47 double calculate_spread_cost(double shares, double price) const {
48 double dollar_amount = shares * price;
49 return dollar_amount * (typical_spread_bps / 10000.0) * 0.5; // Half spread
50 }
51
55 double estimate_price_impact(double shares, double price) const {
56 if (average_daily_volume <= 0)
57 return 0.0;
58
59 double participation_rate = shares / average_daily_volume;
60 double volatility_factor = volatility * price;
61
62 // Square root impact model: impact = coefficient * volatility * sqrt(participation_rate)
63 return price_impact_coefficient * volatility_factor * std::sqrt(participation_rate);
64 }

189 Result<SecurityCapacityResult> analyze_security_capacity(const Symbol& symbol, double target_position_dollars,
190 double current_price) const {
191 auto market_it = market_data_.find(symbol);
192 if (market_it == market_data_.end()) {
193 return Result<SecurityCapacityResult>::error(ErrorCode::InvalidSymbol,
194 "No market data available for symbol: " + symbol);
195 }
196
197 const auto& market_data = market_it->second;
198 SecurityCapacityResult result;
199 result.symbol = symbol;
200
201 if (current_price <= 0) {
202 return Result<SecurityCapacityResult>::error(ErrorCode::InvalidInput, "Current price must be positive");
203 }
204
205 // Calculate maximum position based on different constraints
206 double max_shares_adv = market_data.average_daily_volume * constraints_.max_adv_participation;
207 double max_shares_market_cap = (market_data.market_cap * constraints_.max_market_cap_percent) / current_price;
208
209 // Take the most restrictive constraint
210 result.max_position_shares = std::min(max_shares_adv, max_shares_market_cap);
211 result.max_position_dollars = result.max_position_shares * current_price;
212
213 // Determine binding constraint
214 if (max_shares_adv < max_shares_market_cap) {
215 result.binding_constraint = LiquidityConstraint::ADVMultiple;
216 } else {
217 result.binding_constraint = LiquidityConstraint::MarketCapPercent;
218 }
219
220 // Calculate daily trading limits
221 result.max_daily_trade_shares = market_data.average_daily_volume * constraints_.max_single_day_volume;
222 result.max_daily_trade_dollars = result.max_daily_trade_shares * current_price;
223
224 // Estimate trading timeline
225 double target_shares = std::min(target_position_dollars / current_price, result.max_position_shares);
226 if (result.max_daily_trade_shares > 0) {
227 result.estimated_trading_days =
228 static_cast<size_t>(std::ceil(target_shares / result.max_daily_trade_shares));
229 }
230
231 // Calculate trading costs
232 result.estimated_spread_cost = market_data.calculate_spread_cost(target_shares, current_price);
233 result.estimated_impact_cost = market_data.estimate_price_impact(target_shares, current_price) * target_shares;
234 result.total_trading_cost = result.estimated_spread_cost + result.estimated_impact_cost;
235
236 // Check constraint violations
237 double adv_participation = target_shares / market_data.average_daily_volume;
238 double spread_cost_bps = (result.estimated_spread_cost / (target_shares * current_price)) * 10000.0;
239 double impact_bps = (result.estimated_impact_cost / (target_shares * current_price)) * 10000.0;
240
241 if (constraints_.is_violated(adv_participation, spread_cost_bps, impact_bps, result.estimated_trading_days)) {
242 // Adjust position size to meet constraints
243 result =
244 optimize_position_size(symbol, target_position_dollars, current_price, market_data).value_or(result);
245 }
246
247 return Result<SecurityCapacityResult>::success(result);
248 }

253 Result<PortfolioCapacityResult> analyze_portfolio_capacity(const std::map<Symbol, double>& target_weights,
254 double total_portfolio_value,
255 const std::map<Symbol, double>& current_prices) const {
256 if (target_weights.empty()) {
257 return Result<PortfolioCapacityResult>::error(ErrorCode::InvalidInput, "Target weights cannot be empty");
258 }
259
260 PortfolioCapacityResult portfolio_result;
261 portfolio_result.current_portfolio_size = total_portfolio_value;
262 portfolio_result.total_portfolio_capacity = 0.0;
263 portfolio_result.total_estimated_costs = 0.0;
264
265 std::vector<double> trading_days_vector;
266
267 // Analyze each security
268 for (const auto& [symbol, weight] : target_weights) {
269 auto price_it = current_prices.find(symbol);
270 if (price_it == current_prices.end()) {
271 continue; // Skip securities without price data
272 }
273
274 double target_dollars = total_portfolio_value * weight;
275 double current_price = price_it->second;
276
277 auto security_result = analyze_security_capacity(symbol, target_dollars, current_price);
278 if (security_result.is_ok()) {
279 const auto& result = security_result.value();
280 portfolio_result.security_results[symbol] = result;
281
282 // Aggregate portfolio metrics
283 portfolio_result.total_portfolio_capacity += result.max_position_dollars;
284 portfolio_result.total_estimated_costs += result.total_trading_cost;
285
286 if (result.estimated_trading_days > 0) {
287 trading_days_vector.push_back(static_cast<double>(result.estimated_trading_days));
288 }
289
290 // Track capacity constrained securities
291 if (result.is_capacity_constrained()) {
292 portfolio_result.capacity_constrained_securities.push_back(symbol);
293 }
294 }
295 }
296
297 // Calculate portfolio-level metrics
298 if (portfolio_result.total_portfolio_capacity > 0) {
299 portfolio_result.capacity_utilization =
300 portfolio_result.current_portfolio_size / portfolio_result.total_portfolio_capacity;
301 }
302
303 if (!trading_days_vector.empty()) {
304 portfolio_result.average_trading_days =
305 std::accumulate(trading_days_vector.begin(), trading_days_vector.end(), 0.0) /
306 trading_days_vector.size();
307 }
308
309 return Result<PortfolioCapacityResult>::success(std::move(portfolio_result));
310 }

315 Result<TimeSeries<double>> calculate_capacity_decay(const Symbol& symbol, double initial_capacity,
316 const TimeSeries<double>& volume_series) const {
317 if (volume_series.empty()) {
318 return Result<TimeSeries<double>>::error(ErrorCode::InsufficientData, "Volume series cannot be empty");
319 }
320
321 std::vector<double> capacity_values;
322 capacity_values.reserve(volume_series.size());
323
324 auto market_it = market_data_.find(symbol);
325 if (market_it == market_data_.end()) {
326 return Result<TimeSeries<double>>::error(ErrorCode::InvalidSymbol, "No market data for symbol: " + symbol);
327 }
328
329 const auto& market_data = market_it->second;
330 double current_capacity = initial_capacity;
331
332 for (size_t i = 0; i < volume_series.size(); ++i) {
333 double daily_volume = volume_series[i];
334
335 // Update capacity based on volume changes
336 if (market_data.average_daily_volume > 0) {
337 double volume_ratio = daily_volume / market_data.average_daily_volume;
338 current_capacity *= volume_ratio; // Simplistic decay model
339 }
340
341 capacity_values.push_back(current_capacity);
342 }
343
344 return Result<TimeSeries<double>>::success(
345 TimeSeries<double>{volume_series.timestamps(), capacity_values, symbol + "_capacity"});

351 Result<std::vector<double>> simulate_trading_impact(const Symbol& symbol,
352 const transactions::TransactionSeries& transactions,
353 double initial_capacity) const {
354 if (transactions.empty()) {
355 return Result<std::vector<double>>::error(ErrorCode::InsufficientData,
356 "Transaction series cannot be empty");
357 }
358
359 auto market_it = market_data_.find(symbol);
360 if (market_it == market_data_.end()) {
361 return Result<std::vector<double>>::error(ErrorCode::InvalidSymbol, "No market data for symbol: " + symbol);
362 }
363
364 const auto& market_data = market_it->second;
365 std::vector<double> capacity_impact;
366 capacity_impact.reserve(transactions.size());
367
368 double current_capacity = initial_capacity;
369
370 for (const auto& txn : transactions) {
371 if (txn.symbol() != symbol)
372 continue;
373
374 double trade_size = std::abs(txn.shares());
375 double participation_rate = trade_size / market_data.average_daily_volume;
376
377 // Calculate capacity reduction due to trading
378 double impact_factor = 1.0 - (participation_rate * 0.1); // 10% reduction per 1% participation
379 current_capacity *= std::max(0.1, impact_factor); // Minimum 10% of original capacity
380
381 capacity_impact.push_back(current_capacity);
382 }
383
384 return Result<std::vector<double>>::success(std::move(capacity_impact));
385 }

391 std::optional<SecurityCapacityResult> optimize_position_size(const Symbol& symbol, double target_dollars,
392 double current_price,
393 const MarketMicrostructure& market_data) const {
394 SecurityCapacityResult result;
395 result.symbol = symbol;
396
397 // Binary search for optimal position size
398 double min_dollars = 0.0;
399 double max_dollars = target_dollars;
400 double optimal_dollars = 0.0;
401
402 for (int iteration = 0; iteration < 20; ++iteration) { // Max 20 iterations
403 double test_dollars = (min_dollars + max_dollars) / 2.0;
404 double test_shares = test_dollars / current_price;
405
406 // Check constraints
407 double adv_participation = test_shares / market_data.average_daily_volume;
408 double spread_cost = market_data.calculate_spread_cost(test_shares, current_price);
409 double impact_cost = market_data.estimate_price_impact(test_shares, current_price) * test_shares;
410
411 double spread_cost_bps = (spread_cost / test_dollars) * 10000.0;
412 double impact_bps = (impact_cost / test_dollars) * 10000.0;
413
414 size_t trading_days = static_cast<size_t>(
415 std::ceil(test_shares / (market_data.average_daily_volume * constraints_.max_single_day_volume)));
416
417 if (constraints_.is_violated(adv_participation, spread_cost_bps, impact_bps, trading_days)) {
418 max_dollars = test_dollars;
419 } else {
420 min_dollars = test_dollars;
421 optimal_dollars = test_dollars;
422 }
423
424 if (std::abs(max_dollars - min_dollars) < 1000.0) { // $1000 precision
425 break;
426 }
427 }
428
429 if (optimal_dollars > 0) {
430 result.max_position_dollars = optimal_dollars;
431 result.max_position_shares = optimal_dollars / current_price;
432 result.estimated_spread_cost = market_data.calculate_spread_cost(result.max_position_shares, current_price);
433 result.estimated_impact_cost =
434 market_data.estimate_price_impact(result.max_position_shares, current_price) *
435 result.max_position_shares;
436 result.total_trading_cost = result.estimated_spread_cost + result.estimated_impact_cost;
437 result.binding_constraint = LiquidityConstraint::VolumePercent;
438
439 return result;
440 }
441
442 return std::nullopt;
443 }

449inline MarketMicrostructure create_market_microstructure(const Symbol& symbol, double avg_daily_volume,
450 double market_cap, double current_price,
451 double typical_spread_bps = 10.0, double volatility = 0.20) {
452 MarketMicrostructure data;
453 data.symbol = symbol;
454 data.average_daily_volume = avg_daily_volume;
455 data.average_dollar_volume = avg_daily_volume * current_price;
456 data.market_cap = market_cap;
457 data.typical_spread_bps = typical_spread_bps;
458 data.volatility = volatility;
459
460 // Estimate price impact coefficient based on market cap and volume
461 // Larger, more liquid stocks have lower impact
462 double liquidity_factor = std::log(data.average_dollar_volume / 1000000.0); // Log of ADV in millions
463 data.price_impact_coefficient = 0.1 / std::max(1.0, liquidity_factor);
464
465 return data;
466}

471inline Result<double> calculate_turnover_capacity(const PortfolioCapacityResult& capacity_result,
472 double target_turnover, double portfolio_value) {
473 if (target_turnover <= 0.0) {
474 return Result<double>::error(ErrorCode::InvalidInput, "Target turnover must be positive");
475 }
476
477 double total_trading_capacity = 0.0;
478 for (const auto& [symbol, result] : capacity_result.security_results) {
479 total_trading_capacity += result.max_daily_trade_dollars;
480 }
481
482 if (total_trading_capacity <= 0.0) {
483 return Result<double>::success(0.0);
484 }
485
486 // Annual turnover capacity = daily capacity * trading days per year
487 double annual_trading_capacity = total_trading_capacity * 252.0;
488 double max_supportable_aum = annual_trading_capacity / target_turnover;
489
490 return Result<double>::success(max_supportable_aum);
