41template <typename T>
42Result<StatisticalSummary> calculate_summary(const TimeSeries<T>& series) {
43 if (series.empty()) {
44 return Result<StatisticalSummary>::error(ErrorCode::InvalidInput, "Empty time series");
45 }
46
47 auto values = series.values();
48 std::span<const T> data_span(values);
49
50 StatisticalSummary summary;
51
52 // Basic statistics
53 auto mean_result = stats::mean(data_span);
54 if (!mean_result.has_value()) {
55 return Result<StatisticalSummary>::error(ErrorCode::CalculationError,
56 "Failed to calculate mean: " + mean_result.error().message);
57 }
58 summary.mean = mean_result.value();
59
60 auto variance_result = stats::variance(data_span);
61 if (!variance_result.has_value()) {
62 // For single value, variance and std dev are 0
63 summary.variance = 0.0;
64 summary.std_dev = 0.0;
65 } else {
66 summary.variance = variance_result.value();
67 summary.std_dev = std::sqrt(summary.variance);
68 }
69
70 auto skew_result = stats::skewness(data_span);
71 if (!skew_result.has_value()) {
72 // For single value, skewness is undefined but set to 0
73 summary.skewness = 0.0;
74 } else {
75 summary.skewness = skew_result.value();
76 }
77
78 auto kurt_result = stats::kurtosis(data_span);
79 if (!kurt_result.has_value()) {
80 // For single value, kurtosis is undefined but set to 0
81 summary.kurtosis = 0.0;
82 } else {
83 summary.kurtosis = kurt_result.value();
84 }
85
86 // Min/max
87 auto minmax = std::minmax_element(values.begin(), values.end());
88 summary.min_value = *minmax.first;
89 summary.max_value = *minmax.second;
90
91 // Percentiles
92 std::vector<T> sorted_values = values;
93 std::sort(sorted_values.begin(), sorted_values.end());
94
95 size_t n = sorted_values.size();
96 summary.median = sorted_values[n / 2];
97 summary.q25 = sorted_values[n / 4];
98 summary.q75 = sorted_values[3 * n / 4];
99
100 summary.count = n;
101 summary.missing_count = 0; // Assuming no missing values for now
102
103 return Result<StatisticalSummary>::success(summary);
104}

122template <typename T>
123Result<DistributionAnalysis> analyze_distribution(const TimeSeries<T>& series) {
124 if (series.empty()) {
125 return Error("Empty time series");
126 }
127
128 DistributionAnalysis analysis;
129 auto values = series.values();
130
131 // Calculate Jarque-Bera test for normality
132 auto summary_result = calculate_summary(series);
133 if (!summary_result.has_value()) {
134 return Error("Failed to calculate summary statistics");
135 }
136
137 auto summary = summary_result.value();
138
139 // Jarque-Bera statistic
140 double n = static_cast<double>(values.size());
141 analysis.jarque_bera_statistic =
142 (n / 6.0) * (std::pow(summary.skewness, 2) + std::pow(summary.kurtosis - 3.0, 2) / 4.0);
143
144 // Simple p-value approximation (chi-square with 2 df)
145 analysis.jarque_bera_p_value = 1.0 - std::erf(analysis.jarque_bera_statistic / 2.0);
146 analysis.is_normal = analysis.jarque_bera_p_value > 0.05;
147
148 // Create histogram
149 int n_bins = static_cast<int>(std::sqrt(n));
150 analysis.histogram_bins.resize(n_bins + 1);
151 analysis.histogram_counts.resize(n_bins);
152
153 double bin_width = (summary.max_value - summary.min_value) / n_bins;
154 for (int i = 0; i <= n_bins; ++i) {
155 analysis.histogram_bins[i] = summary.min_value + i * bin_width;
156 }
157
158 // Count values in each bin
159 std::fill(analysis.histogram_counts.begin(), analysis.histogram_counts.end(), 0.0);
160 for (const auto& value : values) {
161 int bin = static_cast<int>((value - summary.min_value) / bin_width);
162 if (bin >= n_bins)
163 bin = n_bins - 1;
164 if (bin >= 0)
165 analysis.histogram_counts[bin]++;
166 }
167
168 return analysis;
169}

186template <typename T1, typename T2>
187Result<CorrelationAnalysis> analyze_correlation(const TimeSeries<T1>& series1, const TimeSeries<T2>& series2,
188 size_t rolling_window = 252) {
189 if (series1.empty() || series2.empty()) {
190 return Error("Empty time series");
191 }
192
193 CorrelationAnalysis analysis;
194 analysis.window_size = rolling_window;
195
196 // Align series by dates
197 auto aligned = align_series(series1, series2);
198 if (aligned.first.empty() || aligned.second.empty()) {
199 return Error("No overlapping dates between series");
200 }
201
202 auto values1 = aligned.first.values();
203 auto values2 = aligned.second.values();
204
205 std::span<const T1> span1(values1);
206 std::span<const T2> span2(values2);
207
208 // Pearson correlation
209 auto corr_result = stats::correlation(span1, span2);
210 if (!corr_result.has_value()) {
211 return Error("Failed to calculate correlation");
212 }
213 analysis.pearson_correlation = corr_result.value();
214 analysis.r_squared = analysis.pearson_correlation * analysis.pearson_correlation;
215
216 // Simplified Spearman and Kendall (would need ranking functions)
217 analysis.spearman_correlation = analysis.pearson_correlation; // Approximation
218 analysis.kendall_tau = analysis.pearson_correlation * 0.8; // Approximation
219
220 // Rolling correlations
221 if (values1.size() >= rolling_window) {
222 for (size_t i = rolling_window; i <= values1.size(); ++i) {
223 std::span<const T1> window1(values1.data() + i - rolling_window, rolling_window);
224 std::span<const T2> window2(values2.data() + i - rolling_window, rolling_window);
225
226 auto window_corr = stats::correlation(window1, window2);
227 if (window_corr.has_value()) {
228 analysis.rolling_correlations.push_back(window_corr.value());
229 } else {
230 analysis.rolling_correlations.push_back(0.0);
231 }
232 }
233 }
234
235 return analysis;
236}

241template <typename T1, typename T2>
242std::pair<TimeSeries<T1>, TimeSeries<T2>> align_series(const TimeSeries<T1>& series1, const TimeSeries<T2>& series2) {
243 TimeSeries<T1> aligned1;
244 TimeSeries<T2> aligned2;
245
246 auto dates1 = series1.timestamps();
247 auto dates2 = series2.timestamps();
248 auto values1 = series1.values();
249 auto values2 = series2.values();
250
251 // Simple intersection approach
252 for (size_t i = 0; i < dates1.size(); ++i) {
253 for (size_t j = 0; j < dates2.size(); ++j) {
254 if (dates1[i] == dates2[j]) {
255 aligned1.push_back(dates1[i], values1[i]);
256 aligned2.push_back(dates2[j], values2[j]);
257 break;
258 }
259 }
260 }
261
262 return {aligned1, aligned2};
263}

286 template <typename T>
287 static Result<double> sharpe_ratio(const TimeSeries<T>& returns, double risk_free_rate = 0.0) {
288 auto mean_result = returns.mean();
289 auto std_result = returns.std();
290 if (mean_result.is_error() || std_result.is_error()) {
291 return Result<double>::error(ErrorCode::CalculationError, "Failed to calculate Sharpe ratio");
292 }
293 // Check for zero standard deviation
294 if (std_result.value() == 0.0) {
295 return Result<double>::error(ErrorCode::CalculationError,
296 "Zero standard deviation in Sharpe ratio calculation");
297 }
298
299 // Annualize returns and volatility (assuming daily data)
300 double annualized_return = mean_result.value() * 252.0;
301 double annualized_std = std_result.value() * std::sqrt(252.0);
302 double excess_return = annualized_return - risk_free_rate;
303 double sharpe = excess_return / annualized_std;
304 return Result<double>::success(sharpe);
305 }

307 // Sortino ratio calculation
308 template <typename T>
309 static Result<double> sortino_ratio(const TimeSeries<T>& returns, double risk_free_rate = 0.0) {
310 auto mean_result = returns.mean();
311 if (mean_result.is_error()) {
312 return Result<double>::error(ErrorCode::CalculationError, "Failed to calculate Sortino ratio");
313 }
314
315 // Convert annual risk-free rate to daily
316 double daily_rf = risk_free_rate / 252.0;
317
318 // Calculate downside deviation
319 auto values = returns.values();
320 double sum_negative_sq = 0.0;
321 int negative_count = 0;
322 for (const auto& ret : values) {
323 if (ret < daily_rf) {
324 double diff = ret - daily_rf;
325 sum_negative_sq += diff * diff;
326 negative_count++;
327 }
328 }
329
330 if (negative_count == 0) {
331 return Result<double>::success(std::numeric_limits<double>::infinity());
332 }
333
334 double downside_dev = std::sqrt(sum_negative_sq / negative_count);
335 // Annualize the downside deviation
336 double annualized_downside_dev = downside_dev * std::sqrt(252.0);
337 // Annualize the returns
338 double annualized_return = mean_result.value() * 252.0;
339 double excess_return = annualized_return - risk_free_rate;
340 double sortino = excess_return / annualized_downside_dev;
341 return Result<double>::success(sortino);
342 }

344 // Calmar ratio calculation
345 template <typename T>
346 static Result<double> calmar_ratio(const TimeSeries<T>& returns) {
347 auto annual_return = returns.mean();
348 if (annual_return.is_error()) {
349 return Result<double>::error(ErrorCode::CalculationError, "Failed to calculate Calmar ratio");
350 }
351
352 // Simple max drawdown calculation
353 auto cumulative_result = returns.cumulative_returns();
354 if (cumulative_result.is_error()) {
355 return cumulative_result.map([](const auto&) { return 0.0; });
356 }
357
358 auto cum_values = cumulative_result.value().values();
359 double max_value = 0.0;
360 double max_drawdown = 0.0;
361
362 for (const auto& val : cum_values) {
363 if (val > max_value) {
364 max_value = val;
365 }
366 double drawdown = (val - max_value) / max_value;
367 if (drawdown < max_drawdown) {
368 max_drawdown = drawdown;
369 }
370 }
371
372 if (max_drawdown == 0.0) {
373 return Result<double>::success(std::numeric_limits<double>::infinity());
374 }
375
376 double calmar = annual_return.value() / std::abs(max_drawdown);
377 return Result<double>::success(calmar);
378 }

395 // Maximum drawdown calculation
396 template <typename T>
397 static Result<SimpleDrawdownInfo> max_drawdown(const TimeSeries<T>& returns) {
398 auto cumulative_result = returns.cumulative_returns();
399 if (cumulative_result.is_error()) {
400 return Result<SimpleDrawdownInfo>::error(cumulative_result.error().code, cumulative_result.error().message);
401 }
402
403 auto cum_values = cumulative_result.value().values();
404 if (cum_values.empty()) {
405 return Result<SimpleDrawdownInfo>::error(ErrorCode::InvalidInput, "Empty cumulative returns");
406 }
407
408 // Start with first value as peak (cumulative returns usually start at 1.0)
409 double max_value = std::max(1.0, cum_values[0]);
410 double max_drawdown = 0.0;
411 int max_duration = 0;
412 int current_duration = 0;
413
414 for (const auto& val : cum_values) {
415 if (val > max_value) {
416 max_value = val;
417 current_duration = 0;
418 } else {
419 current_duration++;
420 if (current_duration > max_duration) {
421 max_duration = current_duration;
422 }
423 }
424
425 // Calculate drawdown as percentage drop from peak
426 double drawdown = 0.0;
427 if (max_value > 0.0) {
428 drawdown = (val - max_value) / max_value;
429 // Ensure drawdown doesn't exceed -100%
430 drawdown = std::max(drawdown, -1.0);
431 } else {
432 // If max_value is 0, cap at -100%
433 drawdown = -1.0;
434 }
435
436 if (drawdown < max_drawdown) {
437 max_drawdown = drawdown;
438 }
439 }
440
441 SimpleDrawdownInfo info;
442 info.max_drawdown = max_drawdown;
443 info.duration_days = max_duration;
444 return Result<SimpleDrawdownInfo>::success(info);
445 }

476 // Alpha and beta calculation
477 template <typename T>
478 static Result<AlphaBetaResult> alpha_beta(const TimeSeries<T>& returns, const TimeSeries<T>& benchmark_returns,
479 double risk_free_rate = 0.0) {
480 if (returns.size() != benchmark_returns.size()) {
481 return Result<AlphaBetaResult>::error(ErrorCode::InvalidInput, "Return series must have same length");
482 }
483
484 auto portfolio_values = returns.values();
485 auto benchmark_values = benchmark_returns.values();
486
487 // Calculate means
488 double portfolio_mean = 0.0;
489 double benchmark_mean = 0.0;
490 for (size_t i = 0; i < portfolio_values.size(); ++i) {
491 portfolio_mean += portfolio_values[i];
492 benchmark_mean += benchmark_values[i];
493 }
494 portfolio_mean /= portfolio_values.size();
495 benchmark_mean /= benchmark_values.size();
496
497 // Calculate beta (covariance / variance)
498 double covariance = 0.0;
499 double benchmark_variance = 0.0;
500
501 for (size_t i = 0; i < portfolio_values.size(); ++i) {
502 double portfolio_diff = portfolio_values[i] - portfolio_mean;
503 double benchmark_diff = benchmark_values[i] - benchmark_mean;
504 covariance += portfolio_diff * benchmark_diff;
505 benchmark_variance += benchmark_diff * benchmark_diff;
506 }
507
508 covariance /= portfolio_values.size();
509 benchmark_variance /= benchmark_values.size();
510
511 double beta = (benchmark_variance != 0.0) ? covariance / benchmark_variance : 0.0;
512 double alpha = portfolio_mean - risk_free_rate - beta * (benchmark_mean - risk_free_rate);
513
514 AlphaBetaResult result;
515 result.alpha = alpha;
516 result.beta = beta;
517 return Result<AlphaBetaResult>::success(result);
518 }

521 template <typename T>
522 static Result<double> information_ratio(const TimeSeries<T>& returns, const TimeSeries<T>& benchmark_returns) {
523 // Information ratio = (portfolio_return - benchmark_return) / tracking_error
524 if (returns.size() != benchmark_returns.size()) {
525 return Result<double>::error(ErrorCode::InvalidInput, "Return series must have same length");
526 }
527
528 // Calculate excess returns (portfolio - benchmark)
529 auto portfolio_values = returns.values();
530 auto benchmark_values = benchmark_returns.values();
531 std::vector<T> excess_returns_values;
532
533 for (size_t i = 0; i < portfolio_values.size(); ++i) {
534 excess_returns_values.push_back(portfolio_values[i] - benchmark_values[i]);
535 }
536
537 // Calculate mean and std of excess returns
538 double sum = 0.0;
539 for (const auto& val : excess_returns_values) {
540 sum += val;
541 }
542 double mean_excess = sum / excess_returns_values.size();
543
544 double variance = 0.0;
545 for (const auto& val : excess_returns_values) {
546 double diff = val - mean_excess;
547 variance += diff * diff;
548 }
549 variance /= excess_returns_values.size();
550 double std_excess = std::sqrt(variance);
551
552 if (std_excess == 0.0) {
553 return Result<double>::error(ErrorCode::CalculationError, "Zero tracking error");
554 }
555
556 // Annualize for information ratio
557 double annualized_excess_return = mean_excess * 252.0;
558 double annualized_tracking_error = std_excess * std::sqrt(252.0);
559
560 return Result<double>::success(annualized_excess_return / annualized_tracking_error);
561 }

563 template <typename T>
564 static Result<double> tracking_error(const TimeSeries<T>& returns, const TimeSeries<T>& benchmark_returns) {
565 // Tracking error is annualized std dev of excess returns
566 if (returns.size() != benchmark_returns.size()) {
567 return Result<double>::error(ErrorCode::InvalidInput, "Return series must have same length");
568 }
569
570 // Calculate excess returns (portfolio - benchmark)
571 auto portfolio_values = returns.values();
572 auto benchmark_values = benchmark_returns.values();
573 std::vector<T> excess_returns_values;
574
575 for (size_t i = 0; i < portfolio_values.size(); ++i) {
576 excess_returns_values.push_back(portfolio_values[i] - benchmark_values[i]);
577 }
578
579 // Calculate std of excess returns
580 double sum = 0.0;
581 for (const auto& val : excess_returns_values) {
582 sum += val;
583 }
584 double mean_excess = sum / excess_returns_values.size();
585
586 double variance = 0.0;
587 for (const auto& val : excess_returns_values) {
588 double diff = val - mean_excess;
589 variance += diff * diff;
590 }
591 variance /= excess_returns_values.size();
592 double std_excess = std::sqrt(variance);
593
594 // Annualize tracking error
595 double annualized_tracking_error = std_excess * std::sqrt(252.0);
596
