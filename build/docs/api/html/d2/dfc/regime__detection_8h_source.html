306 regime = RegimeType::Volatile;

309 // Trend-based classification for normal volatility

311 regime = RegimeType::Bull;
312 confidence = 0.8;

314 regime = RegimeType::Bear;
315 confidence = 0.8;

318 confidence = 0.7;

330 Result<std::vector<std::pair<RegimeType, double>>> predict_regime_transitions(
331 const RegimeDetectionResult& current_state, size_t forecast_horizon = 21) const {
332 if (current_state.regime_sequence.empty()) {
333 return Result<std::vector<std::pair<RegimeType, double>>>::error(ErrorCode::InvalidInput,
334 "Current state cannot be empty");
335 }
336
337 std::vector<std::pair<RegimeType, double>> predictions;
338 predictions.reserve(forecast_horizon);
339
340 // Calculate transition matrix
341 auto transition_matrix = estimate_transition_matrix(current_state.regime_sequence);
342
343 RegimeType current_regime = current_state.current_regime;
344 double confidence = current_state.current_regime_confidence;
345
346 // Forward simulation
347 for (size_t step = 0; step < forecast_horizon; ++step) {
348 auto next_regime_result = predict_next_regime(current_regime, transition_matrix);
349 if (next_regime_result.is_ok()) {
350 auto [next_regime, transition_prob] = next_regime_result.value();
351 confidence *= transition_prob; // Uncertainty compounds
352 predictions.emplace_back(next_regime, confidence);
353 current_regime = next_regime;
354 } else {
355 break;
356 }
357 }
358
359 return Result<std::vector<std::pair<RegimeType, double>>>::success(std::move(predictions));

382 Result<RegimeDetectionResult> markov_switching_detection(const ReturnSeries& returns, int num_regimes = 2,
383 int max_iterations = 1000, uint32_t seed = 42) const {
384 if (returns.empty()) {
385 return Result<RegimeDetectionResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
386 }
387
388 if (num_regimes < 2 || num_regimes > 5) {
389 return Result<RegimeDetectionResult>::error(ErrorCode::InvalidInput,
390 "Number of regimes must be between 2 and 5");
391 }
392
393 // Initialize random number generator
394 std::mt19937 gen(seed);
395 std::uniform_real_distribution<> uniform(0.0, 1.0);
396 std::normal_distribution<> normal(0.0, 1.0);
397
398 const auto& values = returns.values();
399 size_t n = values.size();
400
401 // Initialize regime parameters
402 std::vector<double> means(num_regimes);
403 std::vector<double> variances(num_regimes);
404 std::vector<std::vector<double>> transition_matrix(num_regimes, std::vector<double>(num_regimes));
405
406 // Initialize parameters with k-means-like approach
407 auto mean_result = stats::mean(std::span<const double>{values});
408 auto var_result = stats::variance(std::span<const double>{values});
409
410 double overall_mean = mean_result.value_or(0.0);
411 double overall_var = var_result.value_or(0.01);
412
413 for (int k = 0; k < num_regimes; ++k) {
414 means[k] = overall_mean + normal(gen) * std::sqrt(overall_var) * 0.5;
415 variances[k] = overall_var * (0.5 + uniform(gen));
416
417 // Initialize transition probabilities
418 for (int j = 0; j < num_regimes; ++j) {
419 transition_matrix[k][j] = uniform(gen);
420 }
421
422 // Normalize transition probabilities
423 double sum = std::accumulate(transition_matrix[k].begin(), transition_matrix[k].end(), 0.0);
424 if (sum > 0) {
425 for (int j = 0; j < num_regimes; ++j) {
426 transition_matrix[k][j] /= sum;
427 }
428 }
429 }
430
431 // EM algorithm
432 std::vector<std::vector<double>> state_probabilities(n, std::vector<double>(num_regimes));
433 double prev_likelihood = -std::numeric_limits<double>::infinity();
434
435 for (int iter = 0; iter < max_iterations; ++iter) {
436 // E-step: Calculate state probabilities using forward-backward algorithm
437 forward_backward_algorithm(values, means, variances, transition_matrix, state_probabilities);
438
439 // M-step: Update parameters
440 update_markov_parameters(values, state_probabilities, means, variances, transition_matrix);
441
442 // Check convergence
443 double likelihood = calculate_likelihood(values, means, variances, transition_matrix);
444 if (std::abs(likelihood - prev_likelihood) < 1e-6) {
445 break;
446 }
447 prev_likelihood = likelihood;
448 }
449
450 // Extract most likely regime sequence
451 RegimeDetectionResult result;
452 result.dates = returns.timestamps();
453 result.regime_sequence.reserve(n);
454 result.regime_probabilities.reserve(n);
455
456 for (size_t i = 0; i < n; ++i) {
457 auto max_it = std::max_element(state_probabilities[i].begin(), state_probabilities[i].end());
458 int regime_idx = std::distance(state_probabilities[i].begin(), max_it);
459
460 // Map regime index to RegimeType
461 RegimeType regime = map_regime_index_to_type(regime_idx, means);
462 result.regime_sequence.push_back(regime);
463 result.regime_probabilities.push_back(*max_it);
464 }
465
466 // Set current regime info
467 if (!result.regime_sequence.empty()) {
468 result.current_regime = result.regime_sequence.back();
469 result.current_regime_confidence = result.regime_probabilities.back();
470 result.current_regime_duration = calculate_current_regime_duration(result.regime_sequence);
471 }
472
473 // Calculate characteristics and transitions
474 result.regime_characteristics = calculate_regime_characteristics(returns, result.regime_sequence);
475 result.transitions = calculate_regime_transitions(result.regime_sequence, result.dates);
476
477 return Result<RegimeDetectionResult>::success(std::move(result));
478 }

491 Result<RegimeDetectionResult> structural_break_detection(const ReturnSeries& returns,
492 double significance_level = 0.05) const {
493 if (returns.empty()) {
494 return Result<RegimeDetectionResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
495 }
496
497 const auto& values = returns.values();
498 size_t n = values.size();
499
500 if (n < 20) {
501 return Result<RegimeDetectionResult>::error(ErrorCode::InsufficientData,
502 "Need at least 20 observations for structural break detection");
503 }
504
505 std::vector<size_t> break_points;
506
507 // CUSUM test for structural breaks
508 auto mean_result = stats::mean(std::span<const double>{values});
509 double overall_mean = mean_result.value_or(0.0);
510
511 std::vector<double> cusum(n);
512 cusum[0] = values[0] - overall_mean;
513
514 for (size_t i = 1; i < n; ++i) {
515 cusum[i] = cusum[i - 1] + (values[i] - overall_mean);
516 }
517
518 // Find potential break points
519 auto var_result = stats::variance(std::span<const double>{values});
520 double overall_var = var_result.value_or(0.01);
521 double threshold = std::sqrt(overall_var * n) * 2.0; // Simplified threshold
522
523 for (size_t i = 10; i < n - 10; ++i) {
524 if (std::abs(cusum[i]) > threshold) {
525 break_points.push_back(i);
526 }
527 }
528
529 // Create regime sequence based on break points
530 RegimeDetectionResult result;
531 result.dates = returns.timestamps();
532 result.regime_sequence.resize(n);
533 result.regime_probabilities.resize(n, 0.8); // High confidence for structural breaks
534
535 // Assign regimes based on break points
536 RegimeType current_regime = RegimeType::Bull;
537 size_t last_break = 0;
538
539 for (size_t break_point : break_points) {
540 // Determine regime type based on mean return in segment
541 auto segment_mean =
542 stats::mean(std::span<const double>{values.data() + last_break, break_point - last_break});
543
544 if (segment_mean.is_ok()) {
545 current_regime = (segment_mean.value() > 0) ? RegimeType::Bull : RegimeType::Bear;
546 }
547
548 for (size_t i = last_break; i < break_point && i < n; ++i) {
549 result.regime_sequence[i] = current_regime;
550 }
551
552 last_break = break_point;
553 // Alternate regime for simplicity
554 current_regime = (current_regime == RegimeType::Bull) ? RegimeType::Bear : RegimeType::Bull;
555 }
556
557 // Fill remaining observations
558 for (size_t i = last_break; i < n; ++i) {
559 result.regime_sequence[i] = current_regime;
560 }
561
562 // Set current regime info
563 if (!result.regime_sequence.empty()) {
564 result.current_regime = result.regime_sequence.back();
565 result.current_regime_confidence = 0.8;
566 result.current_regime_duration = calculate_current_regime_duration(result.regime_sequence);
567 }
568
569 // Calculate characteristics and transitions
570 result.regime_characteristics = calculate_regime_characteristics(returns, result.regime_sequence);
571 result.transitions = calculate_regime_transitions(result.regime_sequence, result.dates);
572
573 return Result<RegimeDetectionResult>::success(std::move(result));
574 }

580 if (returns.empty()) {
581 return Result<RegimeDetectionResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
582 }
583
584 const auto& values = returns.values();
585 size_t n = values.size();
586
587 // Calculate rolling volatility
588 std::vector<double> rolling_volatility;
589 rolling_volatility.reserve(n);
590
591 for (size_t i = 0; i < n; ++i) {
592 size_t start = (i >= lookback_window_) ? i - lookback_window_ + 1 : 0;
593 std::vector<double> window(values.begin() + start, values.begin() + i + 1);
594
595 auto vol_result = stats::standard_deviation(std::span<const double>{window});
596 rolling_volatility.push_back(vol_result.value_or(0.01));
597 }
598
599 // Apply threshold-based regime classification
600 auto median_vol_result = stats::median(std::span<const double>{rolling_volatility});
601 double median_vol = median_vol_result.value_or(0.01);
602
603 double high_vol_threshold = median_vol * 1.5;
604 double low_vol_threshold = median_vol * 0.5;
605
606 RegimeDetectionResult result;
607 result.dates = returns.timestamps();
608 result.regime_sequence.reserve(n);
609 result.regime_probabilities.reserve(n);
610
611 for (size_t i = 0; i < n; ++i) {
612 double vol = rolling_volatility[i];
613 RegimeType regime;
614 double confidence;
615
616 if (vol > high_vol_threshold * 2.0) {
617 regime = RegimeType::Crisis;
618 confidence = 0.9;
619 } else if (vol > high_vol_threshold) {
620 regime = RegimeType::Volatile;
621 confidence = 0.8;
622 } else if (vol < low_vol_threshold) {
623 regime = RegimeType::Stable;
624 confidence = 0.8;
625 } else {
626 // Determine based on return sign
627 regime = (values[i] > 0) ? RegimeType::Bull : RegimeType::Bear;
628 confidence = 0.6;
629 }
630
631 result.regime_sequence.push_back(regime);
632 result.regime_probabilities.push_back(confidence);

637 result.current_regime = result.regime_sequence.back();

824
825 const auto& values = returns.values();

827 // Group returns by regime
828 for (size_t i = 0; i < std::min(sequence.size(), values.size()); ++i) {
829 regime_returns[sequence[i]].push_back(values[i]);
830 }
831
832 // Calculate durations
833 if (!sequence.empty()) {
834 RegimeType current_regime = sequence[0];
835 size_t current_duration = 1;
836
837 for (size_t i = 1; i < sequence.size(); ++i) {

844 }
845 }
846 regime_durations[current_regime].push_back(current_duration);

851 for (const auto& [regime, rets] : regime_returns) {

857 auto vol_result = stats::standard_deviation(std::span<const double>{rets});

862
863 // Calculate average duration
864 if (!regime_durations[regime].empty()) {
865 double avg_duration =
866 std::accumulate(regime_durations[regime].begin(), regime_durations[regime].end(), 0.0) /
867 regime_durations[regime].size();
868 char_data.persistence = avg_duration;
869 }
870
871 char_data.description = generate_regime_description(char_data);
872 }

874 characteristics.push_back(char_data);
875 }

883 std::vector<RegimeTransition> calculate_regime_transitions(const std::vector<RegimeType>& sequence,
884 const std::vector<DateTime>& dates) const {

886 std::map<RegimeType, int> regime_counts;
887
888 // Count transitions
889 for (size_t i = 1; i < sequence.size(); ++i) {
890 RegimeType from = sequence[i - 1];
891 RegimeType to = sequence[i];
892
893 transition_counts[{from, to}]++;
894 regime_counts[from]++;
895 }
896
897 std::vector<RegimeTransition> transitions;
898
899 for (const auto& [transition, count] : transition_counts) {
900 RegimeTransition trans;
901 trans.from_regime = transition.first;
902 trans.to_regime = transition.second;
903
904 if (regime_counts[transition.first] > 0) {
905 trans.probability = static_cast<double>(count) / regime_counts[transition.first];
906 trans.expected_duration = 1.0 / trans.probability;
907 }
908
909 transitions.push_back(trans);
910 }
911
912 return transitions;

958 RegimeType current_regime, const std::map<std::pair<RegimeType, RegimeType>, double>& transition_matrix) const {

960
961 for (const auto& [transition, prob] : transition_matrix) {

974 [](const auto& a, const auto& b) { return a.second < b.second; });

982 void forward_backward_algorithm(const std::vector<double>& observations, const std::vector<double>& means,
983 const std::vector<double>& variances,
984 const std::vector<std::vector<double>>& transition_matrix,
985 std::vector<std::vector<double>>& state_probabilities) const {
986 size_t n = observations.size();
987 size_t num_states = means.size();
988
989 // Forward pass
990 std::vector<std::vector<double>> forward(n, std::vector<double>(num_states));
991
992 // Initialize forward probabilities
993 for (size_t k = 0; k < num_states; ++k) {
994 forward[0][k] = gaussian_pdf(observations[0], means[k], variances[k]) / num_states;
995 }
996
997 // Forward recursion
998 for (size_t t = 1; t < n; ++t) {
999 for (size_t k = 0; k < num_states; ++k) {
1000 forward[t][k] = 0.0;
1001 for (size_t j = 0; j < num_states; ++j) {
1002 forward[t][k] += forward[t - 1][j] * transition_matrix[j][k];
1003 }
1004 forward[t][k] *= gaussian_pdf(observations[t], means[k], variances[k]);
1005 }
1006
1007 // Normalize to prevent underflow
1008 double sum = std::accumulate(forward[t].begin(), forward[t].end(), 0.0);
1009 if (sum > 0) {
1010 for (size_t k = 0; k < num_states; ++k) {
1011 forward[t][k] /= sum;
1012 }
1013 }
1014 }
1015
1016 // Backward pass
1017 std::vector<std::vector<double>> backward(n, std::vector<double>(num_states, 1.0));
1018
1019 // Backward recursion
1020 for (int t = static_cast<int>(n) - 2; t >= 0; --t) {
1021 for (size_t k = 0; k < num_states; ++k) {
1022 backward[t][k] = 0.0;
1023 for (size_t j = 0; j < num_states; ++j) {
1024 backward[t][k] += transition_matrix[k][j] *
1025 gaussian_pdf(observations[t + 1], means[j], variances[j]) * backward[t + 1][j];
1026 }
1027 }
1028
1029 // Normalize
1030 double sum = std::accumulate(backward[t].begin(), backward[t].end(), 0.0);
1031 if (sum > 0) {
1032 for (size_t k = 0; k < num_states; ++k) {
1033 backward[t][k] /= sum;
1034 }
1035 }
1036 }
1037
1038 // Combine forward and backward probabilities
1039 for (size_t t = 0; t < n; ++t) {
1040 double sum = 0.0;
1041 for (size_t k = 0; k < num_states; ++k) {
1042 state_probabilities[t][k] = forward[t][k] * backward[t][k];
1043 sum += state_probabilities[t][k];
1044 }
1045
1046 // Normalize
1047 if (sum > 0) {
1048 for (size_t k = 0; k < num_states; ++k) {
1049 state_probabilities[t][k] /= sum;
1050 }
1051 }
1052 }
1053 }

1058 void update_markov_parameters(const std::vector<double>& observations,
1059 const std::vector<std::vector<double>>& state_probabilities,
1060 std::vector<double>& means, std::vector<double>& variances,
1061 std::vector<std::vector<double>>& transition_matrix) const {
1062 size_t n = observations.size();
1063 size_t num_states = means.size();
1064
1065 // Update means and variances
1066 for (size_t k = 0; k < num_states; ++k) {
1067 double sum_prob = 0.0;
1068 double sum_obs = 0.0;
1069
1070 for (size_t t = 0; t < n; ++t) {
1071 sum_prob += state_probabilities[t][k];
1072 sum_obs += state_probabilities[t][k] * observations[t];
1073 }
1074
1075 if (sum_prob > 1e-6) {
1076 means[k] = sum_obs / sum_prob;
1077
1078 // Update variance
1079 double sum_var = 0.0;
1080 for (size_t t = 0; t < n; ++t) {
1081 double diff = observations[t] - means[k];
1082 sum_var += state_probabilities[t][k] * diff * diff;
1083 }
1084 variances[k] = std::max(1e-6, sum_var / sum_prob);
1085 }
1086 }
1087
1088 // Update transition matrix
1089 for (size_t i = 0; i < num_states; ++i) {
1090 double sum_from_i = 0.0;
1091 for (size_t j = 0; j < num_states; ++j) {
1092 double sum_transitions = 0.0;
1093
1094 for (size_t t = 0; t < n - 1; ++t) {
1095 sum_transitions += state_probabilities[t][i] * state_probabilities[t + 1][j];
1096 }
1097
1098 transition_matrix[i][j] = sum_transitions;
1099 sum_from_i += sum_transitions;
1100 }
1101
1102 // Normalize
1103 if (sum_from_i > 1e-6) {
1104 for (size_t j = 0; j < num_states; ++j) {
1105 transition_matrix[i][j] /= sum_from_i;
1106 }
1107 }
1108 }
1109 }

1114 double calculate_likelihood(const std::vector<double>& observations, const std::vector<double>& means,
1115 const std::vector<double>& variances,
1116 const std::vector<std::vector<double>>& transition_matrix) const {
1117 double log_likelihood = 0.0;
1118 size_t n = observations.size();
1119 size_t num_states = means.size();
1120
1121 std::vector<double> current_probs(num_states, 1.0 / num_states);
1122
1123 for (size_t t = 0; t < n; ++t) {
1124 std::vector<double> next_probs(num_states, 0.0);
1125 double observation_prob = 0.0;
1126
1127 for (size_t k = 0; k < num_states; ++k) {
1128 double emission_prob = gaussian_pdf(observations[t], means[k], variances[k]);
1129 observation_prob += current_probs[k] * emission_prob;
1130
1131 // Update state probabilities
1132 for (size_t j = 0; j < num_states; ++j) {
1133 next_probs[j] += current_probs[k] * transition_matrix[k][j] * emission_prob;
1134 }
1135 }
1136
1137 log_likelihood += std::log(std::max(1e-10, observation_prob));
1138
1139 // Normalize next probabilities
1140 double sum = std::accumulate(next_probs.begin(), next_probs.end(), 0.0);
1141 if (sum > 1e-10) {
1142 for (size_t k = 0; k < num_states; ++k) {
1143 next_probs[k] /= sum;
1144 }
1145 }
1146
1147 current_probs = next_probs;
1148 }
1149
1150 return log_likelihood;
1151 }
