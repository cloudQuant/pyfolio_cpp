200 std::vector<DateTime> selected_index;
201
202 for (size_type i = 0; i < size(); ++i) {
203 if (index_[i] >= start && index_[i] <= end) {
204 selected_indices.push_back(i);
205 selected_index.push_back(index_[i]);

209 if (selected_indices.empty()) {
210 return Result<DataFrame>::error(ErrorCode::MissingData, "No data found in specified date range");
211 }

214 DataFrame result{selected_index};
215

217 std::visit(

219 using T = typename std::decay_t<decltype(column_data)>::value_type;
220 std::vector<T> selected_data;

222
223 for (size_type idx : selected_indices) {
224 selected_data.push_back(column_data[idx]);
225 }
226
227 result.add_column(column_name, selected_data);
228 },
229 columns_.at(column_name));
230 }

354 Result<DataFrame> describe() const {
355 std::vector<std::string> stats_names = {"count", "mean", "std", "min", "max"};
356 std::vector<DateTime> stats_index;
357 for (const auto& name : stats_names) {
358 stats_index.push_back(DateTime::parse(name, "%s").value_or(DateTime::now()));
359 }
360
361 DataFrame stats{stats_index};
362
363 for (const auto& column_name : column_names_) {
364 std::visit(
365 [&](const auto& column_data) {
366 using T = typename std::decay_t<decltype(column_data)>::value_type;
367
368 if constexpr (std::is_arithmetic_v<T>) {
369 std::vector<double> column_stats;
370
371 // Count
372 column_stats.push_back(static_cast<double>(column_data.size()));
373
374 // Mean
375 double sum = std::accumulate(column_data.begin(), column_data.end(), 0.0);
376 double mean = sum / column_data.size();
377 column_stats.push_back(mean);
378
379 // Standard deviation
380 double sq_sum =
381 std::accumulate(column_data.begin(), column_data.end(), 0.0,
382 [mean](double acc, T val) { return acc + (val - mean) * (val - mean); });
383 double std_dev = std::sqrt(sq_sum / column_data.size());
384 column_stats.push_back(std_dev);
385
386 // Min/Max
387 auto [min_it, max_it] = std::minmax_element(column_data.begin(), column_data.end());
388 column_stats.push_back(static_cast<double>(*min_it));
389 column_stats.push_back(static_cast<double>(*max_it));
390
391 stats.add_column(column_name, column_stats);
392 }
393 },
394 columns_.at(column_name));
395 }
396
397 return Result<DataFrame>::success(std::move(stats));
398 }

458 DateTime get_period_start(const DateTime& timestamp, Frequency freq) const {
459 // Simplified period calculation (same as TimeSeries)
460 auto date = timestamp.to_date();
461
462 switch (freq) {
463 case Frequency::Monthly:
464 return DateTime{date.year() / date.month() / std::chrono::day{1}};
465 case Frequency::Quarterly: {
466 auto month = date.month();
467 auto quarter_start_month = ((static_cast<unsigned>(month) - 1) / 3) * 3 + 1;
468 return DateTime{date.year() / std::chrono::month{quarter_start_month} / std::chrono::day{1}};
469 }
470 case Frequency::Yearly:
471 return DateTime{date.year() / std::chrono::January / std::chrono::day{1}};
472 default:
473 return timestamp;
474 }
