274inline Result<void> save_returns_to_csv(const TimeSeries<Return>& returns, const std::string& file_path,
275 const CSVConfig& config) {
276 std::ofstream file(file_path);
277 if (!file.is_open()) {
278 return Result<void>::error(ErrorCode::FileNotFound, "Cannot open file for writing: " + file_path);
279 }
280
281 // Write header if configured
282 if (config.has_header) {
283 file << config.date_column << config.delimiter << config.return_column << "\n";
284 }
285
286 // Write data
287 auto timestamps = returns.timestamps();
288 auto values = returns.values();
289
290 for (size_t i = 0; i < timestamps.size(); ++i) {
291 // Format date as YYYY-MM-DD
292 file << timestamps[i].year() << "-" << std::setfill('0') << std::setw(2) << timestamps[i].month() << "-"
293 << std::setfill('0') << std::setw(2) << timestamps[i].day() << config.delimiter << values[i] << "\n";
294 }
295
296 file.close();
297 return Result<void>::success();
298}

360inline Result<TimeSeries<Return>> load_returns_from_csv(const std::string& file_path, const CSVConfig& config) {
361 std::ifstream file(file_path);
362 if (!file.is_open()) {
363 return Result<TimeSeries<Return>>::error(ErrorCode::FileNotFound, "Cannot open file for reading: " + file_path);
364 }
365
366 TimeSeries<Return> returns;
367 std::string line;
368 bool first_line = true;
369
370 while (std::getline(file, line)) {
371 if (first_line && config.has_header) {
372 first_line = false;
373 continue;
374 }
375
376 std::stringstream ss(line);
377 std::string date_str, return_str;
378
379 if (std::getline(ss, date_str, config.delimiter) && std::getline(ss, return_str, config.delimiter)) {
380 try {
381 // Simple date parsing (assumes YYYY-MM-DD format)
382 std::stringstream date_ss(date_str);
383 std::string year_str, month_str, day_str;
384
385 if (std::getline(date_ss, year_str, '-') && std::getline(date_ss, month_str, '-') &&
386 std::getline(date_ss, day_str)) {
387 int year = std::stoi(year_str);
388 int month = std::stoi(month_str);
389 int day = std::stoi(day_str);
390 double return_val = std::stod(return_str);
391
392 returns.push_back(DateTime(year, month, day), return_val);
393 }
394 } catch (const std::exception&) {
395 // Skip invalid lines
396 continue;
397 }
398 }
399 }
400
401 file.close();
402 return Result<TimeSeries<Return>>::success(std::move(returns));
403}

511inline Result<std::vector<std::vector<std::string>>> CSVParser::parse_string(const std::string& content) {
512 std::vector<std::vector<std::string>> rows;
513 std::istringstream stream(content);
514 std::string line;
515
516 while (std::getline(stream, line)) {
517 if (config_.skip_empty_lines && line.empty()) {
518 continue;
519 }
520
521 auto fields = split_csv_line(line);
522 rows.push_back(std::move(fields));
523 }
524
525 return Result<std::vector<std::vector<std::string>>>::success(std::move(rows));
526}

539inline std::vector<std::string> CSVParser::split_csv_line(const std::string& line) {
540 std::vector<std::string> fields;
541 std::string field;
542 bool in_quotes = false;
543
544 for (size_t i = 0; i < line.length(); ++i) {
545 char c = line[i];
546
547 if (c == config_.quote_char && !in_quotes) {
548 in_quotes = true;
549 } else if (c == config_.quote_char && in_quotes) {
550 in_quotes = false;
551 } else if (c == config_.delimiter && !in_quotes) {
552 fields.push_back(trim(field));
553 field.clear();
554 } else {
555 field += c;
556 }
557 }
558
559 if (!field.empty() || !fields.empty()) {
560 fields.push_back(trim(field));
561 }
562
563 return fields;
564}

566inline std::string CSVParser::trim(const std::string& str) {
567 size_t start = str.find_first_not_of(" \t\r\n");
568 if (start == std::string::npos)
569 return "";
570
571 size_t end = str.find_last_not_of(" \t\r\n");
572 return str.substr(start, end - start + 1);
573}
