333inline json PlotlyEngine::generate_plotly_config(const InteractivePlotConfig& config) {
334 json plotly_config = {
335 { "displayModeBar", true},
336 { "displaylogo", false},
337 {"modeBarButtonsToRemove", json::array()},
338 { "responsive", config.responsive}
339 };
340
341 if (!config.enable_zoom) {
342 plotly_config["modeBarButtonsToRemove"].push_back("zoom2d");
343 plotly_config["modeBarButtonsToRemove"].push_back("zoomIn2d");
344 plotly_config["modeBarButtonsToRemove"].push_back("zoomOut2d");
345 }
346
347 if (!config.enable_pan) {
348 plotly_config["modeBarButtonsToRemove"].push_back("pan2d");
349 }
350
351 // Merge custom config
352 if (!config.custom_config.empty()) {
353 plotly_config.update(config.custom_config);
354 }
355
356 return plotly_config;
357}

464inline Result<std::string> PlotlyEngine::create_time_series_chart(const std::vector<TimeSeries<double>>& series,
465 const std::vector<std::string>& labels,
466 const InteractivePlotConfig& config) {
467 if (series.empty()) {
468 return Result<std::string>::error(ErrorCode::InvalidInput, "No data series provided");
469 }
470
471 if (labels.size() != series.size()) {
472 return Result<std::string>::error(ErrorCode::InvalidInput, "Labels count must match series count");
473 }
474
475 json data = json::array();
476 std::vector<std::string> colors = config.colors;
477 if (colors.empty()) {
478 colors = {"#1f77b4", "#ff7f0e", "#2ca02c", "#d62728", "#9467bd", "#8c564b", "#e377c2"};
479 }
480
481 for (size_t i = 0; i < series.size(); ++i) {
482 const auto& ts = series[i];
483
484 json trace = {
485 {"type", "scatter"},
486 {"mode", "lines"},
487 {"name", labels[i]},
488 { "x", json::array()},
489 { "y", json::array()},
490 {"line", {{"color", colors[i % colors.size()]}}}
491 };
492
493 const auto& timestamps = ts.timestamps();
494 const auto& values = ts.values();
495
496 for (size_t j = 0; j < timestamps.size(); ++j) {
497 trace["x"].push_back(datetime_to_plotly_string(timestamps[j]));
498 trace["y"].push_back(values[j]);
499 }
500
501 if (config.enable_hover) {
502 trace["hovertemplate"] = labels[i] + ": %{y:.4f}<br>%{x}<extra></extra>";
503 }
504
505 data.push_back(trace);
506 }
507
508 auto layout = generate_layout(config, config.title);
509 layout["xaxis"]["title"] = config.xlabel;
510 layout["yaxis"]["title"] = config.ylabel;
511
512 auto plotly_config = generate_plotly_config(config);
513
514 return Result<std::string>::success(generate_html_wrapper(data, layout, plotly_config));
515}

517inline Result<std::string> PlotlyEngine::create_candlestick_chart(const OHLCData& ohlc_data,
518 const InteractivePlotConfig& config) {
519 if (ohlc_data.timestamps.empty()) {
520 return Result<std::string>::error(ErrorCode::InvalidInput, "No OHLC data provided");
521 }
522
523 json data = json::array();
524
525 // Candlestick trace
526 json candlestick = {
527 { "type", "candlestick"},
528 { "name", ohlc_data.name.empty() ? "Price" : ohlc_data.name},
529 { "x", json::array()},
530 { "open", json::array()},
531 { "high", json::array()},
532 { "low", json::array()},
533 { "close", json::array()},
534 {"increasing", {{"line", {{"color", "#26a69a"}}}}},
535 {"decreasing", {{"line", {{"color", "#ef5350"}}}}},
536 { "xaxis", "x"},
537 { "yaxis", "y"}
538 };
539
540 for (size_t i = 0; i < ohlc_data.timestamps.size(); ++i) {
541 candlestick["x"].push_back(datetime_to_plotly_string(ohlc_data.timestamps[i]));
542 candlestick["open"].push_back(ohlc_data.open[i]);
543 candlestick["high"].push_back(ohlc_data.high[i]);
544 candlestick["low"].push_back(ohlc_data.low[i]);
545 candlestick["close"].push_back(ohlc_data.close[i]);
546 }
547
548 data.push_back(candlestick);
549
550 // Volume trace if available
551 if (!ohlc_data.volume.empty()) {
552 json volume_trace = {
553 { "type", "bar"},
554 { "name", "Volume"},
555 { "x", json::array()},
556 { "y", json::array()},
557 {"marker", {{"color", "rgba(158,202,225,0.5)"}}},
558 { "xaxis", "x"},
559 { "yaxis", "y2"}
560 };
561
562 for (size_t i = 0; i < ohlc_data.timestamps.size(); ++i) {
563 volume_trace["x"].push_back(datetime_to_plotly_string(ohlc_data.timestamps[i]));
564 volume_trace["y"].push_back(ohlc_data.volume[i]);
565 }
566
567 data.push_back(volume_trace);
568 }
569
570 auto layout = generate_layout(config, config.title);
571
572 // Dual y-axis for volume
573 if (!ohlc_data.volume.empty()) {
574 layout["yaxis2"] = {
575 { "title", "Volume"},
576 {"overlaying", "y"},
577 { "side", "right"},
578 { "showgrid", false}
579 };
580 }
581
582 layout["xaxis"]["rangeslider"] = {
583 {"visible", false}
584 }; // Disable rangeslider for candlestick
585
586 auto plotly_config = generate_plotly_config(config);
587
588 return Result<std::string>::success(generate_html_wrapper(data, layout, plotly_config));
589}

591inline Result<std::string> PlotlyEngine::create_correlation_heatmap(
592 const std::vector<std::vector<double>>& correlation_matrix, const std::vector<std::string>& labels,
593 const InteractivePlotConfig& config) {
594 if (correlation_matrix.empty() || labels.empty()) {
595 return Result<std::string>::error(ErrorCode::InvalidInput, "Empty correlation matrix or labels");
596 }
597
598 json data = json::array();
599
600 json heatmap = {
601 { "type", "heatmap"},
602 { "z", correlation_matrix},
603 { "x", labels},
604 { "y", labels},
605 { "colorscale", "RdBu"},
606 { "zmid", 0},
607 { "colorbar", {{"title", "Correlation"}}},
608 {"hovertemplate", "%{x} vs %{y}<br>Correlation: %{z:.3f}<extra></extra>"}
609 };
610
611 data.push_back(heatmap);
612
613 auto layout = generate_layout(config, config.title.empty() ? "Correlation Matrix" : config.title);
614 layout["xaxis"]["title"] = "";
615 layout["yaxis"]["title"] = "";
616 layout["width"] = 600;
617 layout["height"] = 600;
618
619 auto plotly_config = generate_plotly_config(config);
620
621 return Result<std::string>::success(generate_html_wrapper(data, layout, plotly_config));
622}
