518inline Result<double> calculate_average_price(const TransactionSeries& transactions, const Symbol& symbol) {
519 auto symbol_txns_result = transactions.filter_by_symbol(symbol);
520
521 if (symbol_txns_result.is_error()) {
522 return Result<double>::error(symbol_txns_result.error().code, symbol_txns_result.error().message);

524

540 }

548inline Result<double> calculate_turnover(const TransactionSeries& transactions, double portfolio_value,
549 const DateTime& start, const DateTime& end) {
550 if (portfolio_value <= 0) {
551 return Result<double>::error(ErrorCode::InvalidInput, "Portfolio value must be positive");
552 }
553
554 auto period_txns_result = transactions.filter_by_date_range(start, end);
555
556 if (period_txns_result.is_error()) {
557 return Result<double>::error(period_txns_result.error().code, period_txns_result.error().message);
558 }
559
560 auto period_txns = period_txns_result.value();
561 if (period_txns.empty()) {
562 return Result<double>::success(0.0);
563 }
564
565 // Calculate half of total traded value (to avoid double counting)
566 double traded_value = period_txns.total_value() / 2.0;
567
568 // Annualize based on period length
569 int days = start.business_days_until(end);
570 if (days <= 0) {
571 return Result<double>::error(ErrorCode::InvalidDateRange, "Invalid date range for turnover calculation");
572 }
573
574 double annualization_factor = constants::TRADING_DAYS_PER_YEAR / days;
575 double annualized_turnover = (traded_value / portfolio_value) * annualization_factor;
576
577 return Result<double>::success(annualized_turnover);
578}
