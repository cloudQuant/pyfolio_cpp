47constexpr double power(double base, int exp) noexcept {
48 if (exp == 0)
49 return 1.0;
50 if (exp == 1)
51 return base;
52 if (exp < 0)
53 return 1.0 / power(base, -exp);
54
55 double result = 1.0;
56 while (exp > 0) {
57 if (exp & 1) {
58 result *= base;
59 }
60 base *= base;
61 exp >>= 1;
62 }
63 return result;
64}

124template <Numeric T>
125Result<double> variance(std::span<const T> data, bool sample = true) {
126 if (data.size() < (sample ? 2 : 1)) {
127 return Result<double>::error(ErrorCode::InsufficientData, "Insufficient data for variance calculation");
128 }
129
130 auto mean_result = mean(data);
131 if (mean_result.is_error()) {
132 return mean_result;
133 }
134
135 double data_mean = mean_result.value();
136 double sum_sq_diff = 0.0;
137
138 for (const auto& value : data) {
139 double diff = static_cast<double>(value) - data_mean;
140 sum_sq_diff += diff * diff;
141 }
142
143 size_t denominator = sample ? data.size() - 1 : data.size();
144 return Result<double>::success(sum_sq_diff / denominator);
145}

163template <Numeric T>
164Result<double> skewness(std::span<const T> data) {
165 if (data.size() < 3) {
166 return Result<double>::error(ErrorCode::InsufficientData,
167 "Need at least 3 data points for skewness calculation");
168 }
169
170 auto mean_result = mean(data);
171 auto std_result = standard_deviation(data);
172
173 if (mean_result.is_error())
174 return mean_result;
175 if (std_result.is_error())
176 return std_result;
177
178 double data_mean = mean_result.value();
179 double data_std = std_result.value();
180
181 if (data_std == 0.0) {
182 return Result<double>::error(ErrorCode::DivisionByZero,
183 "Cannot calculate skewness with zero standard deviation");
184 }
185
186 double sum_cubed = 0.0;
187 for (const auto& value : data) {
188 double standardized = (static_cast<double>(value) - data_mean) / data_std;
189 sum_cubed += standardized * standardized * standardized;
190 }
191
192 return Result<double>::success(sum_cubed / data.size());
193}

198template <Numeric T>
199Result<double> kurtosis(std::span<const T> data, bool excess = true) {
200 if (data.size() < 4) {
201 return Result<double>::error(ErrorCode::InsufficientData,
202 "Need at least 4 data points for kurtosis calculation");
203 }
204
205 auto mean_result = mean(data);
206 auto std_result = standard_deviation(data);
207
208 if (mean_result.is_error())
209 return mean_result;
210 if (std_result.is_error())
211 return std_result;
212
213 double data_mean = mean_result.value();
214 double data_std = std_result.value();
215
216 if (data_std == 0.0) {
217 return Result<double>::error(ErrorCode::DivisionByZero,
218 "Cannot calculate kurtosis with zero standard deviation");
219 }
220
221 double sum_fourth = 0.0;
222 for (const auto& value : data) {
223 double standardized = (static_cast<double>(value) - data_mean) / data_std;
224 double fourth_power = standardized * standardized * standardized * standardized;
225 sum_fourth += fourth_power;
226 }
227
228 double kurt = sum_fourth / data.size();
229 return Result<double>::success(excess ? kurt - 3.0 : kurt);
230}

235template <Numeric T1, Numeric T2>
236Result<double> correlation(std::span<const T1> x, std::span<const T2> y) {
237 if (x.size() != y.size()) {
238 return Result<double>::error(ErrorCode::InvalidInput,
239 "Data series must have the same length for correlation calculation");
240 }
241
242 if (x.size() < 2) {
243 return Result<double>::error(ErrorCode::InsufficientData,
244 "Need at least 2 data points for correlation calculation");
245 }
246
247 auto x_mean_result = mean(x);
248 auto y_mean_result = mean(y);
249
250 if (x_mean_result.is_error())
251 return x_mean_result;
252 if (y_mean_result.is_error())
253 return y_mean_result;
254
255 double x_mean = x_mean_result.value();
256 double y_mean = y_mean_result.value();
257
258 double numerator = 0.0;
259 double x_sum_sq = 0.0;
260 double y_sum_sq = 0.0;
261
262 for (size_t i = 0; i < x.size(); ++i) {
263 double x_diff = static_cast<double>(x[i]) - x_mean;
264 double y_diff = static_cast<double>(y[i]) - y_mean;
265
266 numerator += x_diff * y_diff;
267 x_sum_sq += x_diff * x_diff;
268 y_sum_sq += y_diff * y_diff;
269 }
270
271 double denominator = std::sqrt(x_sum_sq * y_sum_sq);
272
273 if (denominator == 0.0) {
274 return Result<double>::error(ErrorCode::DivisionByZero, "Cannot calculate correlation with zero variance");
275 }
276
277 return Result<double>::success(numerator / denominator);
278}

285 if (x.size() != y.size()) {
286 return Result<double>::error(ErrorCode::InvalidInput,
287 "Data series must have the same length for covariance calculation");
288 }
289
290 if (x.size() < (sample ? 2 : 1)) {
291 return Result<double>::error(ErrorCode::InsufficientData, "Insufficient data for covariance calculation");
292 }
293
294 auto x_mean_result = mean(x);
295 auto y_mean_result = mean(y);
296
297 if (x_mean_result.is_error())
298 return x_mean_result;

300 return y_mean_result;
301
302 double x_mean = x_mean_result.value();
303 double y_mean = y_mean_result.value();
304

319template <Numeric T>
320Result<double> percentile(std::span<const T> data, double p) {
321 if (data.empty()) {
322 return Result<double>::error(ErrorCode::InsufficientData, "Cannot calculate percentile of empty data");
323 }
324
325 if (p < 0.0 || p > 100.0) {
326 return Result<double>::error(ErrorCode::InvalidInput, "Percentile must be between 0 and 100");
327 }
328
329 std::vector<T> sorted_data(data.begin(), data.end());
330 std::sort(sorted_data.begin(), sorted_data.end());
331
332 if (p == 0.0)
333 return Result<double>::success(static_cast<double>(sorted_data.front()));
334 if (p == 100.0)
335 return Result<double>::success(static_cast<double>(sorted_data.back()));
336
337 double index = (p / 100.0) * (sorted_data.size() - 1);
338 size_t lower_index = static_cast<size_t>(std::floor(index));
339 size_t upper_index = static_cast<size_t>(std::ceil(index));
340
341 if (lower_index == upper_index) {
342 return Result<double>::success(static_cast<double>(sorted_data[lower_index]));
343 }
344
345 double weight = index - lower_index;
346 double interpolated = static_cast<double>(sorted_data[lower_index]) * (1.0 - weight) +
347 static_cast<double>(sorted_data[upper_index]) * weight;
348
349 return Result<double>::success(interpolated);
350}

375template <Numeric T>
376Result<double> value_at_risk(std::span<const T> returns, double confidence_level = 0.95) {
377 if (confidence_level <= 0.0 || confidence_level >= 1.0) {
378 return Result<double>::error(ErrorCode::InvalidInput, "Confidence level must be between 0 and 1 (exclusive)");
379 }
380
381 double percentile_level = (1.0 - confidence_level) * 100.0;
382 auto var_result = percentile(returns, percentile_level);
383
384 if (var_result.is_error()) {
385 return var_result;
386 }
387
388 // VaR should be negative for losses (negative returns)
389 // Since we're calculating the percentile of negative returns, return as-is
390 return Result<double>::success(var_result.value());
391}

396template <Numeric T>
397Result<double> conditional_value_at_risk(std::span<const T> returns, double confidence_level = 0.95) {
398 auto var_result = value_at_risk(returns, confidence_level);
399 if (var_result.is_error()) {
400 return var_result;
401 }
402
403 double var_threshold = var_result.value(); // Use VaR threshold directly
404
405 // Calculate mean of returns below VaR threshold
406 std::vector<double> tail_returns;
407 for (const auto& ret : returns) {
408 if (static_cast<double>(ret) <= var_threshold) {
409 tail_returns.push_back(static_cast<double>(ret));
410 }
411 }
412
413 if (tail_returns.empty()) {
414 return Result<double>::error(ErrorCode::InsufficientData, "No returns below VaR threshold found");
415 }
416
417 auto tail_mean_result = mean(std::span<const double>{tail_returns});
418 if (tail_mean_result.is_error()) {
419 return tail_mean_result;
420 }
421
422 // CVaR should be negative for losses (same as VaR convention)
423 return Result<double>::success(tail_mean_result.value());
424}

429template <typename T, typename F>
430Result<std::vector<double>> rolling_statistic(std::span<const T> data, size_t window_size, F&& func) {
431 if (window_size == 0 || window_size > data.size()) {
432 return Result<std::vector<double>>::error(ErrorCode::InvalidInput,
433 "Invalid window size for rolling calculation");
434 }
435
436 std::vector<double> results;
437 results.reserve(data.size() - window_size + 1);
438
439 for (size_t i = window_size - 1; i < data.size(); ++i) {
440 std::span<const T> window{data.data() + i - window_size + 1, window_size};
441 auto result = func(window);
442
443 if (result.is_error()) {
444 return Result<std::vector<double>>::error(result.error().code,
445 "Rolling calculation failed: " + result.error().message);
446 }
447
448 results.push_back(result.value());
449 }
450
451 return Result<std::vector<double>>::success(std::move(results));
452}

522inline double normal_ppf(double p) {
523 if (p <= 0.0 || p >= 1.0) {
524 if (p == 0.0)
525 return -std::numeric_limits<double>::infinity();
526 if (p == 1.0)
527 return std::numeric_limits<double>::infinity();
528 return std::numeric_limits<double>::quiet_NaN();
529 }
530
531 // Coefficients for Beasley-Springer-Moro algorithm
532 static const double a[] = {-3.969683028665376e+01, 2.209460984245205e+02, -2.759285104469687e+02,
533 1.383577518672690e+02, -3.066479806614716e+01, 2.506628277459239e+00};
534
535 static const double b[] = {-5.447609879822406e+01, 1.615858368580409e+02, -1.556989798598866e+02,
536 6.680131188771972e+01, -1.328068155288572e+01};
537
538 static const double c[] = {-7.784894002430293e-03, -3.223964580411365e-01, -2.400758277161838e+00,
539 -2.549732539343734e+00, 4.374664141464968e+00, 2.938163982698783e+00};
540
541 static const double d[] = {7.784695709041462e-03, 3.224671290700398e-01, 2.445134137142996e+00,
542 3.754408661907416e+00};
543
544 double x, r;
545
546 if (p < 0.02425) {
547 // Tail region
548 x = std::sqrt(-2.0 * std::log(p));
549 x = (((((c[0] * x + c[1]) * x + c[2]) * x + c[3]) * x + c[4]) * x + c[5]) /
550 ((((d[0] * x + d[1]) * x + d[2]) * x + d[3]) * x + 1.0);
551 } else if (p > 0.97575) {
552 // Upper tail region
553 x = std::sqrt(-2.0 * std::log(1.0 - p));
554 x = -(((((c[0] * x + c[1]) * x + c[2]) * x + c[3]) * x + c[4]) * x + c[5]) /
555 ((((d[0] * x + d[1]) * x + d[2]) * x + d[3]) * x + 1.0);
556 } else {
557 // Central region
558 x = p - 0.5;
559 r = x * x;
560 x = (((((a[0] * r + a[1]) * r + a[2]) * r + a[3]) * r + a[4]) * r + a[5]) * x /
561 (((((b[0] * r + b[1]) * r + b[2]) * r + b[3]) * r + b[4]) * r + 1.0);
562 }
563
564 return x;
565}
