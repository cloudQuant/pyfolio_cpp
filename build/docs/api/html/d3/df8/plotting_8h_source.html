223inline std::vector<double> utils::calculate_cumulative_returns(const TimeSeries<Return>& returns) {
224 std::vector<double> cum_returns;
225 cum_returns.reserve(returns.size());
226
227 double cum_prod = 1.0;
228 for (size_t i = 0; i < returns.size(); ++i) {
229 cum_prod *= (1.0 + returns[i]);
230 cum_returns.push_back(cum_prod - 1.0); // Convert to percentage
231 }
232
233 return cum_returns;
234}

236inline std::pair<std::vector<std::string>, std::vector<double>> utils::calculate_annual_returns(
237 const TimeSeries<Return>& returns) {
238 std::unordered_map<int, std::vector<double>> yearly_returns;
239
240 const auto& timestamps = returns.timestamps();
241 const auto& values = returns.values();
242
243 for (size_t i = 0; i < timestamps.size(); ++i) {
244 int year = timestamps[i].year();
245 yearly_returns[year].push_back(values[i]);
246 }
247
248 std::vector<std::string> years;
249 std::vector<double> annual_rets;
250
251 for (const auto& [year, rets] : yearly_returns) {
252 years.push_back(std::to_string(year));
253
254 // Calculate annual return as compound growth
255 double annual_ret = 1.0;
256 for (double ret : rets) {
257 annual_ret *= (1.0 + ret);
258 }
259 annual_rets.push_back(annual_ret - 1.0);
260 }
261
262 return {years, annual_rets};
263}

556inline std::string PlotEngine::generate_svg_plot(const std::vector<PlotData>& series, const PlotConfig& config) {
557 // Simple SVG generation for basic line plots
558 std::ostringstream svg;
559
560 svg << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
561 << "<svg width=\"" << config.figsize.first * 80 << "\" height=\"" << config.figsize.second * 80 << "\" "
562 << "xmlns=\"http://www.w3.org/2000/svg\">\n"
563 << " <rect width=\"100%\" height=\"100%\" fill=\"white\"/>\n"
564 << " <g transform=\"translate(60,20)\">\n"

567 << " \n"
568 << " <!-- Plot area -->\n"
569 << " <rect x=\"0\" y=\"40\" width=\"" << config.figsize.first * 70 << "\" height=\""
570 << config.figsize.second * 60 << "\" "
571 << "fill=\"none\" stroke=\"black\" stroke-width=\"1\"/>\n"
572 << " \n"
573 << " <!-- Simple line plot -->\n";
574
575 if (!series.empty()) {
576 const auto& first_series = series[0];
577 if (!first_series.values.empty()) {
578 svg << "\n <polyline fill=\"none\" stroke=\"" << (series[0].color.empty() ? "#1f77b4" : series[0].color)
579 << "\" stroke-width=\"2\" points=\"";
580
581 // Scale data to fit plot area
582 double min_val = *std::min_element(first_series.values.begin(), first_series.values.end());
583 double max_val = *std::max_element(first_series.values.begin(), first_series.values.end());
584 double range = max_val - min_val;
585 if (range == 0)
586 range = 1.0;
587
588 double plot_width = config.figsize.first * 70.0;
589 double plot_height = config.figsize.second * 60.0;
590
591 for (size_t i = 0; i < first_series.values.size(); ++i) {
592 double x = (static_cast<double>(i) / (first_series.values.size() - 1)) * plot_width;
593 double y = plot_height - ((first_series.values[i] - min_val) / range) * plot_height + 40;
594
595 if (i > 0)
596 svg << " ";
597 svg << x << "," << y;
598 }
599 svg << R"("/>)";
600 }
601 }
602
603 svg << R"(
604 </g>
605</svg>)";
606
607 return svg.str();
608}

669inline Result<std::string> plots::plot_returns_distribution(const TimeSeries<Return>& returns, int bins,
670 const PlotConfig& config) {
671 // Simple histogram implementation
672 auto values = returns.values();
673 if (values.empty()) {
674 return Result<std::string>::error(ErrorCode::InsufficientData, "No returns data");
675 }
676
677 double min_val = *std::min_element(values.begin(), values.end());
678 double max_val = *std::max_element(values.begin(), values.end());
679 double bin_width = (max_val - min_val) / bins;
680
681 std::vector<int> histogram(bins, 0);
682 for (double val : values) {
683 int bin = static_cast<int>((val - min_val) / bin_width);
684 if (bin >= bins)
685 bin = bins - 1;
686 if (bin >= 0)
687 histogram[bin]++;
688 }
689
690 // Convert to plot data
691 PlotData data;
692 data.label = "Returns Distribution";
693 data.color = "#d62728";
694 data.style = "bar";
695
696 for (int i = 0; i < bins; ++i) {
697 data.values.push_back(static_cast<double>(histogram[i]));
698 data.timestamps.push_back(DateTime(2023, 1, 1 + i)); // Dummy dates for x-axis
699 }
700
701 PlotEngine engine;
702 return engine.create_line_plot({data}, config);
703}
