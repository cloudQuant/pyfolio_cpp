150 void handle_performance_analysis(const httplib::Request& req, httplib::Response& res) {
151 try {
152 json request_data = json::parse(req.body);
153
154 // Parse returns time series
155 auto returns_result = JsonSerializer::parse_time_series<double>(request_data["returns"]);
156 if (returns_result.is_error()) {
157 res.status = 400;
158 res.set_content(
159 JsonSerializer::create_error_response(returns_result.error().code, returns_result.error().message)
160 .dump(),
161 "application/json");
162 return;
163 }
164
165 // Parse optional benchmark
166 std::optional<TimeSeries<double>> benchmark;
167 if (request_data.contains("benchmark")) {
168 auto bench_result = JsonSerializer::parse_time_series<double>(request_data["benchmark"]);
169 if (bench_result.is_ok()) {
170 benchmark = bench_result.value();
171 }
172 }
173
174 // Perform analysis
175 auto analysis_result = analysis_suite_->analyze_performance_parallel(returns_result.value(), benchmark);
176
177 if (analysis_result.is_error()) {
178 res.status = 500;
179 res.set_content(
180 JsonSerializer::create_error_response(analysis_result.error().code, analysis_result.error().message)
181 .dump(),
182 "application/json");
183 return;
184 }
185
186 // Convert analysis report to JSON
187 const auto& report = analysis_result.value();
188 json response_data;
189 response_data["metrics"] = {
190 { "total_return", report.total_return},
191 { "annual_return", report.annual_return},
192 {"annual_volatility", report.annual_volatility},
193 { "sharpe_ratio", report.sharpe_ratio},
194 { "sortino_ratio", report.sortino_ratio},
195 { "max_drawdown", report.max_drawdown},
196 { "calmar_ratio", report.calmar_ratio},
197 { "var_95", report.var_95},
198 { "cvar_95", report.cvar_95},
199 { "skewness", report.skewness},
200 { "kurtosis", report.kurtosis}
201 };
202
203 if (report.alpha.has_value()) {
204 response_data["benchmark_metrics"] = {
205 { "alpha", report.alpha.value()},
206 { "beta", report.beta.value()},
207 {"information_ratio", report.information_ratio.value_or(0)},
208 { "tracking_error", report.tracking_error.value_or(0)}
209 };
210 }
211
212 response_data["risk_analysis"] = {
213 {"passed_risk_checks", report.passed_risk_checks},
214 { "warnings", report.warnings},
215 { "recommendations", report.recommendations}
216 };
217
218 response_data["computation_time_ms"] = report.computation_time.count();
219
220 json response = JsonSerializer::create_api_response(true, response_data);
221 res.set_content(response.dump(), "application/json");
222
223 } catch (const std::exception& e) {
