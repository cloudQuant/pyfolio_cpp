175 static Result<double> annual_return(const TimeSeries<T>& returns) {
176 auto mean_result = returns.mean();
177 if (mean_result.is_error()) {
178 return mean_result;
179 }
180 // Annualize assuming daily returns
181 double annual = mean_result.value() * constants::TRADING_DAYS_PER_YEAR;
182 return Result<double>::success(annual);
183 }
