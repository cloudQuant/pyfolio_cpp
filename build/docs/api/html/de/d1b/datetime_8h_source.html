249 DateTime add_months(int months) const {
250 auto date = to_date();
251 auto new_year = date.year();
252 auto new_month = static_cast<int>(static_cast<unsigned>(date.month())) + months;
253
254 // Handle month overflow/underflow
255 while (new_month > 12) {
256 new_month -= 12;
257 new_year = std::chrono::year{static_cast<int>(new_year) + 1};
258 }
259 while (new_month < 1) {
260 new_month += 12;
261 new_year = std::chrono::year{static_cast<int>(new_year) - 1};
262 }
263
264 auto new_date = new_year / std::chrono::month{static_cast<unsigned>(new_month)} / date.day();
265
266 // Handle invalid dates (e.g., Feb 30 -> Feb 28/29)
267 if (!new_date.ok()) {
268 // Use last day of month instead
269 new_date = new_year / std::chrono::month{static_cast<unsigned>(new_month)} / std::chrono::last;
270 }
271
272 return DateTime{new_date};
