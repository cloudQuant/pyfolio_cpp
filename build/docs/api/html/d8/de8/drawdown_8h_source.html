28inline Result<ReturnSeries> calculate_drawdowns(const ReturnSeries& returns) {
29 if (returns.empty()) {
30 return Result<ReturnSeries>::error(ErrorCode::InsufficientData,
31 "Cannot calculate drawdowns for empty return series");
32 }
33
34 // Calculate cumulative returns starting from 1.0
35 auto cumulative_result = calculate_cumulative_returns(returns, 1.0);
36 if (cumulative_result.is_error()) {
37 return Result<ReturnSeries>::error(cumulative_result.error().code, cumulative_result.error().message);
38 }
39
40 const auto& cumulative_returns = cumulative_result.value();
41
42 std::vector<Return> drawdowns;
43 drawdowns.reserve(cumulative_returns.size());
44
45 double running_max = 1.0; // Starting value
46
47 for (const auto& cum_ret : cumulative_returns) {
48 double current_value = 1.0 + cum_ret;
49
50 if (current_value > running_max) {
51 running_max = current_value;
52 }
53
54 // Calculate drawdown as percentage from peak
55 double drawdown = (running_max - current_value) / running_max;
56 drawdowns.push_back(drawdown);
57 }
58
59 return Result<ReturnSeries>::success(
60 ReturnSeries{cumulative_returns.timestamps(), std::move(drawdowns), returns.name() + "_drawdowns"});
61}

86inline Result<DrawdownInfo> max_drawdown_info(const ReturnSeries& returns) {
87 if (returns.empty()) {
88 return Result<DrawdownInfo>::error(ErrorCode::InsufficientData,
89 "Cannot calculate drawdown info for empty return series");
90 }
91
92 auto cumulative_result = calculate_cumulative_returns(returns, 1.0);
93 if (cumulative_result.is_error()) {
94 return Result<DrawdownInfo>::error(cumulative_result.error().code, cumulative_result.error().message);
95 }
96
97 const auto& cumulative_returns = cumulative_result.value();
98
99 double running_max = 1.0;
100 double max_drawdown_value = 0.0;
101
102 size_t peak_index = 0;
103 size_t trough_index = 0;
104 size_t recovery_index = 0;
105 bool found_recovery = false;
106
107 for (size_t i = 0; i < cumulative_returns.size(); ++i) {
108 double current_value = 1.0 + cumulative_returns[i];
109
110 if (current_value > running_max) {
111 running_max = current_value;
112 peak_index = i;
113
114 // Check if we've recovered from previous drawdown
115 if (!found_recovery && max_drawdown_value > 0.0) {
116 recovery_index = i;
117 found_recovery = true;
118 }
119 }
120
121 double drawdown = (running_max - current_value) / running_max;
122
123 if (drawdown > max_drawdown_value) {
124 max_drawdown_value = drawdown;
125 trough_index = i;
126 found_recovery = false; // Reset recovery flag for new max drawdown
127 }
128 }
129
130 // Check if recovery occurred after the maximum drawdown
131 if (!found_recovery && max_drawdown_value > 0.0) {
132 double peak_value = 1.0 + cumulative_returns[peak_index];
133
134 for (size_t i = trough_index + 1; i < cumulative_returns.size(); ++i) {
135 double current_value = 1.0 + cumulative_returns[i];
136 if (current_value >= peak_value) {
137 recovery_index = i;
138 found_recovery = true;
139 break;
140 }
141 }
142 }
143
144 DrawdownInfo info;
145 info.max_drawdown = max_drawdown_value;
146 info.start_date = returns.timestamp(peak_index);
147 info.end_date = returns.timestamp(trough_index);
148 info.recovered = found_recovery;
149
150 if (found_recovery) {
151 info.recovery_date = returns.timestamp(recovery_index);
152 info.recovery_days = static_cast<int>(recovery_index - trough_index);
153 } else {
154 info.recovery_date = DateTime::now(); // Placeholder
155 info.recovery_days = -1; // Not recovered
156 }
157
158 info.duration_days = static_cast<int>(trough_index - peak_index);
159
160 return Result<DrawdownInfo>::success(info);
161}

205inline Result<std::vector<DrawdownInfo>> find_drawdown_periods(const ReturnSeries& returns,
206 double min_drawdown = 0.01) {
207 if (returns.empty()) {
208 return Result<std::vector<DrawdownInfo>>::error(ErrorCode::InsufficientData,
209 "Cannot find drawdown periods for empty return series");
210 }
211
212 auto cumulative_result = calculate_cumulative_returns(returns, 1.0);
213 if (cumulative_result.is_error()) {
214 return Result<std::vector<DrawdownInfo>>::error(cumulative_result.error().code,
215 cumulative_result.error().message);
216 }
217
218 const auto& cumulative_returns = cumulative_result.value();
219
220 std::vector<DrawdownInfo> drawdown_periods;
221
222 double running_max = 1.0;
223 bool in_drawdown = false;
224 size_t peak_index = 0;
225 size_t trough_index = 0;
226 double period_max_drawdown = 0.0;
227
228 for (size_t i = 0; i < cumulative_returns.size(); ++i) {
229 double current_value = 1.0 + cumulative_returns[i];
230
231 if (current_value > running_max) {
232 // New peak reached
233 if (in_drawdown && period_max_drawdown >= min_drawdown) {
234 // End of significant drawdown period
235 DrawdownInfo info;
236 info.max_drawdown = period_max_drawdown;
237 info.start_date = returns.timestamp(peak_index);
238 info.end_date = returns.timestamp(trough_index);
239 info.recovery_date = returns.timestamp(i);
240 info.duration_days = static_cast<int>(trough_index - peak_index);
241 info.recovery_days = static_cast<int>(i - trough_index);
242 info.recovered = true;
243
244 drawdown_periods.push_back(info);
245 }
246
247 running_max = current_value;
248 peak_index = i;
249 in_drawdown = false;
250 period_max_drawdown = 0.0;
251 } else {
252 // Below peak - in drawdown
253 double drawdown = (running_max - current_value) / running_max;
254
255 if (!in_drawdown && drawdown >= min_drawdown) {
256 in_drawdown = true;
257 }
258
259 if (drawdown > period_max_drawdown) {
260 period_max_drawdown = drawdown;
261 trough_index = i;
262 }
263 }
264 }
265
266 // Handle case where series ends in drawdown
267 if (in_drawdown && period_max_drawdown >= min_drawdown) {
268 DrawdownInfo info;
269 info.max_drawdown = period_max_drawdown;
270 info.start_date = returns.timestamp(peak_index);
271 info.end_date = returns.timestamp(trough_index);
272 info.recovery_date = DateTime::now(); // Not recovered
273 info.duration_days = static_cast<int>(trough_index - peak_index);
274 info.recovery_days = -1;
275 info.recovered = false;
276
277 drawdown_periods.push_back(info);
278 }
279
280 return Result<std::vector<DrawdownInfo>>::success(std::move(drawdown_periods));
281}

286inline Result<double> average_drawdown(const ReturnSeries& returns) {
287 auto drawdowns_result = calculate_drawdowns(returns);
288 if (drawdowns_result.is_error()) {
289 return Result<double>::error(drawdowns_result.error().code, drawdowns_result.error().message);
290 }
291
292 const auto& drawdowns = drawdowns_result.value();
293
294 if (drawdowns.empty()) {
295 return Result<double>::error(ErrorCode::InsufficientData, "No drawdown data available");
296 }
297
298 // Only include actual drawdown periods (non-zero values)
299 std::vector<double> non_zero_drawdowns;
300 for (const auto& dd : drawdowns) {
301 if (dd > 0.0) {
302 non_zero_drawdowns.push_back(dd);
303 }
304 }
305
306 if (non_zero_drawdowns.empty()) {
307 return Result<double>::success(0.0);
308 }
309
310 double sum = std::accumulate(non_zero_drawdowns.begin(), non_zero_drawdowns.end(), 0.0);
311 double average = sum / non_zero_drawdowns.size();
312
313 return Result<double>::success(average);

319inline Result<ReturnSeries> rolling_max_drawdown(const ReturnSeries& returns, size_t window_size) {
320 if (window_size == 0 || window_size > returns.size()) {
321 return Result<ReturnSeries>::error(ErrorCode::InvalidInput, "Invalid window size for rolling maximum drawdown");
322 }
323
324 std::vector<Return> rolling_max_drawdowns;
325 std::vector<DateTime> rolling_timestamps;
326
327 rolling_max_drawdowns.reserve(returns.size() - window_size + 1);
328 rolling_timestamps.reserve(returns.size() - window_size + 1);
329
330 for (size_t i = window_size - 1; i < returns.size(); ++i) {
331 // Extract window
332 std::vector<DateTime> window_timestamps;
333 std::vector<Return> window_returns;
334
335 for (size_t j = i - window_size + 1; j <= i; ++j) {
336 window_timestamps.push_back(returns.timestamp(j));
337 window_returns.push_back(returns[j]);
338 }
339
340 ReturnSeries window_series{std::move(window_timestamps), std::move(window_returns)};
341
342 auto max_dd_result = max_drawdown(window_series);
343 if (max_dd_result.is_error()) {
344 rolling_max_drawdowns.push_back(std::numeric_limits<double>::quiet_NaN());
345 } else {
346 rolling_max_drawdowns.push_back(max_dd_result.value());
347 }
348
349 rolling_timestamps.push_back(returns.timestamp(i));
350 }
351
352 return Result<ReturnSeries>::success(ReturnSeries{std::move(rolling_timestamps), std::move(rolling_max_drawdowns),
353 returns.name() + "_rolling_max_drawdown"});

370

374 recovery_times.push_back(period.recovery_days);

376 }
377
378 if (recovery_times.empty()) {
379 return Result<double>::error(ErrorCode::InsufficientData, "No recovered drawdown periods found");
380 }
381
