175 template <typename TimestampIter, typename ValueIter>
176 [[nodiscard]] Result<void> bulk_insert(TimestampIter timestamps_begin, TimestampIter timestamps_end,
177 ValueIter values_begin, ValueIter values_end) {
178 size_type new_count = std::distance(timestamps_begin, timestamps_end);
179 size_type values_count = std::distance(values_begin, values_end);
180
181 if (new_count != values_count) {
182 return Result<void>::error(ErrorCode::InvalidInput, "Bulk insert: timestamp and value counts must match");
183 }
184
185 // Reserve space to avoid multiple reallocations
186 size_type old_size = size();
187 timestamps_.reserve(old_size + new_count);
188 values_.reserve(old_size + new_count);
189
190 // Insert new data
191 timestamps_.insert(timestamps_.end(), timestamps_begin, timestamps_end);
192 values_.insert(values_.end(), values_begin, values_end);
193
194 // Sort if needed (only if we added data that might be out of order)
195 if (old_size > 0 && new_count > 0) {
196 bool needs_sort = false;
197 DateTime last_old = timestamps_[old_size - 1];
198
199 // Check if any new timestamp is earlier than the last old timestamp
200 for (auto it = timestamps_begin; it != timestamps_end; ++it) {
201 if (*it < last_old) {
202 needs_sort = true;
203 break;
204 }
205 }
206
207 if (needs_sort) {
208 sort_by_time();
209 }
210 }
211
212 return Result<void>::success();
213 }

286 // Create pairs for sorting to maintain data integrity
287 std::vector<std::pair<DateTime, T>> paired_data;
288 paired_data.reserve(size());

350 requires std::floating_point<T>
351 {
352 // Default aggregator is mean
353 auto mean_aggregator = [](const std::vector<T>& values) -> T {
354 if (values.empty())
355 return T{0};
356 T sum = T{0};
357 for (const auto& val : values) {

359 }
360 return sum / static_cast<T>(values.size());

436 [[nodiscard]] Result<TimeSeries<T>> shift(int periods) const {
437 if (empty()) {
438 return Result<TimeSeries<T>>::success(*this);
439 }
440
441 std::vector<T> shifted_values(size());
442
443 if (periods > 0) {
444 // Forward shift - fill beginning with NaN
445 std::fill_n(shifted_values.begin(), std::min(static_cast<size_type>(periods), size()),
446 std::numeric_limits<T>::quiet_NaN());
447
448 if (static_cast<size_type>(periods) < size()) {
449 std::copy(values_.begin(), values_.end() - periods, shifted_values.begin() + periods);
450 }
451 } else if (periods < 0) {
452 // Backward shift - fill end with NaN
453 size_type abs_periods = static_cast<size_type>(-periods);
454
455 if (abs_periods < size()) {
456 std::copy(values_.begin() + abs_periods, values_.end(), shifted_values.begin());
457 }
458
459 std::fill(shifted_values.end() - std::min(abs_periods, size()), shifted_values.end(),
460 std::numeric_limits<T>::quiet_NaN());
461 } else {
462 // No shift
463 shifted_values = values_;
464 }
465
466 return Result<TimeSeries<T>>::success(
467 TimeSeries<T>{timestamps_, std::move(shifted_values), name_ + "_shifted"});
468 }

581 requires std::floating_point<T>
582 {
583 if (window == 0 || window > size()) {
584 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput,
585 "Invalid window size: " + std::to_string(window));

587
588 std::vector<DateTime> result_timestamps;
589 std::vector<T> result_values;
590 result_timestamps.reserve(size());

592
593 // Running sum for efficient calculation
594 T running_sum = T{0};
595
596 for (size_type i = 0; i < size(); ++i) {
597 result_timestamps.push_back(timestamps_[i]);
598
599 if (i < window) {
600 // Building up the initial window
601 running_sum += values_[i];

605 } else {

607 result_values.push_back(std::numeric_limits<T>::quiet_NaN());
608 }
609 } else {

611 running_sum = running_sum - values_[i - window] + values_[i];
612 result_values.push_back(running_sum / static_cast<T>(window));
613 }
614 }

623 [[nodiscard]] Result<TimeSeries<T>> rolling_std(size_type window) const
624 requires std::floating_point<T>
625 {
626 if (window == 0 || window > size()) {
627 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput,
628 "Invalid window size: " + std::to_string(window));
629 }
630
631 std::vector<DateTime> result_timestamps;
632 std::vector<T> result_values;
633 result_timestamps.reserve(size());
634 result_values.reserve(size());
635
636 // Running sums for efficient calculation
637 T running_sum = T{0};
638 T running_sum_sq = T{0};
639
640 for (size_type i = 0; i < size(); ++i) {
641 result_timestamps.push_back(timestamps_[i]);
642
643 if (i < window) {
644 // Building up the initial window
645 running_sum += values_[i];
646 running_sum_sq += values_[i] * values_[i];
647
648 if (i == window - 1) {
649 // First complete window
650 T mean = running_sum / static_cast<T>(window);
651 T variance = (running_sum_sq / static_cast<T>(window)) - (mean * mean);
652 result_values.push_back(std::sqrt(std::max(T{0}, variance)));
653 } else {
654 // Incomplete window
655 result_values.push_back(std::numeric_limits<T>::quiet_NaN());
656 }
657 } else {
658 // Sliding window: remove oldest, add newest
659 T old_val = values_[i - window];
660 T new_val = values_[i];
661
662 running_sum = running_sum - old_val + new_val;
663 running_sum_sq = running_sum_sq - (old_val * old_val) + (new_val * new_val);
664
665 T mean = running_sum / static_cast<T>(window);
666 T variance = (running_sum_sq / static_cast<T>(window)) - (mean * mean);
667 result_values.push_back(std::sqrt(std::max(T{0}, variance)));
668 }
669 }
670
671 return Result<TimeSeries<T>>::success(
672 TimeSeries<T>{std::move(result_timestamps), std::move(result_values), name_ + "_rolling_std"});

927 [[nodiscard]] Result<TimeSeries<T>> fill_missing(const std::vector<DateTime>& target_dates,
928 FillMethod method) const {
929 if (target_dates.empty()) {
930 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput, "Target dates cannot be empty");
931 }
932
933 std::vector<T> filled_values;
934 filled_values.reserve(target_dates.size());
935
936 for (const auto& target_date : target_dates) {
937 auto it = std::find(timestamps_.begin(), timestamps_.end(), target_date);
938
939 if (it != timestamps_.end()) {
940 // Date exists, use actual value
941 size_t idx = std::distance(timestamps_.begin(), it);
942 filled_values.push_back(values_[idx]);
943 } else {
944 // Date missing, apply fill method
945 switch (method) {
946 case FillMethod::Forward:
947 // Find last available value before or at this date
948 {
949 T fill_value = T{0};
950 bool found = false;
951
952 for (size_t i = 0; i < timestamps_.size(); ++i) {
953 if (timestamps_[i] <= target_date) {
954 fill_value = values_[i];
955 found = true;
956 } else {
957 break; // Stop at first date after target
958 }
959 }
960
961 if (found) {
962 filled_values.push_back(fill_value);
963 } else if (!empty()) {
964 // Use first value if no prior value found
965 filled_values.push_back(values_[0]);
966 } else {
967 return Result<TimeSeries<T>>::error(ErrorCode::InsufficientData,
968 "No data to fill from");
969 }
970 }
971 break;
972
973 case FillMethod::Backward:
974 // Find next available value after this date
975 for (size_t i = 0; i < timestamps_.size(); ++i) {
976 if (timestamps_[i] > target_date) {
977 filled_values.push_back(values_[i]);
978 goto next_date;
979 }
980 }
981 // If no future value found, use last value
982 if (!empty()) {
983 filled_values.push_back(values_.back());
984 } else {
985 return Result<TimeSeries<T>>::error(ErrorCode::InsufficientData, "No data to fill from");
986 }
987 break;
988
989 case FillMethod::Drop:
990 // For Drop method, we'll skip missing values (not implemented fully)
991 // For now, use zero as fallback
992 filled_values.push_back(T{0});
993 break;
994
995 default:
996 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput, "Unknown fill method");
997 }
998 next_date:;
999 }
1000 }
1001
1002 return Result<TimeSeries<T>>::success(TimeSeries<T>{target_dates, filled_values, name_ + "_filled"});
1003 }

1152 [[nodiscard]] Result<double> correlation(const TimeSeries<T>& other) const
1153 requires std::floating_point<T>
1154 {
1155 if (size() != other.size() || size() < 2) {
1156 return Result<double>::error(ErrorCode::InsufficientData,
1157 "Need matching series with at least 2 points for correlation");
1158 }
1159
1160 if constexpr (std::is_same_v<T, double>) {
1161 // Use SIMD-optimized version for double
1162 double result = simd::correlation_simd(std::span<const double>(values_.data(), values_.size()),
1163 std::span<const double>(other.values_.data(), other.values_.size()));
1164 return Result<double>::success(result);
1165 } else {
1166 // Fallback implementation for other types
1167 auto mean1_result = mean();
1168 auto mean2_result = other.mean();
1169
1170 if (mean1_result.is_error() || mean2_result.is_error()) {
1171 return Result<double>::error(ErrorCode::InsufficientData, "Failed to calculate means for correlation");
1172 }
1173
1174 double mean1 = static_cast<double>(mean1_result.value());
1175 double mean2 = static_cast<double>(mean2_result.value());
1176
1177 double numerator = 0.0, sum_sq1 = 0.0, sum_sq2 = 0.0;
1178
1179 for (size_t i = 0; i < size(); ++i) {
1180 double diff1 = static_cast<double>(values_[i]) - mean1;
1181 double diff2 = static_cast<double>(other.values_[i]) - mean2;
1182 numerator += diff1 * diff2;
1183 sum_sq1 += diff1 * diff1;
1184 sum_sq2 += diff2 * diff2;
1185 }
1186
1187 double denominator = std::sqrt(sum_sq1 * sum_sq2);
1188 if (denominator == 0.0) {
1189 return Result<double>::success(0.0);
1190 }
1191
1192 return Result<double>::success(numerator / denominator);
1193 }
