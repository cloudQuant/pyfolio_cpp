31 template <typename T>
32 Result<void> calculate_risk_metrics_parallel(const TimeSeries<T>& returns, AnalysisReport& report) const {
33 // Use parallel algorithms for large datasets
34 if (returns.size() >= 10000) {
35 // Parallel VaR and CVaR calculation
36 auto sorted_returns = returns.values();
37
38 // Sort the returns
39 std::sort(sorted_returns.begin(), sorted_returns.end());
40
41 if (!sorted_returns.empty()) {
42 size_t var_index = static_cast<size_t>(sorted_returns.size() * 0.05);
43 report.var_95 = sorted_returns[var_index];
44
45 // Parallel CVaR calculation
46 auto cvar_result = parallel_algo_->parallel_reduce(
47 std::vector<double>(sorted_returns.begin(), sorted_returns.begin() + var_index + 1), 0.0,
48 std::plus<double>());
49
50 if (cvar_result.is_ok()) {
51 report.cvar_95 = cvar_result.value() / (var_index + 1);
52 }
53 }
54
55 // Parallel downside deviation calculation
56 auto mean_result = parallel_algo_->parallel_mean(returns);
57 if (mean_result.is_error())
58 return Result<void>::error(mean_result.error().code, mean_result.error().message);
59
60 double mean_return = mean_result.value();
61 const auto& values = returns.values();
62
63 // Parallel calculation of downside deviations
64 auto downside_values_result = parallel_algo_->parallel_map(values, [mean_return](double val) -> double {
65 return val < mean_return ? (val - mean_return) * (val - mean_return) : 0.0;
66 });
67
68 if (downside_values_result.is_ok()) {
69 auto downside_sum_result =
70 parallel_algo_->parallel_reduce(downside_values_result.value(), 0.0, std::plus<double>());
71
72 if (downside_sum_result.is_ok()) {
73 size_t downside_count = std::count_if(values.begin(), values.end(),
74 [mean_return](double val) { return val < mean_return; });
75
76 report.downside_deviation =
77 downside_count > 0 ? std::sqrt(downside_sum_result.value() / downside_count) : 0.0;
78 }
79 }
80 } else {
81 // Use serial implementation for smaller datasets
82 return PerformanceAnalysisSuite::calculate_risk_metrics(returns, report);
83 }
84
85 return Result<void>::success();
86 }

91 template <typename T>
92 Result<void> calculate_rolling_metrics_parallel(const TimeSeries<T>& returns, AnalysisReport& report) const {
93 for (size_t window : get_config().rolling_windows) {
94 if (window < returns.size()) {
95 // Use parallel rolling calculations for large datasets
96 if (returns.size() >= 10000) {
97 auto rolling_mean_result = parallel_algo_->parallel_rolling_mean(returns, window);
98 if (rolling_mean_result.is_ok()) {
99 report.rolling_returns[window] = rolling_mean_result.value();
100 }
101
102 auto rolling_std_result = parallel_algo_->parallel_rolling_std(returns, window);
103 if (rolling_std_result.is_ok()) {
104 report.rolling_volatility[window] = rolling_std_result.value();
105
106 // Calculate rolling Sharpe ratio
107 const auto& vol_series = rolling_std_result.value();
108 const auto& ret_series = report.rolling_returns[window];
109
110 if (vol_series.size() == ret_series.size()) {
111 std::vector<DateTime> dates = vol_series.timestamps();
112 std::vector<double> sharpe_values;
113
114 double daily_rf = get_config().risk_free_rate / get_config().periods_per_year;
115
116 // Parallel Sharpe ratio calculation
117 std::vector<std::pair<double, double>> vol_ret_pairs;
118 for (size_t i = 0; i < vol_series.size(); ++i) {
119 vol_ret_pairs.emplace_back(vol_series[i], ret_series[i]);
120 }
121
122 auto sharpe_result = parallel_algo_->parallel_map(
123 vol_ret_pairs, [daily_rf](const std::pair<double, double>& pair) -> double {
124 double vol = pair.first;
125 double ret = pair.second;
126 if (vol > 0) {
127 double excess_return = ret - daily_rf;
128 return excess_return / vol;
129 }
130 return 0.0;
131 });
132
133 if (sharpe_result.is_ok()) {
134 report.rolling_sharpe[window] = TimeSeries<double>(
135 dates, sharpe_result.value(), "rolling_sharpe_" + std::to_string(window));
136 }
137 }
138 }
139 } else {
140 // Use cached calculations for smaller datasets
141 return PerformanceAnalysisSuite::calculate_rolling_metrics(returns, report);
142 }
143 }
144 }
145
146 return Result<void>::success();

158 [[nodiscard]] Result<AnalysisReport> analyze_performance_parallel(
159 const TimeSeries<T>& returns, const std::optional<TimeSeries<T>>& benchmark = std::nullopt) const {
160 auto start_time = std::chrono::high_resolution_clock::now();
161 AnalysisReport report{};
162 report.analysis_timestamp = std::chrono::steady_clock::now();
163
164 try {
165 // Determine if we should use parallel processing
166 bool use_parallel = returns.size() >= 10000;
167
168 if (use_parallel) {
169 // Parallel basic performance metrics
170 auto mean_result = parallel_algo_->parallel_mean(returns);
171 if (mean_result.is_error())
172 return Result<AnalysisReport>::error(mean_result.error().code, mean_result.error().message);
173
174 auto std_result = parallel_algo_->parallel_std_deviation(returns);
175 if (std_result.is_error())
176 return Result<AnalysisReport>::error(std_result.error().code, std_result.error().message);
177
178 // Calculate cumulative returns for total return
179 auto cumulative_result = returns.cumsum();
180 if (cumulative_result.is_error())
181 return Result<AnalysisReport>::error(cumulative_result.error().code,
182 cumulative_result.error().message);
183
184 const auto& cum_returns = cumulative_result.value();
185 report.total_return = cum_returns.empty() ? 0.0 : cum_returns[cum_returns.size() - 1];
186
187 // Annualized metrics
188 report.annual_return = mean_result.value() * get_config().periods_per_year;
189 report.annual_volatility = std_result.value() * std::sqrt(get_config().periods_per_year);
190
191 // Parallel Sharpe ratio calculation
192 double daily_rf = get_config().risk_free_rate / get_config().periods_per_year;
193 double excess_return = mean_result.value() - daily_rf;
194 report.sharpe_ratio = std_result.value() > 0 ? excess_return / std_result.value() : 0.0;
195
196 // Parallel maximum drawdown calculation
197 auto max_dd_result = parallel_algo_->parallel_reduce(
198 cum_returns.values(), std::pair<double, double>{cum_returns[0], 0.0}, // {peak, max_drawdown}
199 [](const std::pair<double, double>& acc, double val) {
200 double new_peak = std::max(acc.first, val);
201 double drawdown = new_peak > 0 ? (new_peak - val) / new_peak : 0.0;
202 double max_dd = std::max(acc.second, drawdown);
203 return std::make_pair(new_peak, max_dd);
204 });
205
206 if (max_dd_result.is_ok()) {
207 report.max_drawdown = max_dd_result.value().second;
208 }
209
210 // Calmar ratio
211 report.calmar_ratio = report.max_drawdown > 0 ? report.annual_return / report.max_drawdown : 0.0;
212
213 // Parallel risk metrics
214 auto risk_result = calculate_risk_metrics_parallel(returns, report);
215 if (risk_result.is_error())
216 return Result<AnalysisReport>::error(risk_result.error().code, risk_result.error().message);
217
218 // Parallel rolling metrics
219 auto rolling_result = calculate_rolling_metrics_parallel(returns, report);
220 if (rolling_result.is_error())
221 return Result<AnalysisReport>::error(rolling_result.error().code, rolling_result.error().message);
222
223 // Parallel benchmark comparison if provided
224 if (benchmark) {
225 auto benchmark_corr = parallel_algo_->parallel_correlation(returns, *benchmark);
226 if (benchmark_corr.is_ok()) {
227 auto benchmark_mean = parallel_algo_->parallel_mean(*benchmark);
228 auto benchmark_std = parallel_algo_->parallel_std_deviation(*benchmark);
229
230 if (benchmark_mean.is_ok() && benchmark_std.is_ok()) {
231 // Beta calculation
232 double covariance = benchmark_corr.value() * std_result.value() * benchmark_std.value();
233 double benchmark_variance = benchmark_std.value() * benchmark_std.value();
234 report.beta = benchmark_variance > 0 ? covariance / benchmark_variance : 0.0;
235
236 // Alpha calculation
237 double benchmark_annual_return = benchmark_mean.value() * get_config().periods_per_year;
238 report.alpha =
239 report.annual_return -
240 (get_config().risk_free_rate +
241 report.beta.value() * (benchmark_annual_return - get_config().risk_free_rate));
242
243 // Parallel tracking error calculation
244 std::vector<double> excess_returns;
245 const auto& ret_vals = returns.values();
246 const auto& bench_vals = benchmark->values();
247
248 size_t min_size = std::min(ret_vals.size(), bench_vals.size());
249 excess_returns.reserve(min_size);
250
251 for (size_t i = 0; i < min_size; ++i) {
252 excess_returns.push_back(ret_vals[i] - bench_vals[i]);
253 }
254
255 if (!excess_returns.empty()) {
256 // Create TimeSeries for parallel processing
257 std::vector<DateTime> excess_dates(
258 returns.timestamps().begin(), returns.timestamps().begin() + excess_returns.size());
259
260 TimeSeries<double> excess_ts(excess_dates, excess_returns, "excess_returns");
261
262 auto excess_std = parallel_algo_->parallel_std_deviation(excess_ts);
263 auto excess_mean = parallel_algo_->parallel_mean(excess_ts);
264
265 if (excess_std.is_ok() && excess_mean.is_ok()) {
266 report.tracking_error =
267 excess_std.value() * std::sqrt(get_config().periods_per_year);
268 report.information_ratio =
269 report.tracking_error.value() > 0
270 ? (excess_mean.value() * get_config().periods_per_year) /
271 report.tracking_error.value()
272 : 0.0;
273 }
274 }
275 }
276 }
277 }
278 } else {
279 // Use cached performance analysis suite for smaller datasets
280 return PerformanceAnalysisSuite::analyze_performance(returns, benchmark);
281 }
282
283 // Statistical moments calculation (parallel for large datasets)
284 const auto& values = returns.values();
285 if (!values.empty() && use_parallel) {
286 auto mean_result = parallel_algo_->parallel_mean(returns);
287 if (mean_result.is_ok()) {
288 double mean_val = mean_result.value();
289
290 // Parallel moments calculation
291 auto moments_result = parallel_algo_->parallel_reduce(
292 values, std::array<double, 3>{0.0, 0.0, 0.0}, // {variance_sum, skew_sum, kurt_sum}
293 [mean_val](const std::array<double, 3>& acc, double val) {
294 double diff = val - mean_val;
295 double diff2 = diff * diff;
296 return std::array<double, 3>{acc[0] + diff2, acc[1] + diff2 * diff, acc[2] + diff2 * diff2};
297 });
298
299 if (moments_result.is_ok()) {
300 const auto& moments = moments_result.value();
301 double variance = moments[0] / values.size();
302 double std_dev = std::sqrt(variance);
303
304 if (std_dev > 0) {
305 report.skewness = (moments[1] / values.size()) / (std_dev * std_dev * std_dev);
306 report.kurtosis = (moments[2] / values.size()) / (variance * variance) - 3.0;
307 }
308 }
309 }
310 }
311
312 // Sortino ratio calculation
313 report.sortino_ratio = report.downside_deviation > 0
314 ? (report.annual_return - get_config().risk_free_rate) /
315 (report.downside_deviation * std::sqrt(get_config().periods_per_year))
316 : 0.0;
317
318 // Performance analysis and risk checks
319 // Risk analysis performed in parallel above
320
321 // Cache statistics (from parent class)
322 // Note: Parallel processing doesn't use the cache directly
323
324 // Timing information
325 auto end_time = std::chrono::high_resolution_clock::now();
326 report.computation_time = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
327
328 return Result<AnalysisReport>::success(std::move(report));
329
330 } catch (const std::exception& e) {
331 return Result<AnalysisReport>::error(ErrorCode::CalculationError,
332 "Parallel analysis failed: " + std::string(e.what()));
