15inline Result<ReturnSeries> calculate_returns(const PriceSeries& prices) {
16 if (prices.size() < 2) {
17 return Result<ReturnSeries>::error(ErrorCode::InsufficientData,
18 "Need at least 2 price points to calculate returns");
19 }
20
21 std::vector<DateTime> return_timestamps;
22 std::vector<Return> returns;
23
24 return_timestamps.reserve(prices.size() - 1);
25 returns.reserve(prices.size() - 1);
26
27 for (size_t i = 1; i < prices.size(); ++i) {
28 double prev_price = prices[i - 1];
29 double curr_price = prices[i];
30
31 if (prev_price <= 0.0) {
32 return Result<ReturnSeries>::error(ErrorCode::InvalidInput,
33 "Price must be positive for return calculation");
34 }
35
36 double simple_return = (curr_price - prev_price) / prev_price;
37
38 return_timestamps.push_back(prices.timestamp(i));
39 returns.push_back(simple_return);
40 }
41
42 return Result<ReturnSeries>::success(
43 ReturnSeries{std::move(return_timestamps), std::move(returns), prices.name() + "_returns"});
44}

125inline Result<double> annualize_returns(const ReturnSeries& returns, Frequency frequency = Frequency::Daily) {
126 if (returns.empty()) {
127 return Result<double>::error(ErrorCode::InsufficientData, "Cannot annualize empty return series");
128 }
129
130 auto mean_result = stats::mean(std::span<const Return>{returns.values()});
131 if (mean_result.is_error()) {
132 return mean_result;
133 }
134
135 double period_return = mean_result.value();
136 double periods_per_year = frequency::to_annual_factor(frequency);
137
138 // Compound the returns
139 double annualized = std::pow(1.0 + period_return, periods_per_year) - 1.0;
140
141 return Result<double>::success(annualized);

275inline Result<double> calculate_cagr(const ReturnSeries& returns, Frequency frequency = Frequency::Daily) {
276 if (returns.empty()) {
277 return Result<double>::error(ErrorCode::InsufficientData, "Cannot calculate CAGR for empty return series");
278 }
279
280 auto total_return_result = total_return(returns);
281 if (total_return_result.is_error()) {
282 return total_return_result;
283 }
284
285 double total_ret = total_return_result.value();
286 double periods_per_year = frequency::to_annual_factor(frequency);
287 double years = returns.size() / periods_per_year;
288
289 if (years <= 0.0) {
290 return Result<double>::error(ErrorCode::InvalidInput, "Need at least one year of data for CAGR calculation");
291 }
292
293 double cagr = std::pow(1.0 + total_ret, 1.0 / years) - 1.0;
294
295 return Result<double>::success(cagr);
296}
