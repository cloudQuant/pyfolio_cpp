132 Result<PortfolioFactorExposure> calculate_portfolio_exposures(
133 const std::map<Symbol, double>& portfolio_weights,
134 const std::map<Symbol, double>& benchmark_weights = {}) const {
135 if (portfolio_weights.empty()) {
136 return Result<PortfolioFactorExposure>::error(ErrorCode::InvalidInput, "Portfolio weights cannot be empty");
137 }
138
139 // Validate weights sum to 1
140 double total_weight = std::accumulate(portfolio_weights.begin(), portfolio_weights.end(), 0.0,
141 [](double sum, const auto& pair) { return sum + std::abs(pair.second); });
142
143 if (std::abs(total_weight - 1.0) > 1e-6) {
144 return Result<PortfolioFactorExposure>::error(ErrorCode::InvalidInput, "Portfolio weights must sum to 1.0");
145 }
146
147 PortfolioFactorExposure result;
148
149 // Get all unique factors
150 std::set<std::string> all_factors;
151 for (const auto& [symbol, exposure] : security_exposures_) {
152 for (const auto& [factor, loading] : exposure.factor_loadings) {
153 all_factors.insert(factor);
154 }
155 }
156
157 // Calculate portfolio exposures
158 for (const std::string& factor : all_factors) {
159 double net_exposure = 0.0;
160 double gross_exposure = 0.0;
161 double benchmark_exposure = 0.0;
162
163 // Portfolio exposure
164 for (const auto& [symbol, weight] : portfolio_weights) {
165 auto exposure_it = security_exposures_.find(symbol);
166 if (exposure_it != security_exposures_.end()) {
167 double factor_loading = exposure_it->second.get_factor_exposure(factor);
168 net_exposure += weight * factor_loading;
169 gross_exposure += std::abs(weight * factor_loading);
170 }
171 }
172
173 // Benchmark exposure (if provided)
174 if (!benchmark_weights.empty()) {
175 for (const auto& [symbol, weight] : benchmark_weights) {
176 auto exposure_it = security_exposures_.find(symbol);
177 if (exposure_it != security_exposures_.end()) {
178 double factor_loading = exposure_it->second.get_factor_exposure(factor);
179 benchmark_exposure += weight * factor_loading;
180 }
181 }
182 }
183
184 result.net_exposures[factor] = net_exposure;
185 result.gross_exposures[factor] = gross_exposure;
186 result.active_exposures[factor] = net_exposure - benchmark_exposure;
187
188 // Calculate concentration (sum of squares)
189 double concentration = 0.0;
190 for (const auto& [symbol, weight] : portfolio_weights) {
191 auto exposure_it = security_exposures_.find(symbol);
192 if (exposure_it != security_exposures_.end()) {
193 double factor_loading = exposure_it->second.get_factor_exposure(factor);
194 concentration += (weight * factor_loading) * (weight * factor_loading);
195 }
196 }
197 result.exposure_concentrations[factor] = concentration;
198 }
199
200 return Result<PortfolioFactorExposure>::success(std::move(result));
201 }

206 Result<FactorRiskAttribution> calculate_risk_attribution(
207 const std::map<Symbol, double>& portfolio_weights, const std::map<Symbol, double>& specific_risks = {}) const {
208 // Calculate portfolio factor exposures
209 auto exposure_result = calculate_portfolio_exposures(portfolio_weights);
210 if (exposure_result.is_error()) {
211 return Result<FactorRiskAttribution>::error(exposure_result.error().code, exposure_result.error().message);
212 }
213
214 const auto& exposures = exposure_result.value();
215 FactorRiskAttribution result;
216
217 // Calculate factor risk contributions
218 for (const auto& [factor1, exposure1] : exposures.net_exposures) {
219 double factor_contribution = 0.0;
220
221 for (const auto& [factor2, exposure2] : exposures.net_exposures) {
222 // Get factor covariance
223 double covariance = get_factor_covariance(factor1, factor2);
224 factor_contribution += exposure1 * exposure2 * covariance;
225 }
226
227 result.factor_contributions[factor1] = factor_contribution;
228 }
229
230 // Calculate specific risk contribution
231 result.specific_risk_contribution = 0.0;
232 for (const auto& [symbol, weight] : portfolio_weights) {
233 auto specific_risk_it = specific_risks.find(symbol);
234 if (specific_risk_it != specific_risks.end()) {
235 double specific_risk = specific_risk_it->second;
236 result.specific_risk_contribution += weight * weight * specific_risk * specific_risk;
237 }
238 }
239
240 // Calculate total risk
241 result.total_risk = 0.0;
242 for (const auto& [factor, contribution] : result.factor_contributions) {
243 result.total_risk += contribution;
244 }
245 result.total_risk += result.specific_risk_contribution;
246 result.total_risk = std::sqrt(std::max(0.0, result.total_risk));
247
248 return Result<FactorRiskAttribution>::success(std::move(result));
249 }

352inline Result<FactorModel> create_factor_model_from_returns(
353 const std::map<std::string, TimeSeries<double>>& factor_returns,
354 const std::map<Symbol, std::map<std::string, double>>& factor_loadings) {
355 if (factor_returns.empty()) {
356 return Result<FactorModel>::error(ErrorCode::InvalidInput, "Factor returns cannot be empty");
357 }
358
359 FactorModel model;
360 model.set_factor_returns(factor_returns);
361
362 // Convert loadings to SecurityFactorExposure format
363 std::map<Symbol, SecurityFactorExposure> exposures;
364 for (const auto& [symbol, loadings] : factor_loadings) {
365 SecurityFactorExposure exposure;
366 exposure.symbol = symbol;
367 exposure.factor_loadings = loadings;
368 exposures[symbol] = exposure;
369 }
370 model.set_security_exposures(exposures);
371
372 // Calculate factor covariance matrix
373 std::map<std::string, std::map<std::string, double>> covariance;
374 std::vector<std::string> factor_names;
375 for (const auto& [factor, ts] : factor_returns) {
376 factor_names.push_back(factor);
377 }
378
379 for (const auto& factor1 : factor_names) {
380 for (const auto& factor2 : factor_names) {
381 const auto& ts1 = factor_returns.at(factor1);
382 const auto& ts2 = factor_returns.at(factor2);
383
384 if (ts1.size() == ts2.size() && ts1.size() > 1) {
385 auto cov_result =
386 stats::covariance(std::span<const double>{ts1.values()}, std::span<const double>{ts2.values()});
387
388 if (cov_result.is_ok()) {
389 covariance[factor1][factor2] = cov_result.value();
390 } else {
391 covariance[factor1][factor2] = (factor1 == factor2) ? 1.0 : 0.0;
392 }
393 } else {
394 covariance[factor1][factor2] = (factor1 == factor2) ? 1.0 : 0.0;
395 }
396 }
397 }
398
399 model.set_factor_covariance(covariance);
400
401 return Result<FactorModel>::success(std::move(model));
402}
