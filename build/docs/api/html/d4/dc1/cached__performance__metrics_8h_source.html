29template <typename T>
30struct CacheEntry {
31 T value;
32 std::chrono::steady_clock::time_point timestamp;
33 size_t data_hash;
34 size_t data_size;
35
36 CacheEntry() = default;
37 CacheEntry(T val, size_t hash, size_t size)
38 : value(std::move(val)), timestamp(std::chrono::steady_clock::now()), data_hash(hash), data_size(size) {}
39
40 bool is_valid(size_t current_hash, size_t current_size,
41 std::chrono::milliseconds max_age = std::chrono::milliseconds(5000)) const {
42 if (data_hash != current_hash || data_size != current_size) {
43 return false;
44 }
45
46 auto now = std::chrono::steady_clock::now();
47 return (now - timestamp) < max_age;
48 }
49};

100 template <typename T>
101 size_t calculate_hash(const TimeSeries<T>& series) const {
102 std::hash<T> hasher;
103 size_t seed = series.size();
104
105 // Hash a sample of values for large series (for performance)
106 size_t step = std::max(size_t{1}, series.size() / 100);
107
108 for (size_t i = 0; i < series.size(); i += step) {
109 seed ^= hasher(series[i]) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
110 }
111
112 // Also hash first and last few values for better collision detection
113 size_t sample_size = std::min(size_t{10}, series.size());
114 for (size_t i = 0; i < sample_size; ++i) {
115 seed ^= hasher(series[i]) + 0x9e3779b97f4a7c15ULL + (seed << 6) + (seed >> 2);
116 }
117
118 if (series.size() > sample_size) {
119 for (size_t i = series.size() - sample_size; i < series.size(); ++i) {
120 seed ^= hasher(series[i]) + 0x9e3779b97f4a7c15ULL + (seed << 6) + (seed >> 2);
121 }
122 }
123
124 return seed;
125 }

130 template <typename... Args>
131 std::string make_cache_key(const std::string& function_name, Args&&... args) const {
132 std::string key = function_name;
133
134 auto append_arg = [&key](const auto& arg) {
135 if constexpr (std::is_arithmetic_v<std::decay_t<decltype(arg)>>) {
136 key += "_" + std::to_string(arg);
137 } else if constexpr (std::is_same_v<std::decay_t<decltype(arg)>, std::string>) {
138 key += "_" + arg;
139 }
140 };
141
142 (append_arg(args), ...);
143
144 return key;

150 void check_and_cleanup_cache() const {
151 auto now = std::chrono::steady_clock::now();
152 if (!config_.enable_auto_cleanup || (now - last_cleanup_) < config_.cleanup_interval) {
153 return;
154 }
155
156 std::unique_lock lock(cache_mutex_);
157
158 // Clean up expired entries
159 auto cleanup_map = [&](auto& cache_map) {
160 for (auto it = cache_map.begin(); it != cache_map.end();) {
161 if ((now - it->second.timestamp) > config_.max_age) {
162 it = cache_map.erase(it);
163 } else {
164 ++it;
165 }
166 }
167 };
168
169 cleanup_map(scalar_cache_);
170 cleanup_map(metrics_cache_);
171 cleanup_map(series_cache_);
172
173 // If cache is too large, remove oldest entries
174 auto trim_cache = [&](auto& cache_map) {
175 if (cache_map.size() > config_.max_entries) {
176 std::vector<std::pair<std::string, std::chrono::steady_clock::time_point>> entries;
177 for (const auto& [key, entry] : cache_map) {
178 entries.emplace_back(key, entry.timestamp);
179 }
180
181 std::sort(entries.begin(), entries.end(),
182 [](const auto& a, const auto& b) { return a.second < b.second; });
183
184 size_t to_remove = cache_map.size() - config_.max_entries + 100; // Remove extra for hysteresis
185 for (size_t i = 0; i < std::min(to_remove, entries.size()); ++i) {
186 cache_map.erase(entries[i].first);
187 }
188 }
189 };
190
191 trim_cache(scalar_cache_);
192 trim_cache(metrics_cache_);
193 trim_cache(series_cache_);
194
195 last_cleanup_ = now;
196 }

202 std::optional<T> try_get_cached(const std::string& key, size_t data_hash, size_t data_size, auto& cache_map) const {
203 std::shared_lock lock(cache_mutex_);
204
205 auto it = cache_map.find(key);
206 if (it != cache_map.end() && it->second.is_valid(data_hash, data_size, config_.max_age)) {

443 template <typename T>
444 [[nodiscard]] Result<double> sharpe_ratio(const TimeSeries<T>& returns, double risk_free_rate = 0.02) const {
445 check_and_cleanup_cache();
446
447 size_t data_hash = calculate_hash(returns);
448 std::string cache_key = make_cache_key("sharpe", data_hash, static_cast<int>(risk_free_rate * 10000));
449
450 if (auto cached = try_get_cached<double>(cache_key, data_hash, returns.size(), scalar_cache_)) {
451 return Result<double>::success(*cached);
452 }
453
454 auto start_time = std::chrono::high_resolution_clock::now();
455
456 // Calculate Sharpe ratio manually for caching individual components
457 auto mean_result = mean(returns); // This will use cached mean if available
458 if (mean_result.is_error()) {
459 return mean_result;
460 }
461
462 auto std_result = std_deviation(returns); // This will use cached std if available
463 if (std_result.is_error()) {
464 return std_result;
465 }
466
467 double excess_return = mean_result.value() - risk_free_rate / 252.0; // Daily risk-free rate
468 double sharpe = excess_return / std_result.value();
469
470 auto end_time = std::chrono::high_resolution_clock::now();
471
472 if ((end_time - start_time) >= config_.min_computation_time_basic) {
473 store_in_cache(cache_key, sharpe, data_hash, returns.size(), scalar_cache_);
474 }
475
476 return Result<double>::success(sharpe);
477 }

482 template <typename T>
483 [[nodiscard]] Result<double> max_drawdown(const TimeSeries<T>& prices) const {
484 check_and_cleanup_cache();
485
486 size_t data_hash = calculate_hash(prices);
487 std::string cache_key = make_cache_key("max_drawdown", data_hash);
488
489 if (auto cached = try_get_cached<double>(cache_key, data_hash, prices.size(), scalar_cache_)) {
490 return Result<double>::success(*cached);
491 }
492
493 auto start_time = std::chrono::high_resolution_clock::now();
494
495 // Calculate maximum drawdown
496 if (prices.empty()) {
497 return Result<double>::error(ErrorCode::InvalidInput, "Empty price series");
498 }
499
500 double max_price = prices[0];
501 double max_dd = 0.0;
502
503 for (size_t i = 1; i < prices.size(); ++i) {
504 max_price = std::max(max_price, static_cast<double>(prices[i]));
505 double drawdown = (max_price - static_cast<double>(prices[i])) / max_price;
506 max_dd = std::max(max_dd, drawdown);
507 }
508
509 auto end_time = std::chrono::high_resolution_clock::now();
510
511 if ((end_time - start_time) >= config_.min_computation_time_basic) {
512 store_in_cache(cache_key, max_dd, data_hash, prices.size(), scalar_cache_);
513 }
514
515 return Result<double>::success(max_dd);
516 }
