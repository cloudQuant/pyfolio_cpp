198 template <typename T>
199 static Result<TimeSeries<T>> parse_time_series(const json& data) {
200 try {
201 if (!data.contains("data") || !data["data"].is_array()) {
202 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput,
203 "Invalid JSON: missing or invalid 'data' field");
204 }
205
206 std::vector<DateTime> timestamps;
207 std::vector<T> values;
208
209 for (const auto& point : data["data"]) {
210 if (!point.contains("timestamp") || !point.contains("value")) {
211 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput,
212 "Invalid data point: missing timestamp or value");
213 }
214
215 auto dt_result = DateTime::parse(point["timestamp"].get<std::string>());
216 if (dt_result.is_error()) {
217 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput, "Invalid timestamp format");
218 }
219
220 timestamps.push_back(dt_result.value());
221 values.push_back(point["value"].get<T>());
222 }
223
224 std::string name = data.value("name", "series");
225 return TimeSeries<T>::create(timestamps, values, name);
226
227 } catch (const std::exception& e) {
228 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput,
229 std::string("JSON parsing error: ") + e.what());
230 }
231 }

236 static Result<transactions::TransactionRecord> parse_transaction_record(const json& data) {
237 try {
238 if (!data.contains("symbol") || !data.contains("timestamp") || !data.contains("shares") ||
239 !data.contains("price")) {
240 return Result<transactions::TransactionRecord>::error(ErrorCode::InvalidInput,
241 "Missing required transaction fields");
242 }
243
244 std::string symbol = data["symbol"].get<std::string>();
245
246 auto dt_result = DateTime::parse(data["timestamp"].get<std::string>());
247 if (dt_result.is_error()) {
248 return Result<transactions::TransactionRecord>::error(ErrorCode::InvalidInput,
249 "Invalid timestamp format");
250 }
251
252 double shares = data["shares"].get<double>();
253 double price = data["price"].get<double>();
254
255 // Determine transaction type
256 transactions::TransactionType type =
257 shares > 0 ? transactions::TransactionType::Buy : transactions::TransactionType::Sell;
258
259 if (data.contains("type")) {
260 std::string type_str = data["type"].get<std::string>();
261 if (type_str == "sell") {
262 type = transactions::TransactionType::Sell;
263 } else {
264 type = transactions::TransactionType::Buy;
265 }
266 }
267
268 std::string currency = data.value("currency", "USD");
269 double commission = data.value("commission", 0.0);
270 double slippage = data.value("slippage", 0.0);
271
272 // Create transaction record
273 transactions::TransactionRecord txn(symbol, shares, price, dt_result.value(), type, currency, commission,
274 slippage);
275
276 if (data.contains("exchange")) {
277 txn.set_exchange(data["exchange"].get<std::string>());
278 }
279
280 if (data.contains("order_id")) {
281 txn.set_order_id(data["order_id"].get<std::string>());
282 }
283
284 return Result<transactions::TransactionRecord>::success(txn);
285
286 } catch (const std::exception& e) {
287 return Result<transactions::TransactionRecord>::error(ErrorCode::InvalidInput,
288 std::string("JSON parsing error: ") + e.what());
289 }
290 }
