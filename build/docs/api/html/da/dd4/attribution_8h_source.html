102 Result<AttributionResult> calculate_period_attribution(const positions::PortfolioHoldings& portfolio_start,
103 const positions::PortfolioHoldings& portfolio_end,
104 const std::map<Symbol, double>& benchmark_weights,
105 const std::map<Symbol, double>& security_returns) const {
106 if (portfolio_start.holdings().empty() || portfolio_end.holdings().empty()) {
107 return Result<AttributionResult>::error(ErrorCode::InsufficientData, "Portfolio holdings cannot be empty");
108 }
109
110 AttributionResult result{};
111
112 // Calculate portfolio return
113 double portfolio_start_value = portfolio_start.total_value();
114 double portfolio_end_value = portfolio_end.total_value();
115
116 if (portfolio_start_value <= 0) {
117 return Result<AttributionResult>::error(ErrorCode::InvalidInput, "Portfolio start value must be positive");
118 }
119
120 result.portfolio_return = (portfolio_end_value - portfolio_start_value) / portfolio_start_value;
121
122 // Calculate benchmark return
123 result.benchmark_return = 0.0;
124 double total_benchmark_weight = 0.0;
125
126 for (const auto& [symbol, weight] : benchmark_weights) {
127 auto return_it = security_returns.find(symbol);
128 if (return_it != security_returns.end()) {
129 result.benchmark_return += weight * return_it->second;
130 total_benchmark_weight += weight;
131 }
132 }
133
134 if (total_benchmark_weight == 0.0) {
135 return Result<AttributionResult>::error(ErrorCode::InvalidInput, "Benchmark weights sum to zero");
136 }
137
138 result.benchmark_return /= total_benchmark_weight;
139 result.active_return = result.portfolio_return - result.benchmark_return;
140
141 // Calculate attribution effects
142 auto attribution_effects = calculate_attribution_effects(portfolio_start, benchmark_weights, security_returns);
143
144 if (attribution_effects.is_error()) {
145 return Result<AttributionResult>::error(attribution_effects.error().code,
146 attribution_effects.error().message);
147 }
148
149 const auto& effects = attribution_effects.value();
150 result.allocation_effect = effects.allocation_effect;
151 result.selection_effect = effects.selection_effect;
152 result.interaction_effect = effects.interaction_effect;
153 result.total_effect = result.allocation_effect + result.selection_effect + result.interaction_effect;
154
155 return Result<AttributionResult>::success(result);
156 }

161 Result<std::vector<SectorAttribution>> calculate_sector_attribution(
162 const positions::PortfolioHoldings& portfolio, const std::map<Symbol, double>& benchmark_weights,
163 const std::map<Symbol, double>& security_returns) const {
164 // Aggregate portfolio and benchmark by sector
165 std::map<std::string, double> portfolio_sector_weights;
166 std::map<std::string, double> benchmark_sector_weights;
167 std::map<std::string, double> sector_returns;
168 std::map<std::string, std::vector<Symbol>> sector_symbols;
169
170 // Portfolio sector weights
171 for (const auto& [symbol, holding] : portfolio.holdings()) {
172 std::string sector = get_symbol_sector(symbol);
173 portfolio_sector_weights[sector] += holding.weight;
174 sector_symbols[sector].push_back(symbol);
175 }
176
177 // Benchmark sector weights and returns
178 for (const auto& [symbol, weight] : benchmark_weights) {
179 std::string sector = get_symbol_sector(symbol);
180 benchmark_sector_weights[sector] += weight;
181
182 auto return_it = security_returns.find(symbol);
183 if (return_it != security_returns.end()) {
184 // Weight by benchmark weight within sector
185 sector_returns[sector] += weight * return_it->second;
186 }
187 }
188
189 // Normalize sector returns by sector benchmark weights
190 for (auto& [sector, ret] : sector_returns) {
191 if (benchmark_sector_weights[sector] > 0) {
192 ret /= benchmark_sector_weights[sector];
193 }
194 }
195
196 // Calculate portfolio sector returns
197 std::map<std::string, double> portfolio_sector_returns;
198 for (const auto& [sector, symbols] : sector_symbols) {
199 double sector_return = 0.0;
200 double sector_weight = 0.0;
201
202 for (const auto& symbol : symbols) {
203 auto holding_result = portfolio.get_holding(symbol);
204 auto return_it = security_returns.find(symbol);
205
206 if (holding_result.is_ok() && return_it != security_returns.end()) {
207 double weight = holding_result.value().weight;
208 sector_return += weight * return_it->second;
209 sector_weight += weight;
210 }
211 }
212
213 if (sector_weight > 0) {
214 portfolio_sector_returns[sector] = sector_return / sector_weight;
215 }
216 }
217
218 // Calculate attribution for each sector
219 std::vector<SectorAttribution> results;
220 std::set<std::string> all_sectors;
221
222 for (const auto& [sector, weight] : portfolio_sector_weights) {
223 all_sectors.insert(sector);
224 }
225 for (const auto& [sector, weight] : benchmark_sector_weights) {
226 all_sectors.insert(sector);
227 }
228
229 for (const std::string& sector : all_sectors) {
230 SectorAttribution attr;
231 attr.sector = sector;
232 attr.portfolio_weight = portfolio_sector_weights[sector];
233 attr.benchmark_weight = benchmark_sector_weights[sector];
234 attr.portfolio_return = portfolio_sector_returns[sector];
235 attr.benchmark_return = sector_returns[sector];
236
237 // Brinson attribution formulas
238 double weight_diff = attr.portfolio_weight - attr.benchmark_weight;
239 double return_diff = attr.portfolio_return - attr.benchmark_return;
240
241 attr.allocation_effect = weight_diff * attr.benchmark_return;
242 attr.selection_effect = attr.benchmark_weight * return_diff;
243 attr.interaction_effect = weight_diff * return_diff;
244 attr.total_contribution = attr.allocation_effect + attr.selection_effect + attr.interaction_effect;
245
246 results.push_back(attr);
247 }
248
249 // Sort by absolute total contribution
250 std::sort(results.begin(), results.end(), [](const SectorAttribution& a, const SectorAttribution& b) {
251 return std::abs(a.total_contribution) > std::abs(b.total_contribution);
252 });
253
254 return Result<std::vector<SectorAttribution>>::success(std::move(results));
255 }

260 Result<DataFrame> calculate_multi_period_attribution(
261 const positions::HoldingsSeries& holdings_series,
262 const std::map<DateTime, std::map<Symbol, double>>& benchmark_weights_series,
263 const std::map<Symbol, TimeSeries<double>>& return_series) const {
264 if (holdings_series.size() < 2) {
265 return Result<DataFrame>::error(ErrorCode::InsufficientData, "Need at least 2 holdings snapshots");
266 }
267
268 std::vector<DateTime> dates;
269 std::vector<double> portfolio_returns;
270 std::vector<double> benchmark_returns;
271 std::vector<double> active_returns;
272 std::vector<double> allocation_effects;
273 std::vector<double> selection_effects;
274 std::vector<double> interaction_effects;
275
276 for (size_t i = 1; i < holdings_series.size(); ++i) {
277 const auto& start_holdings = holdings_series[i - 1];
278 const auto& end_holdings = holdings_series[i];
279
280 DateTime period_end = end_holdings.timestamp();
281 dates.push_back(period_end);
282
283 // Get benchmark weights for this period
284 auto bench_it = benchmark_weights_series.find(period_end);
285 if (bench_it == benchmark_weights_series.end()) {
286 continue; // Skip period if no benchmark data
287 }
288
289 // Calculate security returns for this period
290 std::map<Symbol, double> period_returns;
291 DateTime period_start = start_holdings.timestamp();
292
293 for (const auto& [symbol, ts] : return_series) {
294 auto start_result = ts.at_time(period_start);
295 auto end_result = ts.at_time(period_end);
296
297 if (start_result.is_ok() && end_result.is_ok()) {
298 double start_price = start_result.value();
299 double end_price = end_result.value();
300 if (start_price > 0) {
301 period_returns[symbol] = (end_price - start_price) / start_price;
302 }
303 }
304 }
305
306 // Calculate attribution for this period
307 auto attribution_result =
308 calculate_period_attribution(start_holdings, end_holdings, bench_it->second, period_returns);
309
310 if (attribution_result.is_ok()) {
311 const auto& attr = attribution_result.value();
312 portfolio_returns.push_back(attr.portfolio_return);
313 benchmark_returns.push_back(attr.benchmark_return);
314 active_returns.push_back(attr.active_return);
315 allocation_effects.push_back(attr.allocation_effect);
316 selection_effects.push_back(attr.selection_effect);
317 interaction_effects.push_back(attr.interaction_effect);
318 } else {
319 // Fill with zeros if calculation fails
320 portfolio_returns.push_back(0.0);
321 benchmark_returns.push_back(0.0);
322 active_returns.push_back(0.0);
323 allocation_effects.push_back(0.0);
324 selection_effects.push_back(0.0);
325 interaction_effects.push_back(0.0);
326 }
327 }
328
329 DataFrame df{dates};
330 df.add_column("portfolio_return", portfolio_returns);
331 df.add_column("benchmark_return", benchmark_returns);
332 df.add_column("active_return", active_returns);
333 df.add_column("allocation_effect", allocation_effects);
334 df.add_column("selection_effect", selection_effects);
335 df.add_column("interaction_effect", interaction_effects);
336
337 return Result<DataFrame>::success(std::move(df));
338 }

352 Result<AttributionEffects> calculate_attribution_effects(const positions::PortfolioHoldings& portfolio,
353 const std::map<Symbol, double>& benchmark_weights,
354 const std::map<Symbol, double>& security_returns) const {
355 AttributionEffects effects{0.0, 0.0, 0.0};
356
357 std::set<Symbol> all_symbols;
358 for (const auto& [symbol, holding] : portfolio.holdings()) {
359 all_symbols.insert(symbol);
360 }
361 for (const auto& [symbol, weight] : benchmark_weights) {
362 all_symbols.insert(symbol);
363 }
364
365 for (const Symbol& symbol : all_symbols) {
366 double portfolio_weight = 0.0;
367 auto holding_result = portfolio.get_holding(symbol);
368 if (holding_result.is_ok()) {
369 portfolio_weight = holding_result.value().weight;
370 }
371
372 double benchmark_weight = 0.0;
373 auto bench_it = benchmark_weights.find(symbol);
374 if (bench_it != benchmark_weights.end()) {
375 benchmark_weight = bench_it->second;
376 }
377
378 double security_return = 0.0;
379 auto return_it = security_returns.find(symbol);
380 if (return_it != security_returns.end()) {
381 security_return = return_it->second;
382 }
383
384 // Brinson attribution formulas
385 double weight_diff = portfolio_weight - benchmark_weight;
386
387 effects.allocation_effect += weight_diff * security_return;
388 effects.selection_effect +=
389 benchmark_weight * security_return * (portfolio_weight > 0 ? 1.0 : 0.0); // Simplified
390 effects.interaction_effect +=
391 weight_diff * security_return * (portfolio_weight > 0 ? 1.0 : 0.0); // Simplified
392 }
393
394 return Result<AttributionEffects>::success(effects);
395 }

417 Result<AlphaBetaResult> calculate(const ReturnSeries& portfolio_returns, const ReturnSeries& benchmark_returns,
418 double risk_free_rate = constants::DEFAULT_RISK_FREE_RATE) const {
419 if (portfolio_returns.size() != benchmark_returns.size()) {
420 return Result<AlphaBetaResult>::error(ErrorCode::InvalidInput,
421 "Portfolio and benchmark returns must have same length");
422 }
423
424 if (portfolio_returns.size() < 3) {
425 return Result<AlphaBetaResult>::error(ErrorCode::InsufficientData,
426 "Need at least 3 observations for regression");
427 }
428
429 // Convert to excess returns
430 auto portfolio_excess_result =
431 pyfolio::performance::calculate_excess_returns(portfolio_returns, risk_free_rate);
432 auto benchmark_excess_result =
433 pyfolio::performance::calculate_excess_returns(benchmark_returns, risk_free_rate);
434
435 if (portfolio_excess_result.is_error() || benchmark_excess_result.is_error()) {
436 return Result<AlphaBetaResult>::error(ErrorCode::CalculationError, "Failed to calculate excess returns");
437 }
438
439 const auto& port_excess = portfolio_excess_result.value();
440 const auto& bench_excess = benchmark_excess_result.value();
441
442 // Linear regression: portfolio_excess = alpha + beta * benchmark_excess
443 auto regression_result = linear_regression(bench_excess.values(), port_excess.values());
444 if (regression_result.is_error()) {
445 return regression_result;
446 }
447
448 AlphaBetaResult result = regression_result.value();
449
450 // Calculate additional metrics
451 std::vector<double> active_returns;
452 for (size_t i = 0; i < portfolio_returns.size(); ++i) {
453 active_returns.push_back(portfolio_returns[i] - benchmark_returns[i]);
454 }
455
456 auto mean_active_result = pyfolio::stats::mean(std::span<const double>{active_returns});
457 auto std_active_result = pyfolio::stats::standard_deviation(std::span<const double>{active_returns});
458
459 if (mean_active_result.is_ok() && std_active_result.is_ok()) {
460 result.active_return = mean_active_result.value();
461 result.tracking_error = std_active_result.value();
462
463 if (result.tracking_error > 0) {
464 result.information_ratio = result.active_return / result.tracking_error;
465 }
466 }
467
468 // Risk decomposition
469 auto bench_vol_result = pyfolio::stats::standard_deviation(std::span<const Return>{benchmark_returns.values()});
470 if (bench_vol_result.is_ok()) {
471 double benchmark_vol = bench_vol_result.value();
472 result.systematic_risk = std::abs(result.beta) * benchmark_vol;
473
474 auto port_vol_result =
475 pyfolio::stats::standard_deviation(std::span<const Return>{portfolio_returns.values()});
476 if (port_vol_result.is_ok()) {
477 double portfolio_vol = port_vol_result.value();
478 result.specific_risk = std::sqrt(
479 std::max(0.0, portfolio_vol * portfolio_vol - result.systematic_risk * result.systematic_risk));
480 }
481 }
482
483 return Result<AlphaBetaResult>::success(result);
484 }
485
486 private:
487 Result<AlphaBetaResult> linear_regression(const std::vector<Return>& x, const std::vector<Return>& y) const {
488 if (x.size() != y.size() || x.size() < 2) {
489 return Result<AlphaBetaResult>::error(ErrorCode::InvalidInput, "Invalid data for regression");
490 }
491
492 size_t n = x.size();
493 double sum_x = std::accumulate(x.begin(), x.end(), 0.0);
494 double sum_y = std::accumulate(y.begin(), y.end(), 0.0);
495 double sum_xx = std::inner_product(x.begin(), x.end(), x.begin(), 0.0);
496 double sum_xy = std::inner_product(x.begin(), x.end(), y.begin(), 0.0);
497 double sum_yy = std::inner_product(y.begin(), y.end(), y.begin(), 0.0);
498
499 double mean_x = sum_x / n;
500 double mean_y = sum_y / n;
501
502 double sxx = sum_xx - n * mean_x * mean_x;
503 double sxy = sum_xy - n * mean_x * mean_y;
504 double syy = sum_yy - n * mean_y * mean_y;
505
506 if (sxx == 0.0) {
507 return Result<AlphaBetaResult>::error(ErrorCode::DivisionByZero, "Zero variance in benchmark returns");
508 }
509
510 AlphaBetaResult result{};
511 result.beta = sxy / sxx;
512 result.alpha = mean_y - result.beta * mean_x;
513
514 // Calculate R-squared
515 if (syy > 0.0) {
516 result.r_squared = (sxy * sxy) / (sxx * syy);
517 }
518
519 return Result<AlphaBetaResult>::success(result);
520 }
