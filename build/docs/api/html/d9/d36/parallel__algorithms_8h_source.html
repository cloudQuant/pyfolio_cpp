48class ThreadPool {
49 private:
50 std::vector<std::thread> workers_;
51 std::queue<std::function<void()>> tasks_;
52 std::mutex queue_mutex_;
53 std::condition_variable condition_;
54 bool stop_ = false;
55
56 public:
57 explicit ThreadPool(size_t num_threads = std::thread::hardware_concurrency()) {
58 for (size_t i = 0; i < num_threads; ++i) {
59 workers_.emplace_back([this] {
60 for (;;) {
61 std::function<void()> task;
62
63 {
64 std::unique_lock<std::mutex> lock(queue_mutex_);
65 condition_.wait(lock, [this] { return stop_ || !tasks_.empty(); });
66
67 if (stop_ && tasks_.empty())
68 return;
69
70 task = std::move(tasks_.front());
71 tasks_.pop();
72 }
73
74 task();
75 }
76 });
77 }
78 }
79
80 ~ThreadPool() {
81 {
82 std::unique_lock<std::mutex> lock(queue_mutex_);
83 stop_ = true;
84 }
85
86 condition_.notify_all();
87
88 for (std::thread& worker : workers_) {
89 worker.join();
90 }
91 }

93 template <typename F, typename... Args>
94 auto enqueue(F&& f, Args&&... args) -> std::future<typename std::invoke_result<F, Args...>::type> {
95 using return_type = typename std::invoke_result<F, Args...>::type;
96
97 auto task = std::make_shared<std::packaged_task<return_type()>>(
98 std::bind(std::forward<F>(f), std::forward<Args>(args)...));
99
100 std::future<return_type> result = task->get_future();
101
102 {
103 std::unique_lock<std::mutex> lock(queue_mutex_);
104
105 if (stop_) {
106 throw std::runtime_error("enqueue on stopped ThreadPool");
107 }
108
109 tasks_.emplace([task]() { (*task)(); });
110 }
111
112 condition_.notify_one();
113 return result;
114 }
115
116 size_t size() const { return workers_.size(); }
117};

137 size_t calculate_chunk_size(size_t total_size, size_t num_threads) const {
138 if (!config_.adaptive_chunking) {
139 return std::max(config_.min_chunk_size, total_size / num_threads);
140 }
141
142 // Adaptive chunking based on data size and thread count
143 size_t base_chunk = total_size / (num_threads * config_.chunk_size_factor);
144 size_t optimal_chunk = std::max(config_.min_chunk_size, base_chunk);
145
146 // Ensure chunk size is reasonable for cache efficiency
147 return std::min(optimal_chunk, total_size / 2);
148 }

163 template <typename T, typename UnaryOp>
164 [[nodiscard]] Result<std::vector<T>> parallel_map(const std::vector<T>& input, UnaryOp op) const {
165 if (input.empty()) {
166 return Result<std::vector<T>>::success(std::vector<T>{});
167 }
168
169 if (!should_use_parallel(input.size())) {
170 // Serial execution for small datasets
171 std::vector<T> result;
172 result.reserve(input.size());
173
174 for (const auto& item : input) {
175 result.push_back(op(item));
176 }
177
178 return Result<std::vector<T>>::success(std::move(result));
179 }
180
181 // Parallel execution
182 const size_t num_threads = std::min(config_.max_threads, input.size());
183 const size_t chunk_size = calculate_chunk_size(input.size(), num_threads);
184
185 std::vector<T> result(input.size());
186 std::vector<std::future<void>> futures;
187
188 auto& pool = get_global_thread_pool();
189
190 for (size_t i = 0; i < input.size(); i += chunk_size) {
191 size_t end = std::min(i + chunk_size, input.size());
192
193 futures.push_back(pool.enqueue([&, i, end]() {
194 for (size_t j = i; j < end; ++j) {
195 result[j] = op(input[j]);
196 }
197 }));
198 }
199
200 // Wait for all tasks to complete
201 for (auto& future : futures) {
202 future.wait();
203 }
204
205 return Result<std::vector<T>>::success(std::move(result));
206 }

211 template <typename T, typename BinaryOp>
212 [[nodiscard]] Result<T> parallel_reduce(const std::vector<T>& input, T init, BinaryOp op) const {
213 if (input.empty()) {
214 return Result<T>::success(init);
215 }
216
217 if (!should_use_parallel(input.size())) {
218 // Serial execution
219 return Result<T>::success(std::accumulate(input.begin(), input.end(), init, op));
220 }
221
222 // Parallel execution
223 const size_t num_threads = std::min(config_.max_threads, input.size());
224 const size_t chunk_size = calculate_chunk_size(input.size(), num_threads);
225
226 std::vector<std::future<T>> futures;
227 auto& pool = get_global_thread_pool();
228
229 for (size_t i = 0; i < input.size(); i += chunk_size) {
230 size_t end = std::min(i + chunk_size, input.size());
231
232 futures.push_back(pool.enqueue([&, i, end]() -> T {
233 T local_result = (i == 0) ? init : T{};
234
235 for (size_t j = i; j < end; ++j) {
236 if (i == 0 && j == i) {
237 local_result = op(local_result, input[j]);
238 } else {
239 local_result = (j == i) ? input[j] : op(local_result, input[j]);
240 }
241 }
242
243 return local_result;
244 }));
245 }
246
247 // Combine results from all threads
248 T final_result = init;
249 bool first = true;
250
251 for (auto& future : futures) {
252 T partial_result = future.get();
253 if (first && !futures.empty()) {
254 final_result = partial_result;
255 first = false;
256 } else {
257 final_result = op(final_result, partial_result);
258 }
259 }
260
261 return Result<T>::success(final_result);
262 }

268 [[nodiscard]] Result<double> parallel_mean(const TimeSeries<T>& series) const {
269 if (series.empty()) {
270 return Result<double>::error(ErrorCode::InvalidInput, "Empty time series");
271 }
272
273 const auto& values = series.values();
274
275 auto sum_result = parallel_reduce(values, 0.0, std::plus<double>());
276 if (sum_result.is_error()) {
277 return Result<double>::error(sum_result.error().code, sum_result.error().message);
278 }
279
280 return Result<double>::success(sum_result.value() / values.size());

286 template <typename T>
287 [[nodiscard]] Result<double> parallel_std_deviation(const TimeSeries<T>& series) const {
288 auto mean_result = parallel_mean(series);
289 if (mean_result.is_error()) {
290 return Result<double>::error(mean_result.error().code, mean_result.error().message);
291 }
292
293 double mean_val = mean_result.value();
294 const auto& values = series.values();
295
296 // Parallel variance calculation
297 auto variance_sum_result = parallel_reduce(values, 0.0, [mean_val](double acc, double val) {
298 double diff = val - mean_val;
299 return acc + diff * diff;
300 });
301
302 if (variance_sum_result.is_error()) {
303 return Result<double>::error(variance_sum_result.error().code, variance_sum_result.error().message);
304 }
305
306 double variance = variance_sum_result.value() / values.size();
307 return Result<double>::success(std::sqrt(variance));

313 template <typename T>
314 [[nodiscard]] Result<double> parallel_correlation(const TimeSeries<T>& series1,
315 const TimeSeries<T>& series2) const {
316 if (series1.size() != series2.size()) {
317 return Result<double>::error(ErrorCode::InvalidInput, "Series size mismatch");
318 }
319
320 if (series1.empty()) {
321 return Result<double>::error(ErrorCode::InvalidInput, "Empty time series");
322 }
323
324 // Calculate means in parallel
325 auto mean1_result = parallel_mean(series1);
326 auto mean2_result = parallel_mean(series2);
327
328 if (mean1_result.is_error())
329 return Result<double>::error(mean1_result.error().code, mean1_result.error().message);
330 if (mean2_result.is_error())
331 return Result<double>::error(mean2_result.error().code, mean2_result.error().message);
332
333 double mean1 = mean1_result.value();
334 double mean2 = mean2_result.value();
335
336 const auto& values1 = series1.values();
337 const auto& values2 = series2.values();
338
339 // Create combined data for parallel processing
340 std::vector<std::pair<double, double>> combined_data;
341 combined_data.reserve(values1.size());
342
343 for (size_t i = 0; i < values1.size(); ++i) {
344 combined_data.emplace_back(values1[i], values2[i]);
345 }
346
347 // Parallel covariance calculation
348 std::vector<double> covariance_terms;
349 covariance_terms.reserve(combined_data.size());
350
351 for (const auto& pair : combined_data) {
352 covariance_terms.push_back((pair.first - mean1) * (pair.second - mean2));
353 }
354
355 auto covariance_result = parallel_reduce(covariance_terms, 0.0, std::plus<double>());
356
357 // Parallel variance calculations
358 auto variance1_result = parallel_reduce(values1, 0.0, [mean1](double acc, double val) {
359 double diff = val - mean1;
360 return acc + diff * diff;
361 });
362
363 auto variance2_result = parallel_reduce(values2, 0.0, [mean2](double acc, double val) {
364 double diff = val - mean2;
365 return acc + diff * diff;
366 });
367
368 if (covariance_result.is_error())
369 return Result<double>::error(covariance_result.error().code, covariance_result.error().message);
370 if (variance1_result.is_error())
371 return Result<double>::error(variance1_result.error().code, variance1_result.error().message);
372 if (variance2_result.is_error())
373 return Result<double>::error(variance2_result.error().code, variance2_result.error().message);
374
375 double covariance = covariance_result.value() / values1.size();
376 double std1 = std::sqrt(variance1_result.value() / values1.size());
377 double std2 = std::sqrt(variance2_result.value() / values2.size());
378
379 if (std1 == 0.0 || std2 == 0.0) {
380 return Result<double>::success(0.0);
381 }
382
383 return Result<double>::success(covariance / (std1 * std2));
384 }

389 template <typename T, typename WindowOp>
390 [[nodiscard]] Result<TimeSeries<T>> parallel_rolling_operation(const TimeSeries<T>& series, size_t window_size,
391 WindowOp op) const {
392 if (window_size == 0 || window_size > series.size()) {
393 return Result<TimeSeries<T>>::error(ErrorCode::InvalidInput, "Invalid window size");
394 }
395
396 const size_t result_size = series.size() - window_size + 1;
397 const auto& timestamps = series.timestamps();
398 const auto& values = series.values();
399
400 if (!should_use_parallel(result_size)) {
401 // Serial execution for small datasets
402 std::vector<DateTime> result_dates;
403 std::vector<T> result_values;
404
405 result_dates.reserve(result_size);
406 result_values.reserve(result_size);
407
408 for (size_t i = 0; i < result_size; ++i) {
409 auto window_begin = values.begin() + i;
410 auto window_end = window_begin + window_size;
411
412 result_dates.push_back(timestamps[i + window_size - 1]);
413 result_values.push_back(op(window_begin, window_end));
414 }
415
416 return TimeSeries<T>::create(result_dates, result_values,
417 series.name() + "_rolling_" + std::to_string(window_size));
418 }
419
420 // Parallel execution
421 std::vector<DateTime> result_dates(result_size);
422 std::vector<T> result_values(result_size);
423
424 const size_t num_threads = std::min(config_.max_threads, result_size);
425 const size_t chunk_size = calculate_chunk_size(result_size, num_threads);
426
427 std::vector<std::future<void>> futures;
428 auto& pool = get_global_thread_pool();
429
430 for (size_t i = 0; i < result_size; i += chunk_size) {
431 size_t end = std::min(i + chunk_size, result_size);
432
433 futures.push_back(pool.enqueue([&, i, end]() {
434 for (size_t j = i; j < end; ++j) {
435 auto window_begin = values.begin() + j;
436 auto window_end = window_begin + window_size;
437
438 result_dates[j] = timestamps[j + window_size - 1];
439 result_values[j] = op(window_begin, window_end);
440 }
441 }));
442 }
443
444 // Wait for completion
445 for (auto& future : futures) {
446 future.wait();
447 }
448
449 return TimeSeries<T>::create(result_dates, result_values,
450 series.name() + "_rolling_" + std::to_string(window_size));
451 }

466 template <typename T>
467 [[nodiscard]] Result<TimeSeries<T>> parallel_rolling_std(const TimeSeries<T>& series, size_t window_size) const {
468 return parallel_rolling_operation(series, window_size, [](auto begin, auto end) -> T {
469 size_t n = std::distance(begin, end);
470 T mean = std::accumulate(begin, end, T{}) / static_cast<T>(n);
471
472 T variance = std::accumulate(begin, end, T{},
473 [mean](T acc, T val) {
474 T diff = val - mean;
475 return acc + diff * diff;
476 }) /
477 static_cast<T>(n);
478
479 return std::sqrt(variance);
