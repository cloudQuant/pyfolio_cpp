144 Result<VaRResult> calculate_historical_var(const ReturnSeries& returns, double confidence_level = 0.95,
145 VaRHorizon horizon = VaRHorizon::Daily) const {
146 if (returns.empty()) {
147 return Result<VaRResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
148 }
149
150 if (confidence_level <= 0.0 || confidence_level >= 1.0) {
151 return Result<VaRResult>::error(ErrorCode::InvalidInput, "Confidence level must be between 0 and 1");
152 }
153
154 const auto& values = returns.values();
155
156 // Calculate basic statistics
157 auto mean_result = stats::mean(std::span<const double>{values});
158 auto std_result = stats::standard_deviation(std::span<const double>{values});
159 auto skew_result = stats::skewness(std::span<const double>{values});
160 auto kurt_result = stats::kurtosis(std::span<const double>{values});
161
162 if (mean_result.is_error() || std_result.is_error()) {
163 return Result<VaRResult>::error(ErrorCode::CalculationError, "Failed to calculate basic statistics");
164 }
165
166 // Calculate VaR and CVaR
167 auto var_result = stats::value_at_risk(std::span<const double>{values}, confidence_level);
168 auto cvar_result = stats::conditional_value_at_risk(std::span<const double>{values}, confidence_level);
169
170 if (var_result.is_error() || cvar_result.is_error()) {
171 return Result<VaRResult>::error(ErrorCode::CalculationError, "Failed to calculate VaR or CVaR");
172 }
173
174 // Calculate maximum drawdown
175 double max_dd = 0.0;
176 double peak = 0.0;
177 double cumulative = 0.0;
178
179 for (double ret : values) {
180 cumulative += ret;
181 peak = std::max(peak, cumulative);
182 double drawdown = peak - cumulative;
183 max_dd = std::max(max_dd, drawdown);
184 }
185
186 VaRResult result;
187 result.var_estimate = var_result.value();
188 result.cvar_estimate = cvar_result.value();
189 result.confidence_level = confidence_level;
190 result.method = VaRMethod::Historical;
191 result.horizon = horizon;
192 result.sample_size = values.size();
193 result.portfolio_volatility = std_result.value();
194 result.skewness = skew_result.is_ok() ? skew_result.value() : 0.0;
195 result.kurtosis = kurt_result.is_ok() ? kurt_result.value() : 3.0;
196 result.max_drawdown = max_dd;
197
198 return Result<VaRResult>::success(result);
199 }

204 Result<VaRResult> calculate_parametric_var(const ReturnSeries& returns, double confidence_level = 0.95,
205 VaRHorizon horizon = VaRHorizon::Daily) const {
206 if (returns.empty()) {
207 return Result<VaRResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
208 }
209
210 const auto& values = returns.values();
211
212 auto mean_result = stats::mean(std::span<const double>{values});
213 auto std_result = stats::standard_deviation(std::span<const double>{values});
214
215 if (mean_result.is_error() || std_result.is_error()) {
216 return Result<VaRResult>::error(ErrorCode::CalculationError,
217 "Failed to calculate mean or standard deviation");
218 }
219
220 double mean = mean_result.value();
221 double std_dev = std_result.value();
222
223 // Normal distribution VaR
224 double z_score = stats::normal_ppf(1.0 - confidence_level);
225 double var_estimate = mean + z_score * std_dev;
226
227 // For parametric CVaR (normal distribution)
228 double phi_z = stats::normal_pdf(z_score);
229 double cvar_estimate = mean - std_dev * phi_z / (1.0 - confidence_level);
230
231 VaRResult result;
232 result.var_estimate = var_estimate;
233 result.cvar_estimate = cvar_estimate;
234 result.confidence_level = confidence_level;
235 result.method = VaRMethod::Parametric;
236 result.horizon = horizon;
237 result.sample_size = values.size();
238 result.portfolio_volatility = std_dev;
239
240 // Calculate skewness and kurtosis for diagnostics
241 auto skew_result = stats::skewness(std::span<const double>{values});
242 auto kurt_result = stats::kurtosis(std::span<const double>{values});
243 result.skewness = skew_result.is_ok() ? skew_result.value() : 0.0;
244 result.kurtosis = kurt_result.is_ok() ? kurt_result.value() : 3.0;
245
246 return Result<VaRResult>::success(result);
247 }

252 Result<VaRResult> calculate_cornish_fisher_var(const ReturnSeries& returns, double confidence_level = 0.95,
253 VaRHorizon horizon = VaRHorizon::Daily) const {
254 if (returns.empty()) {
255 return Result<VaRResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
256 }
257
258 const auto& values = returns.values();
259
260 auto mean_result = stats::mean(std::span<const double>{values});
261 auto std_result = stats::standard_deviation(std::span<const double>{values});
262 auto skew_result = stats::skewness(std::span<const double>{values});
263 auto kurt_result = stats::kurtosis(std::span<const double>{values});
264
265 if (mean_result.is_error() || std_result.is_error() || skew_result.is_error() || kurt_result.is_error()) {
266 return Result<VaRResult>::error(ErrorCode::CalculationError, "Failed to calculate required statistics");
267 }
268
269 double mean = mean_result.value();
270 double std_dev = std_result.value();
271 double skewness = skew_result.value();
272 double kurtosis = kurt_result.value();
273 double excess_kurtosis = kurtosis - 3.0;
274
275 // Normal quantile
276 double z = stats::normal_ppf(1.0 - confidence_level);
277
278 // Cornish-Fisher expansion
279 double cf_adjustment = (1.0 / 6.0) * skewness * (z * z - 1.0) +
280 (1.0 / 24.0) * excess_kurtosis * (z * z * z - 3.0 * z) -
281 (1.0 / 36.0) * skewness * skewness * (2.0 * z * z * z - 5.0 * z);
282
283 double z_cf = z + cf_adjustment;
284 double var_estimate = mean + z_cf * std_dev;
285
286 // Approximate CVaR using adjusted quantile
287 double phi_z_cf = stats::normal_pdf(z_cf);
288 double cvar_estimate = mean - std_dev * phi_z_cf / (1.0 - confidence_level);
289
290 VaRResult result;
291 result.var_estimate = var_estimate;
292 result.cvar_estimate = cvar_estimate;
293 result.confidence_level = confidence_level;
294 result.method = VaRMethod::CornishFisher;
295 result.horizon = horizon;
296 result.sample_size = values.size();
297 result.portfolio_volatility = std_dev;
298 result.skewness = skewness;
299 result.kurtosis = kurtosis;
300
301 return Result<VaRResult>::success(result);
302 }

308 VaRHorizon horizon = VaRHorizon::Daily,
309 size_t num_simulations = 10000) const {
310 if (returns.empty()) {
311 return Result<VaRResult>::error(ErrorCode::InsufficientData, "Return series cannot be empty");
312 }
313
314 const auto& values = returns.values();
315
316 auto mean_result = stats::mean(std::span<const double>{values});
317 auto std_result = stats::standard_deviation(std::span<const double>{values});
318
319 if (mean_result.is_error() || std_result.is_error()) {
320 return Result<VaRResult>::error(ErrorCode::CalculationError,
321 "Failed to calculate mean or standard deviation");
322 }
323
324 double mean = mean_result.value();
325 double std_dev = std_result.value();
326
327 // Generate Monte Carlo simulations
328 std::vector<double> simulated_returns;
329 simulated_returns.reserve(num_simulations);
330
331 std::normal_distribution<double> normal_dist(mean, std_dev);
332

335 }
336
337 // Calculate VaR and CVaR from simulated returns
338 auto var_result = stats::value_at_risk(std::span<const double>{simulated_returns}, confidence_level);
339 auto cvar_result =
340 stats::conditional_value_at_risk(std::span<const double>{simulated_returns}, confidence_level);
341
342 if (var_result.is_error() || cvar_result.is_error()) {
343 return Result<VaRResult>::error(ErrorCode::CalculationError,
344 "Failed to calculate VaR from Monte Carlo simulation");
345 }
346
347 VaRResult result;
348 result.var_estimate = var_result.value();
349 result.cvar_estimate = cvar_result.value();
350 result.confidence_level = confidence_level;
351 result.method = VaRMethod::MonteCarlo;
352 result.horizon = horizon;
353 result.sample_size = num_simulations;
354 result.portfolio_volatility = std_dev;
355
356 // Calculate skewness and kurtosis from original data
357 auto skew_result = stats::skewness(std::span<const double>{values});
358 auto kurt_result = stats::kurtosis(std::span<const double>{values});
359 result.skewness = skew_result.is_ok() ? skew_result.value() : 0.0;
360 result.kurtosis = kurt_result.is_ok() ? kurt_result.value() : 3.0;
361
362 return Result<VaRResult>::success(result);

368 Result<MarginalVaRResult> calculate_marginal_var(const std::map<Symbol, ReturnSeries>& asset_returns,
369 const std::map<Symbol, double>& portfolio_weights,
370 double confidence_level = 0.95) const {
371 if (asset_returns.empty() || portfolio_weights.empty()) {
372 return Result<MarginalVaRResult>::error(ErrorCode::InvalidInput,
373 "Asset returns and portfolio weights cannot be empty");
374 }
375
376 // Create portfolio return series
377 auto portfolio_returns_result = create_portfolio_return_series(asset_returns, portfolio_weights);
378 if (portfolio_returns_result.is_error()) {
379 return Result<MarginalVaRResult>::error(portfolio_returns_result.error().code,
380 portfolio_returns_result.error().message);
381 }
382
383 const auto& portfolio_returns = portfolio_returns_result.value();
384
385 // Calculate portfolio VaR
386 auto portfolio_var_result = calculate_historical_var(portfolio_returns, confidence_level);
387 if (portfolio_var_result.is_error()) {
388 return Result<MarginalVaRResult>::error(portfolio_var_result.error().code,
389 portfolio_var_result.error().message);
390 }
391
392 double portfolio_var = portfolio_var_result.value().var_estimate;
393
394 MarginalVaRResult result;
395 result.total_var = portfolio_var;
396
397 // Calculate marginal VaR for each asset
398 for (const auto& [symbol, weight] : portfolio_weights) {
399 // Create modified portfolio with small weight change
400 const double delta = 0.001; // 0.1% weight change
401 auto modified_weights = portfolio_weights;
402 modified_weights[symbol] += delta;
403
404 // Renormalize weights
405 double total_weight = 0.0;
406 for (auto& [sym, w] : modified_weights) {
407 total_weight += w;
408 }
409 for (auto& [sym, w] : modified_weights) {
410 w /= total_weight;
411 }
412
413 // Calculate VaR for modified portfolio
414 auto modified_returns_result = create_portfolio_return_series(asset_returns, modified_weights);
415 if (modified_returns_result.is_ok()) {
416 auto modified_var_result = calculate_historical_var(modified_returns_result.value(), confidence_level);
417 if (modified_var_result.is_ok()) {
418 double modified_var = modified_var_result.value().var_estimate;
419
420 // Marginal VaR = (VaR_new - VaR_old) / delta_weight
421 double marginal_var = (modified_var - portfolio_var) / delta;
422 result.marginal_var[symbol] = marginal_var;
423
424 // Component VaR = weight * marginal VaR
425 double component_var = weight * marginal_var;
426 result.component_var[symbol] = component_var;
427
428 // Percentage contribution
429 if (portfolio_var != 0.0) {
430 result.percentage_contribution[symbol] = (component_var / portfolio_var) * 100.0;
431 }
432 }
433 }
434 }
435
436 return Result<MarginalVaRResult>::success(std::move(result));
437 }

442 Result<std::map<std::string, VaRResult>> stress_test(const std::map<Symbol, ReturnSeries>& asset_returns,
443 const std::map<Symbol, double>& portfolio_weights,
444 const std::vector<StressTestScenario>& scenarios,
445 double confidence_level = 0.95) const {
446 std::map<std::string, VaRResult> stress_results;
447
448 for (const auto& scenario : scenarios) {
449 // Apply scenario shocks to each asset
450 std::map<Symbol, ReturnSeries> shocked_returns;
451
452 for (const auto& [symbol, returns] : asset_returns) {
453 auto shocked_values = scenario.apply_to_returns(returns.values(), symbol);
454 shocked_returns[symbol] = ReturnSeries{returns.timestamps(), shocked_values, symbol};
455 }
456
457 // Create stressed portfolio returns
458 auto stressed_portfolio_result = create_portfolio_return_series(shocked_returns, portfolio_weights);
459 if (stressed_portfolio_result.is_ok()) {
460 auto stressed_var_result =
461 calculate_historical_var(stressed_portfolio_result.value(), confidence_level);
462 if (stressed_var_result.is_ok()) {
463 stress_results[scenario.name] = stressed_var_result.value();
464 }
465 }
466 }
467
468 return Result<std::map<std::string, VaRResult>>::success(std::move(stress_results));
469 }

471 private:
475 Result<ReturnSeries> create_portfolio_return_series(const std::map<Symbol, ReturnSeries>& asset_returns,
476 const std::map<Symbol, double>& portfolio_weights) const {
477 if (asset_returns.empty()) {
478 return Result<ReturnSeries>::error(ErrorCode::InvalidInput, "Asset returns cannot be empty");
479 }
480
481 // Find common time period
482 size_t min_size = std::numeric_limits<size_t>::max();
483 for (const auto& [symbol, returns] : asset_returns) {
484 min_size = std::min(min_size, returns.size());
485 }
486
487 if (min_size == 0) {
488 return Result<ReturnSeries>::error(ErrorCode::InsufficientData, "No common return data available");
489 }
490
491 // Get timestamps from first series
492 const auto& first_series = asset_returns.begin()->second;
493 std::vector<DateTime> timestamps(first_series.timestamps().end() - min_size, first_series.timestamps().end());
494
495 // Calculate portfolio returns
496 std::vector<double> portfolio_returns;
497 portfolio_returns.reserve(min_size);
498
499 for (size_t i = 0; i < min_size; ++i) {
500 double portfolio_return = 0.0;
501
502 for (const auto& [symbol, weight] : portfolio_weights) {
503 auto asset_it = asset_returns.find(symbol);
504 if (asset_it != asset_returns.end()) {
505 const auto& asset_values = asset_it->second.values();
506 size_t asset_size = asset_values.size();
507 if (i < asset_size) {
508 size_t index = asset_size - min_size + i;
509 portfolio_return += weight * asset_values[index];
510 }
511 }
512 }
513
514 portfolio_returns.push_back(portfolio_return);
515 }
516
517 return Result<ReturnSeries>::success(ReturnSeries{timestamps, portfolio_returns, "Portfolio"});
518 }
