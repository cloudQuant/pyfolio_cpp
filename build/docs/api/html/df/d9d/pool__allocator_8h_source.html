62 public:
63 static constexpr size_t block_size = sizeof(T);
64 static constexpr size_t alignment = alignof(T);

97 T* allocate() {
98 std::lock_guard<std::mutex> lock(mutex_);
99
100 if (!free_list_) {
101 return nullptr; // Pool exhausted
102 }
103
104 Block* block = free_list_;
105 free_list_ = free_list_->next;
106
107 ++allocated_count_;
108 ++total_allocations_;
109
110 return reinterpret_cast<T*>(block);
111 }

117 if (!ptr)
118 return;

120 std::lock_guard<std::mutex> lock(mutex_);
121
122 Block* block = reinterpret_cast<Block*>(ptr);
123 block->next = free_list_;
124 free_list_ = block;
125
126 --allocated_count_;
127 }

134 const char* pool_start = memory_pool_.get();
135 const char* pool_end = pool_start + pool_size;

137 return char_ptr >= pool_start && char_ptr < pool_end;

143 PoolStats get_stats() const noexcept {
144 PoolStats stats{};

146 stats.used_bytes = allocated_count_.load() * block_size;
147 stats.free_bytes = (BlockCount - allocated_count_.load()) * block_size;
148 stats.num_allocations = total_allocations_.load();
149 stats.num_deallocations = total_allocations_.load() - allocated_count_.load();
150 stats.num_pools = 1;

167 private:
168 void initialize_free_list() {
169 char* ptr = memory_pool_.get();
170
171 for (size_t i = 0; i < BlockCount; ++i) {
172 Block* block = reinterpret_cast<Block*>(ptr + i * block_size);
173 block->next = (i == BlockCount - 1) ? nullptr : reinterpret_cast<Block*>(ptr + (i + 1) * block_size);
174 }

176 free_list_ = reinterpret_cast<Block*>(memory_pool_.get());

240 void* allocate(size_t size, size_t align = alignment) {
241 if (size == 0 || size > max_allocation_size) {
242 return nullptr;
243 }
244
245 // Align size
246 size_t aligned_size = align_size(size, align);
247 size_t total_size = sizeof(Block) + aligned_size;
248
249 std::lock_guard<std::mutex> lock(mutex_);
250
251 // Find suitable block
252 for (auto& pool : pools_) {
253 if (void* ptr = allocate_from_pool(*pool, total_size)) {
254 total_used_ += total_size;
255 ++num_allocations_;
256 return ptr;
257 }
258 }
259
260 // Need new pool
261 add_pool();
262 if (void* ptr = allocate_from_pool(*pools_.back(), total_size)) {
263 total_used_ += total_size;
264 ++num_allocations_;
265 return ptr;
266 }
267
268 return nullptr; // Failed to allocate
269 }
270
274 void deallocate(void* ptr) {
275 if (!ptr)
276 return;
277
278 std::lock_guard<std::mutex> lock(mutex_);
279
280 // Find which pool owns this pointer
281 for (auto& pool : pools_) {
282 if (owns_pointer(*pool, ptr)) {
283 deallocate_from_pool(*pool, ptr);
284 ++num_deallocations_;
285 return;
286 }
287 }
288
289 // Pointer not from our pools - this is an error
290 assert(false && "Attempting to deallocate pointer not from pool");
291 }

333 void* allocate_from_pool(Pool& pool, size_t size) {
334 Block** current = &pool.free_list;
335
336 while (*current) {
337 Block* block = *current;
338
339 if (block->size >= size) {
340 // Remove from free list
341 *current = block->next;
342
343 // Split block if it's much larger
344 if (block->size > size + sizeof(Block) + 32) {
345 Block* new_block = reinterpret_cast<Block*>(reinterpret_cast<char*>(block) + sizeof(Block) + size);
346 new_block->size = block->size - size - sizeof(Block);
347 new_block->next = *current;
348 *current = new_block;
349
350 block->size = size;
351 }
352
353 block->next = nullptr; // Mark as allocated
354 pool.used += sizeof(Block) + block->size;
355
356 return block->data();
357 }
358
359 current = &(block->next);
360 }
361
362 return nullptr;
363 }
364
365 void deallocate_from_pool(Pool& pool, void* ptr) {
366 Block* block = reinterpret_cast<Block*>(static_cast<char*>(ptr) - sizeof(Block));
367
368 // Add to free list
369 block->next = pool.free_list;
370 pool.free_list = block;
371
372 pool.used -= sizeof(Block) + block->size;
373 total_used_ -= sizeof(Block) + block->size;
374 }

378 const char* pool_start = pool.memory.get();
379 const char* pool_end = pool_start + pool.size;

381 return char_ptr >= pool_start && char_ptr < pool_end;

384 void coalesce_free_blocks(Pool& pool) {
385 // Sort free blocks by address
386 std::vector<Block*> free_blocks;
387 Block* current = pool.free_list;
388
389 while (current) {
390 free_blocks.push_back(current);
391 current = current->next;
392 }
393
394 if (free_blocks.size() < 2)
395 return;
396
397 std::sort(free_blocks.begin(), free_blocks.end());
398
399 // Coalesce adjacent blocks
400 pool.free_list = nullptr;
401 Block* prev = nullptr;
402
403 for (Block* block : free_blocks) {
404 if (prev &&
405 (reinterpret_cast<char*>(prev) + sizeof(Block) + prev->size == reinterpret_cast<char*>(block))) {
406 // Merge with previous block
407 prev->size += sizeof(Block) + block->size;
408 } else {
409 // Add to free list
410 if (prev) {
411 prev->next = block;
412 } else {
413 pool.free_list = block;
414 }
415 prev = block;
416 }
417 }
418
419 if (prev) {
420 prev->next = nullptr;
421 }
422 }
