113 template <typename T>
114 Result<void> calculate_risk_metrics(const TimeSeries<T>& returns, AnalysisReport& report) const {
115 // Calculate VaR and CVaR
116 auto sorted_returns = returns.values();
117 std::sort(sorted_returns.begin(), sorted_returns.end());
118
119 if (!sorted_returns.empty()) {
120 size_t var_index = static_cast<size_t>(sorted_returns.size() * 0.05);
121 report.var_95 = sorted_returns[var_index];
122
123 // CVaR is the mean of returns below VaR
124 double cvar_sum = 0.0;
125 for (size_t i = 0; i <= var_index; ++i) {
126 cvar_sum += sorted_returns[i];
127 }
128 report.cvar_95 = cvar_sum / (var_index + 1);
129 }
130
131 // Calculate downside deviation
132 auto mean_result = cache_calculator_->mean(returns);
133 if (mean_result.is_error())
134 return Result<void>::error(mean_result.error().code, mean_result.error().message);
135
136 double mean_return = mean_result.value();
137 double downside_sum = 0.0;
138 size_t downside_count = 0;
139
140 for (size_t i = 0; i < returns.size(); ++i) {
141 if (returns[i] < mean_return) {
142 double deviation = returns[i] - mean_return;
143 downside_sum += deviation * deviation;
144 ++downside_count;
145 }
146 }
147
148 report.downside_deviation = downside_count > 0 ? std::sqrt(downside_sum / downside_count) : 0.0;
149
150 return Result<void>::success();
151 }

156 template <typename T>
157 Result<void> calculate_rolling_metrics(const TimeSeries<T>& returns, AnalysisReport& report) const {
158 for (size_t window : config_.rolling_windows) {
159 if (window < returns.size()) {
160 // Use cached rolling calculations
161 auto rolling_mean_result = cache_calculator_->rolling_mean(returns, window);
162 if (rolling_mean_result.is_ok()) {
163 report.rolling_returns[window] = rolling_mean_result.value();
164 }
165
166 auto rolling_std_result = cache_calculator_->rolling_std(returns, window);
167 if (rolling_std_result.is_ok()) {
168 report.rolling_volatility[window] = rolling_std_result.value();
169
170 // Calculate rolling Sharpe ratio
171 const auto& vol_series = rolling_std_result.value();
172 const auto& ret_series = report.rolling_returns[window];
173
174 if (vol_series.size() == ret_series.size()) {
175 std::vector<DateTime> dates = vol_series.timestamps();
176 std::vector<double> sharpe_values;
177
178 double daily_rf = config_.risk_free_rate / config_.periods_per_year;
179
180 for (size_t i = 0; i < vol_series.size(); ++i) {
181 if (vol_series[i] > 0) {
182 double excess_return = ret_series[i] - daily_rf;
183 sharpe_values.push_back(excess_return / vol_series[i]);
184 } else {
185 sharpe_values.push_back(0.0);
186 }
187 }
188
189 report.rolling_sharpe[window] =
190 TimeSeries<double>(dates, sharpe_values, "rolling_sharpe_" + std::to_string(window));
191 }
192 }
193 }
194 }
195
196 return Result<void>::success();
197 }

202 void perform_risk_analysis(AnalysisReport& report) const {
203 report.passed_risk_checks = true;
204 report.warnings.clear();
205 report.recommendations.clear();
206
207 // Check Sharpe ratio
208 if (report.sharpe_ratio < config_.min_sharpe_threshold) {
209 report.passed_risk_checks = false;
210 report.warnings.push_back("Low Sharpe ratio: " + std::to_string(report.sharpe_ratio));
211 report.recommendations.push_back("Consider reducing volatility or improving return generation");
212 }
213
214 // Check maximum drawdown
215 if (report.max_drawdown > config_.max_drawdown_threshold) {
216 report.passed_risk_checks = false;
217 report.warnings.push_back("High maximum drawdown: " + std::to_string(report.max_drawdown * 100) + "%");
218 report.recommendations.push_back("Implement stronger risk controls or position sizing");
219 }
220
221 // Check annual return
222 if (report.annual_return < config_.min_return_threshold) {
223 report.warnings.push_back("Low annual return: " + std::to_string(report.annual_return * 100) + "%");
224 report.recommendations.push_back("Review strategy performance and consider optimization");
225 }
226
227 // Check volatility vs return relationship
228 double risk_adjusted_return = report.annual_return / report.annual_volatility;
229 if (risk_adjusted_return < 0.5) {
230 report.warnings.push_back("Poor risk-adjusted return ratio");
231 report.recommendations.push_back("Focus on reducing volatility while maintaining returns");
232 }
233
234 // Check skewness and kurtosis for tail risks
235 if (report.skewness < -1.0) {
236 report.warnings.push_back("Negative skewness indicates tail risk");
237 report.recommendations.push_back("Consider tail risk hedging strategies");
238 }
239
240 if (report.kurtosis > 5.0) {
241 report.warnings.push_back("High kurtosis indicates fat tail risk");
242 report.recommendations.push_back("Monitor for extreme events and adjust position sizing");
243 }
244 }

254 template <typename T>
255 [[nodiscard]] Result<AnalysisReport> analyze_performance(
256 const TimeSeries<T>& returns, const std::optional<TimeSeries<T>>& benchmark = std::nullopt) const {
257 auto start_time = std::chrono::high_resolution_clock::now();
258 AnalysisReport report{};
259 report.analysis_timestamp = std::chrono::steady_clock::now();
260
261 try {
262 // Basic performance metrics using cached calculations
263 auto mean_result = cache_calculator_->mean(returns);
264 if (mean_result.is_error())
265 return Result<AnalysisReport>::error(mean_result.error().code, mean_result.error().message);
266
267 auto std_result = cache_calculator_->std_deviation(returns);
268 if (std_result.is_error())
269 return Result<AnalysisReport>::error(std_result.error().code, std_result.error().message);
270
271 auto sharpe_result = cache_calculator_->sharpe_ratio(returns, config_.risk_free_rate);
272 if (sharpe_result.is_error())
273 return Result<AnalysisReport>::error(sharpe_result.error().code, sharpe_result.error().message);
274
275 // Calculate cumulative returns for total return
276 auto cumulative_result = returns.cumsum();
277 if (cumulative_result.is_error())
278 return Result<AnalysisReport>::error(cumulative_result.error().code, cumulative_result.error().message);
279
280 const auto& cum_returns = cumulative_result.value();
281 report.total_return = cum_returns.empty() ? 0.0 : cum_returns[cum_returns.size() - 1];
282
283 // Annualized metrics
284 report.annual_return = mean_result.value() * config_.periods_per_year;
285 report.annual_volatility = std_result.value() * std::sqrt(config_.periods_per_year);
286 report.sharpe_ratio = sharpe_result.value();
287
288 // Calculate maximum drawdown using cached function
289 auto max_dd_result = cache_calculator_->max_drawdown(cum_returns);
290 if (max_dd_result.is_error())
291 return Result<AnalysisReport>::error(max_dd_result.error().code, max_dd_result.error().message);
292 report.max_drawdown = max_dd_result.value();
293
294 // Calmar ratio
295 report.calmar_ratio = report.max_drawdown > 0 ? report.annual_return / report.max_drawdown : 0.0;
296
297 // Risk metrics
298 auto risk_result = calculate_risk_metrics(returns, report);
299 if (risk_result.is_error())
300 return Result<AnalysisReport>::error(risk_result.error().code, risk_result.error().message);
301
302 // Statistical moments
303 const auto& values = returns.values();
304 if (!values.empty()) {
305 double mean_val = mean_result.value();
306 double variance = 0.0, skew_sum = 0.0, kurt_sum = 0.0;
307
308 for (double val : values) {
309 double diff = val - mean_val;
310 double diff2 = diff * diff;
311 variance += diff2;
312 skew_sum += diff2 * diff;
313 kurt_sum += diff2 * diff2;
314 }
315
316 variance /= values.size();
317 double std_dev = std::sqrt(variance);
318
319 if (std_dev > 0) {
320 report.skewness = (skew_sum / values.size()) / (std_dev * std_dev * std_dev);
321 report.kurtosis = (kurt_sum / values.size()) / (variance * variance) - 3.0;
322 }
323 }
324
325 // Sortino ratio (downside deviation-adjusted)
326 report.sortino_ratio = report.downside_deviation > 0
327 ? (report.annual_return - config_.risk_free_rate) /
328 (report.downside_deviation * std::sqrt(config_.periods_per_year))
329 : 0.0;
330
331 // Rolling metrics
332 auto rolling_result = calculate_rolling_metrics(returns, report);
333 if (rolling_result.is_error())
334 return Result<AnalysisReport>::error(rolling_result.error().code, rolling_result.error().message);
335
336 // Benchmark comparison if provided
337 if (benchmark) {
338 auto benchmark_corr = cache_calculator_->correlation(returns, *benchmark);
339 if (benchmark_corr.is_ok()) {
340 auto benchmark_mean = cache_calculator_->mean(*benchmark);
341 auto benchmark_std = cache_calculator_->std_deviation(*benchmark);
342
343 if (benchmark_mean.is_ok() && benchmark_std.is_ok()) {
344 // Beta calculation
345 double covariance = benchmark_corr.value() * std_result.value() * benchmark_std.value();
346 double benchmark_variance = benchmark_std.value() * benchmark_std.value();
347 report.beta = benchmark_variance > 0 ? covariance / benchmark_variance : 0.0;
348
349 // Alpha calculation
350 double benchmark_annual_return = benchmark_mean.value() * config_.periods_per_year;
351 report.alpha = report.annual_return -
352 (config_.risk_free_rate +
353 report.beta.value() * (benchmark_annual_return - config_.risk_free_rate));
354
355 // Information ratio and tracking error
356 std::vector<double> excess_returns;
357 for (size_t i = 0; i < std::min(returns.size(), benchmark->size()); ++i) {
358 excess_returns.push_back(returns[i] - (*benchmark)[i]);
359 }
360
361 if (!excess_returns.empty()) {
362 TimeSeries<double> excess_ts(
363 std::vector<DateTime>(returns.timestamps().begin(),
364 returns.timestamps().begin() + excess_returns.size()),
365 excess_returns, "excess_returns");
366
367 auto excess_std = cache_calculator_->std_deviation(excess_ts);
368 auto excess_mean = cache_calculator_->mean(excess_ts);
369
370 if (excess_std.is_ok() && excess_mean.is_ok()) {
371 report.tracking_error = excess_std.value() * std::sqrt(config_.periods_per_year);
372 report.information_ratio = report.tracking_error.value() > 0
373 ? (excess_mean.value() * config_.periods_per_year) /
374 report.tracking_error.value()
375 : 0.0;
376 }
377 }
378 }
379 }
380 }
381
382 // Performance analysis and risk checks
383 perform_risk_analysis(report);
384
385 // Cache statistics
386 report.cache_stats = cache_calculator_->get_cache_stats();
387
388 // Timing information
389 auto end_time = std::chrono::high_resolution_clock::now();
390 report.computation_time = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
391
392 // Update performance tracking
393 ++total_analyses_performed_;
394 total_computation_time_ += std::chrono::duration_cast<std::chrono::nanoseconds>(end_time - start_time);
395 last_analysis_time_ = std::chrono::steady_clock::now();
396
397 return Result<AnalysisReport>::success(std::move(report));
398
399 } catch (const std::exception& e) {
400 return Result<AnalysisReport>::error(ErrorCode::CalculationError,
401 "Analysis failed: " + std::string(e.what()));
402 }
403 }
