29struct PriorDistribution {

32 double variance = 1.0;

38 static PriorDistribution uniform(double lower = -1.0, double upper = 1.0) {
39 PriorDistribution prior;
40 prior.type = PriorType::Uniform;
41 prior.mean = (lower + upper) / 2.0;
42 prior.variance = std::pow(upper - lower, 2) / 12.0;
43 return prior;
44 }
45
49 static PriorDistribution normal(double mean, double variance) {
50 PriorDistribution prior;

138struct BayesianRegimeResult {
139 std::vector<int> regime_sequence; // 0, 1, 2, ... for different regimes
140 std::vector<double> regime_probabilities; // Probability of each regime at each time
141 std::vector<double> transition_probabilities; // Regime transition matrix (flattened)
142 std::vector<double> regime_means; // Mean return for each regime
143 std::vector<double> regime_volatilities; // Volatility for each regime
144 size_t num_regimes;
145
149 double current_regime_probability(int regime) const {
150 if (regime_sequence.empty() || regime < 0 || regime >= static_cast<int>(num_regimes)) {
151 return 0.0;
152 }
153 return regime_probabilities.back() * (regime_sequence.back() == regime ? 1.0 : 0.0);
154 }
155
159 double regime_persistence(int regime) const {
160 if (regime < 0 || regime >= static_cast<int>(num_regimes) ||
161 transition_probabilities.size() != num_regimes * num_regimes) {
162 return 0.0;
163 }
164
165 // Diagonal element of transition matrix
166 double stay_prob = transition_probabilities[regime * num_regimes + regime];
167 return 1.0 / (1.0 - stay_prob);
168 }

228 Result<BayesianPerformanceResult> analyze_performance(
229 const ReturnSeries& portfolio_returns, const ReturnSeries& benchmark_returns,
230 const PriorDistribution& alpha_prior = PriorDistribution::normal(0.0, 0.01),
231 const PriorDistribution& beta_prior = PriorDistribution::normal(1.0, 0.25),
232 double risk_free_rate = 0.02) const {
233 if (portfolio_returns.size() != benchmark_returns.size()) {
234 return Result<BayesianPerformanceResult>::error(ErrorCode::InvalidInput,
235 "Portfolio and benchmark returns must have same length");
236 }
237
238 if (portfolio_returns.size() < 30) {
239 return Result<BayesianPerformanceResult>::error(ErrorCode::InsufficientData,
240 "Need at least 30 observations for Bayesian analysis");
241 }
242
243 // Convert to excess returns
244 const auto& port_returns = portfolio_returns.values();
245 const auto& bench_returns = benchmark_returns.values();
246
247 std::vector<double> excess_port_returns;
248 std::vector<double> excess_bench_returns;
249
250 double daily_rf = risk_free_rate / 252.0;
251
252 for (size_t i = 0; i < port_returns.size(); ++i) {
253 excess_port_returns.push_back(port_returns[i] - daily_rf);
254 excess_bench_returns.push_back(bench_returns[i] - daily_rf);
255 }
256
257 // MCMC sampling for alpha and beta
258 auto mcmc_result = mcmc_alpha_beta_sampling(excess_port_returns, excess_bench_returns, alpha_prior, beta_prior);
259
260 if (mcmc_result.is_error()) {
261 return Result<BayesianPerformanceResult>::error(mcmc_result.error().code, mcmc_result.error().message);
262 }
263
264 BayesianPerformanceResult result = mcmc_result.value();
265
266 // Calculate Sharpe ratio samples
267 result.sharpe_samples.reserve(result.alpha_samples.size());
268 for (size_t i = 0; i < result.alpha_samples.size(); ++i) {
269 // Calculate portfolio volatility for each sample
270 double alpha = result.alpha_samples[i];
271 double beta = result.beta_samples[i];
272
273 // Estimate portfolio return and volatility
274 auto bench_vol_result = stats::standard_deviation(std::span<const double>{excess_bench_returns});
275 if (bench_vol_result.is_ok()) {
276 double bench_vol = bench_vol_result.value();
277 double port_vol = result.volatility_samples[i];
278 double expected_return =
279 alpha + beta * stats::mean(std::span<const double>{excess_bench_returns}).value_or(0.0);
280
281 if (port_vol > 0) {
282 result.sharpe_samples.push_back(expected_return / port_vol);
283 } else {
284 result.sharpe_samples.push_back(0.0);
285 }
286 } else {
287 result.sharpe_samples.push_back(0.0);
288 }
289 }
290
291 // Calculate summary statistics
292 calculate_summary_statistics(result);
293
294 return Result<BayesianPerformanceResult>::success(std::move(result));
295 }

383 Result<std::pair<double, double>> bayesian_var(const ReturnSeries& returns, double confidence_level = 0.95) const {
384 if (returns.empty()) {
385 return Result<std::pair<double, double>>::error(ErrorCode::InsufficientData,
386 "Return series cannot be empty");
387 }
388
389 const auto& return_values = returns.values();
390
391 // Bayesian estimation of VaR distribution
392 std::vector<double> var_samples;
393 var_samples.reserve(num_samples_);
394
395 // Sample from posterior distribution of parameters
396 std::normal_distribution<double> normal_dist;
397 std::uniform_real_distribution<double> uniform_dist(0.0, 1.0);
398
399 for (size_t sample = 0; sample < num_samples_; ++sample) {
400 // Sample mean and variance from posterior
401 auto mean_result = stats::mean(std::span<const double>{return_values});
402 auto var_result = stats::variance(std::span<const double>{return_values});
403
404 if (mean_result.is_error() || var_result.is_error())
405 continue;
406
407 double sample_mean = mean_result.value();
408 double sample_var = var_result.value();
409
410 // Add uncertainty to parameters
411 double uncertain_mean = sample_mean + normal_dist(rng_) * std::sqrt(sample_var / return_values.size());
412 double uncertain_std = std::sqrt(sample_var) * (1.0 + 0.1 * normal_dist(rng_));
413
414 // Calculate VaR for this parameter sample
415 double z_score = stats::normal_ppf(1.0 - confidence_level);
416 double var_estimate = uncertain_mean + z_score * uncertain_std;
417
418 var_samples.push_back(var_estimate);
419 }
420
421 if (var_samples.empty()) {
422 return Result<std::pair<double, double>>::error(ErrorCode::CalculationError,
423 "Failed to generate VaR samples");
424 }
425
426 // Calculate mean and standard deviation of VaR estimates
427 double mean_var = std::accumulate(var_samples.begin(), var_samples.end(), 0.0) / var_samples.size();
428
429 double var_variance = 0.0;
430 for (double var_sample : var_samples) {
431 var_variance += std::pow(var_sample - mean_var, 2);
432 }
433 var_variance /= (var_samples.size() - 1);
434 double var_std = std::sqrt(var_variance);
435
436 return Result<std::pair<double, double>>::success(std::make_pair(mean_var, var_std));
437 }

443 Result<BayesianPerformanceResult> mcmc_alpha_beta_sampling(const std::vector<double>& portfolio_returns,
444 const std::vector<double>& benchmark_returns,
445 const PriorDistribution& alpha_prior,
446 const PriorDistribution& beta_prior) const {
447 BayesianPerformanceResult result;
448 result.alpha_samples.reserve(num_samples_);
449 result.beta_samples.reserve(num_samples_);
450 result.volatility_samples.reserve(num_samples_);
451
452 // Initialize parameters
453 double alpha = 0.0;
454 double beta = 1.0;
455 double sigma = 0.01; // Residual volatility
456
457 std::normal_distribution<double> normal_dist;
458 std::gamma_distribution<double> gamma_dist(2.0, 0.01); // For precision sampling
459
460 size_t total_samples = num_samples_ + burn_in_;
461
462 for (size_t iter = 0; iter < total_samples; ++iter) {
463 // Gibbs sampling
464
465 // Sample alpha | beta, sigma, data
466 double alpha_precision = 1.0 / alpha_prior.variance;
467 double alpha_likelihood_precision = 0.0;
468 double alpha_likelihood_mean = 0.0;
469
470 for (size_t i = 0; i < portfolio_returns.size(); ++i) {
471 double residual = portfolio_returns[i] - beta * benchmark_returns[i];
472 alpha_likelihood_precision += 1.0 / (sigma * sigma);
473 alpha_likelihood_mean += residual / (sigma * sigma);
474 }
475
476 double posterior_precision = alpha_precision + alpha_likelihood_precision;
477 double posterior_mean = (alpha_prior.mean * alpha_precision + alpha_likelihood_mean) / posterior_precision;
478 double posterior_var = 1.0 / posterior_precision;
479
480 alpha = posterior_mean + normal_dist(rng_) * std::sqrt(posterior_var);
481
482 // Sample beta | alpha, sigma, data
483 double beta_precision = 1.0 / beta_prior.variance;
484 double beta_likelihood_precision = 0.0;
485 double beta_likelihood_mean = 0.0;
486
487 for (size_t i = 0; i < portfolio_returns.size(); ++i) {
488 double bench_return = benchmark_returns[i];
489 double residual = portfolio_returns[i] - alpha;
490 beta_likelihood_precision += bench_return * bench_return / (sigma * sigma);
491 beta_likelihood_mean += residual * bench_return / (sigma * sigma);
492 }
493
494 posterior_precision = beta_precision + beta_likelihood_precision;
495 posterior_mean = (beta_prior.mean * beta_precision + beta_likelihood_mean) / posterior_precision;
496 posterior_var = 1.0 / posterior_precision;
497
498 beta = posterior_mean + normal_dist(rng_) * std::sqrt(posterior_var);
499
500 // Sample sigma | alpha, beta, data
501 double sse = 0.0;
502 for (size_t i = 0; i < portfolio_returns.size(); ++i) {
503 double residual = portfolio_returns[i] - alpha - beta * benchmark_returns[i];
504 sse += residual * residual;
505 }
506
507 double shape = 2.0 + portfolio_returns.size() / 2.0;
508 double rate = 1.0 + sse / 2.0;
509
510 std::gamma_distribution<double> precision_dist(shape, 1.0 / rate);
511 double precision = precision_dist(rng_);
512 sigma = 1.0 / std::sqrt(precision);
513
514 // Store samples (after burn-in)
515 if (iter >= burn_in_) {
516 result.alpha_samples.push_back(alpha);
517 result.beta_samples.push_back(beta);
518 result.volatility_samples.push_back(sigma);
519 }
520 }
521
522 return Result<BayesianPerformanceResult>::success(std::move(result));
523 }

528 Result<BayesianRegimeResult> estimate_hmm_parameters(const std::vector<double>& returns, size_t num_regimes) const {
529 BayesianRegimeResult result;
530 result.num_regimes = num_regimes;
531
532 // Initialize parameters randomly
533 std::uniform_real_distribution<double> uniform_dist(0.0, 1.0);
534 std::normal_distribution<double> normal_dist;
535
536 // Initialize regime means and volatilities
537 auto returns_mean = stats::mean(std::span<const double>{returns}).value_or(0.0);
538 auto returns_std = stats::standard_deviation(std::span<const double>{returns}).value_or(0.01);
539
540 result.regime_means.resize(num_regimes);
541 result.regime_volatilities.resize(num_regimes);
542
543 for (size_t i = 0; i < num_regimes; ++i) {
544 result.regime_means[i] = returns_mean + normal_dist(rng_) * returns_std * 0.5;
545 result.regime_volatilities[i] = returns_std * (0.5 + uniform_dist(rng_));
546 }
547
548 // Initialize transition matrix
549 result.transition_probabilities.resize(num_regimes * num_regimes);
550 for (size_t i = 0; i < num_regimes; ++i) {
551 double row_sum = 0.0;
552 for (size_t j = 0; j < num_regimes; ++j) {
553 double prob = uniform_dist(rng_);
554 result.transition_probabilities[i * num_regimes + j] = prob;
555 row_sum += prob;
556 }
557 // Normalize row
558 for (size_t j = 0; j < num_regimes; ++j) {
559 result.transition_probabilities[i * num_regimes + j] /= row_sum;
560 }
561 }
562
563 // Simplified Viterbi algorithm for regime sequence
564 result.regime_sequence.resize(returns.size());
565 result.regime_probabilities.resize(returns.size());
566
567 for (size_t t = 0; t < returns.size(); ++t) {
568 // Simple assignment based on likelihood
569 double best_likelihood = -std::numeric_limits<double>::infinity();
570 int best_regime = 0;
571
572 for (size_t regime = 0; regime < num_regimes; ++regime) {
573 double mean = result.regime_means[regime];
574 double vol = result.regime_volatilities[regime];
575
576 if (vol > 0) {
577 double likelihood =
578 -0.5 * std::log(2.0 * M_PI * vol * vol) - 0.5 * std::pow(returns[t] - mean, 2) / (vol * vol);
579
580 if (likelihood > best_likelihood) {
581 best_likelihood = likelihood;
582 best_regime = static_cast<int>(regime);
583 }
584 }
585 }
586
587 result.regime_sequence[t] = best_regime;
588 result.regime_probabilities[t] = std::exp(best_likelihood);
589 }
590
591 return Result<BayesianRegimeResult>::success(std::move(result));
592 }

597 Result<BayesianForecastResult> bayesian_ar_forecast(const std::vector<double>& returns, size_t forecast_horizon,
598 const PriorDistribution& volatility_prior) const {
599 BayesianForecastResult result;
600 result.return_forecasts.resize(forecast_horizon);
601 result.forecast_lower_95.resize(forecast_horizon);
602 result.forecast_upper_95.resize(forecast_horizon);
603 result.forecast_volatility.resize(forecast_horizon);
604
605 // Estimate AR(1) parameters
606 if (returns.size() < 2) {
607 return Result<BayesianForecastResult>::error(ErrorCode::InsufficientData,
608 "Need at least 2 observations for AR(1) model");
609 }
610
611 // Simple AR(1): r_t = phi * r_{t-1} + epsilon_t
612 std::vector<double> y, x;
613 for (size_t i = 1; i < returns.size(); ++i) {
614 y.push_back(returns[i]);
615 x.push_back(returns[i - 1]);
616 }
617
618 // Bayesian regression for phi
619 double phi = estimate_ar_coefficient(x, y);
620 double sigma = estimate_residual_volatility(x, y, phi);
621
622 // Generate forecasts with uncertainty
623 std::normal_distribution<double> normal_dist;
624
625 for (size_t h = 0; h < forecast_horizon; ++h) {
626 std::vector<double> forecast_samples;
627 forecast_samples.reserve(1000);
628
629 for (size_t sample = 0; sample < 1000; ++sample) {
630 double forecast = 0.0;
631 double last_return = returns.back();
632
633 // Multi-step ahead forecast
634 for (size_t step = 0; step <= h; ++step) {
635 if (step == 0) {
636 forecast = phi * last_return + normal_dist(rng_) * sigma;
637 } else {
638 forecast = phi * forecast + normal_dist(rng_) * sigma;
639 }
640 }
641
642 forecast_samples.push_back(forecast);
643 }
644
645 // Calculate statistics
646 std::sort(forecast_samples.begin(), forecast_samples.end());
647
648 result.return_forecasts[h] =
649 std::accumulate(forecast_samples.begin(), forecast_samples.end(), 0.0) / forecast_samples.size();
650
651 result.forecast_lower_95[h] = forecast_samples[static_cast<size_t>(0.025 * forecast_samples.size())];
652 result.forecast_upper_95[h] = forecast_samples[static_cast<size_t>(0.975 * forecast_samples.size())];
653
654 // Calculate forecast volatility
655 double variance = 0.0;
656 for (double sample : forecast_samples) {
657 variance += std::pow(sample - result.return_forecasts[h], 2);
658 }
659 result.forecast_volatility[h] = std::sqrt(variance / (forecast_samples.size() - 1));
660 }
661
662 return Result<BayesianForecastResult>::success(std::move(result));
663 }

668 void calculate_summary_statistics(BayesianPerformanceResult& result) const {
669 // Alpha statistics
670 if (!result.alpha_samples.empty()) {
671 result.alpha_mean = std::accumulate(result.alpha_samples.begin(), result.alpha_samples.end(), 0.0) /
672 result.alpha_samples.size();
673
674 double alpha_variance = 0.0;
675 for (double alpha : result.alpha_samples) {
676 alpha_variance += std::pow(alpha - result.alpha_mean, 2);
677 }
678 result.alpha_std = std::sqrt(alpha_variance / (result.alpha_samples.size() - 1));
679
680 auto sorted_alpha = result.alpha_samples;
681 std::sort(sorted_alpha.begin(), sorted_alpha.end());
682 result.alpha_credible_lower = sorted_alpha[static_cast<size_t>(0.025 * sorted_alpha.size())];
683 result.alpha_credible_upper = sorted_alpha[static_cast<size_t>(0.975 * sorted_alpha.size())];
684
685 // Calculate probabilities
686 size_t positive_count = std::count_if(result.alpha_samples.begin(), result.alpha_samples.end(),
687 [](double alpha) { return alpha > 0.0; });
688 result.prob_alpha_positive = static_cast<double>(positive_count) / result.alpha_samples.size();
689 result.prob_outperformance = result.prob_alpha_positive;
690 }
691
692 // Beta statistics
693 if (!result.beta_samples.empty()) {
694 result.beta_mean = std::accumulate(result.beta_samples.begin(), result.beta_samples.end(), 0.0) /
695 result.beta_samples.size();
696
697 double beta_variance = 0.0;
698 for (double beta : result.beta_samples) {
699 beta_variance += std::pow(beta - result.beta_mean, 2);
700 }
701 result.beta_std = std::sqrt(beta_variance / (result.beta_samples.size() - 1));
702
703 auto sorted_beta = result.beta_samples;
704 std::sort(sorted_beta.begin(), sorted_beta.end());
705 result.beta_credible_lower = sorted_beta[static_cast<size_t>(0.025 * sorted_beta.size())];
706 result.beta_credible_upper = sorted_beta[static_cast<size_t>(0.975 * sorted_beta.size())];
707
708 size_t greater_one_count = std::count_if(result.beta_samples.begin(), result.beta_samples.end(),
709 [](double beta) { return beta > 1.0; });
710 result.prob_beta_greater_one = static_cast<double>(greater_one_count) / result.beta_samples.size();
711 }
712
713 // Sharpe ratio statistics
714 if (!result.sharpe_samples.empty()) {
715 result.sharpe_mean = std::accumulate(result.sharpe_samples.begin(), result.sharpe_samples.end(), 0.0) /
716 result.sharpe_samples.size();
717
718 double sharpe_variance = 0.0;
719 for (double sharpe : result.sharpe_samples) {
720 sharpe_variance += std::pow(sharpe - result.sharpe_mean, 2);
721 }
722 result.sharpe_std = std::sqrt(sharpe_variance / (result.sharpe_samples.size() - 1));
723
724 auto sorted_sharpe = result.sharpe_samples;
725 std::sort(sorted_sharpe.begin(), sorted_sharpe.end());
726 result.sharpe_credible_lower = sorted_sharpe[static_cast<size_t>(0.025 * sorted_sharpe.size())];
727 result.sharpe_credible_upper = sorted_sharpe[static_cast<size_t>(0.975 * sorted_sharpe.size())];
728 }
729 }
730
734 double estimate_ar_coefficient(const std::vector<double>& x, const std::vector<double>& y) const {
735 if (x.size() != y.size() || x.empty())
736 return 0.0;
737
738 double sum_x = std::accumulate(x.begin(), x.end(), 0.0);
739 double sum_y = std::accumulate(y.begin(), y.end(), 0.0);
740 double sum_xx = std::inner_product(x.begin(), x.end(), x.begin(), 0.0);
741 double sum_xy = std::inner_product(x.begin(), x.end(), y.begin(), 0.0);
742
743 double n = static_cast<double>(x.size());
744 double numerator = sum_xy - (sum_x * sum_y) / n;
745 double denominator = sum_xx - (sum_x * sum_x) / n;
746
747 return (denominator != 0.0) ? numerator / denominator : 0.0;
748 }
749
753 double estimate_residual_volatility(const std::vector<double>& x, const std::vector<double>& y, double phi) const {
754 if (x.size() != y.size() || x.empty())
755 return 0.01;
756
757 double sse = 0.0;
758 for (size_t i = 0; i < x.size(); ++i) {
759 double residual = y[i] - phi * x[i];
760 sse += residual * residual;
761 }
762
763 return std::sqrt(sse / (x.size() - 1));
764 }
