199template <typename T>
200void add_series(const std::vector<T>& a, const std::vector<T>& b, std::vector<T>& result)
201 requires std::floating_point<T>
202{
203 size_t n = std::min({a.size(), b.size(), result.size()});
204 if constexpr (std::is_same_v<T, double>) {
205 vector_add(std::span<const double>(a.data(), n), std::span<const double>(b.data(), n),
206 std::span<double>(result.data(), n));
207 } else {
208 // Fallback for other types
209 std::transform(a.begin(), a.begin() + n, b.begin(), result.begin(), std::plus<T>{});
210 }
211}
