41 IntradayDetectionResult result{.is_intraday = is_intraday,
42 .threshold_ratio = threshold,
43 .optimal_sampling_time = optimal_time,
44 .detection_method = "position_transaction_ratio",
45 .confidence_score = confidence,
46 .symbol_ratios = symbol_ratios};

146double IntradayAnalyzer::calculate_position_transaction_ratio(const PositionSeries& positions,
147 const TransactionSeries& transactions) {
148 if (positions.empty() || transactions.empty()) {
149 return 1.0; // Safe default
150 }
151
152 // Calculate average absolute position values (excluding cash)
153 double total_position_value = 0.0;
154 int position_count = 0;
155
156 for (size_t i = 0; i < positions.size(); ++i) {
157 const auto& snapshot = positions[i];
158 for (const auto& [symbol, value] : snapshot.positions()) {
159 if (symbol != "cash") {
160 total_position_value += std::abs(value);
161 position_count++;
162 }
163 }
164 }
165
166 double avg_position_value = position_count > 0 ? total_position_value / position_count : 0.0;
167
168 // Calculate average transaction volume
169 auto txn_stats_result = transactions.calculate_statistics();
170 if (txn_stats_result.is_error()) {
171 return 1.0; // Safe default
172 }
173
174 double avg_txn_value = txn_stats_result.value().average_transaction_size;
175
176 // Calculate ratio
177 if (avg_txn_value > 0) {
178 return avg_position_value / avg_txn_value;
179 }
180
181 return 1.0; // Safe default
182}

204 total_position_value += std::abs(pos_value);
205 count++;
206 }

208

289double IntradayAnalyzer::calculate_confidence_score(const std::map<std::string, double>& symbol_ratios,
290 double overall_ratio, double threshold) {
291 if (symbol_ratios.empty()) {
292 return 0.5; // Neutral confidence
293 }
294
295 // Calculate consistency across symbols
296 std::vector<bool> symbol_classifications;
297 for (const auto& [symbol, ratio] : symbol_ratios) {
298 symbol_classifications.push_back(ratio < threshold);
299 }
300
301 // Count consistent classifications
302 bool overall_classification = overall_ratio < threshold;
303 int consistent_count = 0;
304
305 for (bool classification : symbol_classifications) {
306 if (classification == overall_classification) {
307 consistent_count++;
308 }
309 }
310
311 double consistency_ratio = static_cast<double>(consistent_count) / symbol_classifications.size();
312
313 // Distance from threshold also affects confidence
314 double threshold_distance = std::abs(overall_ratio - threshold) / threshold;
315 double threshold_confidence = std::min(1.0, threshold_distance);
316
317 // Combine consistency and threshold distance
318 return (consistency_ratio + threshold_confidence) / 2.0;
319}

332std::vector<double> normalize_positions(const std::vector<double>& positions) {
333 if (positions.empty()) {
334 return {};
335 }
336
337 // Calculate mean and standard deviation
338 double sum = std::accumulate(positions.begin(), positions.end(), 0.0);
339 double mean = sum / positions.size();
340
341 double sq_sum = 0.0;
342 for (double pos : positions) {
343 sq_sum += (pos - mean) * (pos - mean);
344 }
345 double std_dev = std::sqrt(sq_sum / positions.size());
346
347 if (std_dev == 0.0) {
348 return std::vector<double>(positions.size(), 0.0);
349 }
350
351 // Normalize
352 std::vector<double> normalized;
353 normalized.reserve(positions.size());
354
355 for (double pos : positions) {
356 normalized.push_back((pos - mean) / std_dev);
357 }
358
359 return normalized;
360}

362double calculate_average_absolute_deviation(const std::vector<double>& values) {
363 if (values.empty()) {
364 return 0.0;
365 }
366
367 // Calculate mean
368 double sum = std::accumulate(values.begin(), values.end(), 0.0);
369 double mean = sum / values.size();
370
371 // Calculate average absolute deviation
372 double aad_sum = 0.0;
373 for (double value : values) {
374 aad_sum += std::abs(value - mean);
375 }
376
377 return aad_sum / values.size();
