27std::pair<TimeSeries<Return>, TimeSeries<Return>> generate_sample_data() {
28 std::vector<DateTime> dates;
29 std::vector<double> strategy_returns;
30 std::vector<double> benchmark_returns;
31
32 std::random_device rd;
33 std::mt19937 gen(rd());
34 std::normal_distribution<double> strategy_dist(0.0008, 0.015); // 0.08% daily return, 1.5% volatility
35 std::normal_distribution<double> benchmark_dist(0.0005, 0.012); // 0.05% daily return, 1.2% volatility
36
37 // Generate 2 years of daily data
38 for (int i = 0; i < 504; ++i) {
39 DateTime base_date(2022, 1, 1);
40 dates.push_back(base_date.add_days(i));
41 strategy_returns.push_back(strategy_dist(gen));
42 benchmark_returns.push_back(benchmark_dist(gen));
43 }
44
45 auto strategy_ts = TimeSeries<Return>::create(dates, strategy_returns).value();
46 auto benchmark_ts = TimeSeries<Return>::create(dates, benchmark_returns).value();
47
48 return {strategy_ts, benchmark_ts};
49}

79OHLCData generate_ohlc_data() {
80 OHLCData ohlc_data;
81 ohlc_data.name = "Sample Stock";
82
83 std::random_device rd;
84 std::mt19937 gen(rd());
85 std::uniform_real_distribution<double> price_change(-2.0, 2.0);
86 std::uniform_real_distribution<double> volume_factor(0.8, 1.2);
87
88 double base_price = 100.0;
89
90 for (int i = 0; i < 60; ++i) { // 2 months of daily data
91 DateTime base_date(2024, 1, 1);
92 ohlc_data.timestamps.push_back(base_date.add_days(i));
93
94 double open = base_price;
95 double change = price_change(gen);
96 double close = open + change;
97 double high = std::max(open, close) + std::abs(price_change(gen) * 0.5);
98 double low = std::min(open, close) - std::abs(price_change(gen) * 0.5);
99 double volume = 1000000 * volume_factor(gen);
100
101 ohlc_data.open.push_back(open);
102 ohlc_data.high.push_back(high);
103 ohlc_data.low.push_back(low);
104 ohlc_data.close.push_back(close);
105 ohlc_data.volume.push_back(volume);
106
107 base_price = close;
108 }
109
110 return ohlc_data;
111}
