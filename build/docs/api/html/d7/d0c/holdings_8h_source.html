80 Result<void> update_holding(const Symbol& symbol, Shares shares, Price average_cost, Price current_price) {
81 if (current_price <= 0) {
82 return Result<void>::error(ErrorCode::InvalidInput, "Current price must be positive");
83 }
84
85 Holding& holding = holdings_[symbol];
86 holding.symbol = symbol;
87 holding.shares = shares;
88 holding.average_cost = average_cost;
89 holding.current_price = current_price;
90
91 // Remove if position is closed
92 if (std::abs(shares) < 1e-6) {
93 holdings_.erase(symbol);
94 } else {
95 holding.calculate_metrics(total_value_);
96 }
97
98 update_total_value();
99
100 // Recalculate weights after value update
101 for (auto& [sym, h] : holdings_) {
102 h.calculate_metrics(total_value_);
103 }
104
105 return Result<void>::success();
106 }

133 PortfolioMetrics calculate_metrics() const {
134 PortfolioMetrics metrics{};
135
136 metrics.cash_weight = total_value_ > 0 ? cash_balance_ / total_value_ : 0.0;
137
138 for (const auto& [symbol, holding] : holdings_) {
139 metrics.num_positions++;
140
141 if (holding.shares > 0) {
142 metrics.num_long_positions++;
143 metrics.long_exposure += holding.market_value;
144 } else {
145 metrics.num_short_positions++;
146 metrics.short_exposure += std::abs(holding.market_value);
147 }
148 }
149
150 metrics.gross_exposure = metrics.long_exposure + metrics.short_exposure;
151 metrics.net_exposure = metrics.long_exposure - metrics.short_exposure;
152
153 // Normalize by total value
154 if (total_value_ > 0) {
155 metrics.gross_exposure /= total_value_;
156 metrics.net_exposure /= total_value_;
157 metrics.long_exposure /= total_value_;
158 metrics.short_exposure /= total_value_;
159 }
160
161 return metrics;
162 }

288 static Result<HoldingsSeries> build_from_transactions(const transactions::TransactionSeries& txns,
289 const std::map<Symbol, PriceSeries>& price_data,
290 double initial_cash = 0.0) {
291 if (txns.empty()) {
292 return Result<HoldingsSeries>::error(ErrorCode::InsufficientData, "No transactions to process");
293 }
294
295 HoldingsSeries result;
296
297 // Track positions and cash
298 std::map<Symbol, std::pair<Shares, Price>> positions; // shares, avg_cost
299 double cash = initial_cash;
300
301 // Process transactions day by day
302 DateTime current_date{txns[0].timestamp().to_date()};
303 DateTime last_date{txns.back().timestamp().to_date()};
304
305 size_t txn_idx = 0;
306
307 while (current_date <= last_date) {
308 // Process all transactions for current date
309 while (txn_idx < txns.size() && DateTime{txns[txn_idx].timestamp().to_date()} == current_date) {
310 const auto& txn = txns[txn_idx];
311 auto& [shares, avg_cost] = positions[txn.symbol()];
312
313 // Update position
314 if (shares == 0) {
315 // New position
316 shares = txn.shares();
317 avg_cost = txn.price();
318 } else if ((shares > 0 && txn.shares() > 0) || (shares < 0 && txn.shares() < 0)) {
319 // Adding to position
320 double total_cost = shares * avg_cost + txn.shares() * txn.price();
321 shares += txn.shares();
322 avg_cost = shares != 0 ? total_cost / shares : 0.0;
323 } else {
324 // Reducing or reversing position
325 shares += txn.shares();
326 if (std::abs(shares) < 1e-6) {
327 shares = 0;
328 avg_cost = 0;
329 } else if (shares * txn.shares() < 0) {
330 // Position reversed
331 avg_cost = txn.price();
332 }
333 }
334
335 // Update cash
336 cash += txn.net_cash_flow();
337
338 txn_idx++;
339 }
340
341 // Create holdings snapshot
342 PortfolioHoldings holdings{current_date, cash};
343
344 // Update all positions with current prices
345 for (const auto& [symbol, pos_data] : positions) {
346 const auto& [shares, avg_cost] = pos_data;
347
348 if (std::abs(shares) < 1e-6)
349 continue;
350
351 // Get current price
352 auto price_it = price_data.find(symbol);
353 if (price_it != price_data.end()) {
354 auto price_result = price_it->second.at_time(current_date);
355 if (price_result.is_ok()) {
356 holdings.update_holding(symbol, shares, avg_cost, price_result.value());
357 }
358 }
359 }
360
361 result.add_holdings(holdings);
362
363 // Move to next business day
364 current_date = current_date.next_business_day();
365 }
366
367 return Result<HoldingsSeries>::success(std::move(result));
368 }
