76class RoundTripAnalyzer {
77 private:
78 std::map<Symbol, std::queue<OpenPosition>> open_positions_;
79 std::vector<RoundTrip> completed_trips_;
80
84 void process_transaction(const TransactionRecord& txn) {
85 const Symbol& symbol = txn.symbol();
86 auto& position_queue = open_positions_[symbol];
87
88 if (txn.shares() > 0) {
89 // Buy transaction - add to open positions
90 position_queue.push(OpenPosition{txn});
91 } else {
92 // Sell transaction - close positions FIFO
93 Shares remaining_shares = std::abs(txn.shares());
94
95 while (remaining_shares > 0 && !position_queue.empty()) {
96 OpenPosition& open_pos = position_queue.front();
97
98 Shares shares_to_close = std::min(remaining_shares, open_pos.shares);
99
100 // Create round trip
101 RoundTrip trip;
102 trip.symbol = symbol;
103 trip.open_date = open_pos.timestamp;
104 trip.close_date = txn.timestamp();
105 trip.shares = shares_to_close;
106 trip.open_price = open_pos.price;
107 trip.close_price = txn.price();
108
109 // Allocate costs proportionally
110 double share_ratio = shares_to_close / std::abs(txn.shares());
111 trip.commission_open = open_pos.commission * (shares_to_close / open_pos.shares);
112 trip.commission_close = txn.commission() * share_ratio;
113 trip.slippage_open = open_pos.slippage * (shares_to_close / open_pos.shares);
114 trip.slippage_close = txn.slippage() * share_ratio;
115
116 completed_trips_.push_back(trip);
117
118 // Update open position
119 open_pos.shares -= shares_to_close;
120 open_pos.commission *= (1.0 - shares_to_close / open_pos.shares);
121 open_pos.slippage *= (1.0 - shares_to_close / open_pos.shares);
122
123 if (open_pos.shares <= 0) {
124 position_queue.pop();
125 }
126
127 remaining_shares -= shares_to_close;
128 }
129
130 // If there are still shares to sell, it's a short position
131 if (remaining_shares > 0) {
132 OpenPosition short_pos{txn};
133 short_pos.shares = -remaining_shares;
134 position_queue.push(short_pos);
135 }
136 }
137 }
138
139 public:
143 Result<std::vector<RoundTrip>> analyze(const TransactionSeries& transactions) {
144 // Clear previous analysis
145 open_positions_.clear();
146 completed_trips_.clear();
147
148 // Process all transactions
149 for (const auto& txn : transactions) {
150 process_transaction(txn);
151 }
152
153 return Result<std::vector<RoundTrip>>::success(completed_trips_);
154 }
155
159 std::map<Symbol, std::vector<OpenPosition>> get_open_positions() const {
160 std::map<Symbol, std::vector<OpenPosition>> result;
161
162 for (const auto& [symbol, queue] : open_positions_) {
163 auto queue_copy = queue;
164 while (!queue_copy.empty()) {
165 result[symbol].push_back(queue_copy.front());
166 queue_copy.pop();
167 }
168 }
169
170 return result;
171 }

177struct RoundTripStatistics {
178 int total_trips;
179 int winning_trips;
180 int losing_trips;
181 double win_rate;
182 double average_pnl;
183 double average_return;
184 double average_duration_days;
185 double total_pnl;
186 double best_trade_pnl;
187 double worst_trade_pnl;
188 double profit_factor;
189
193 static Result<RoundTripStatistics> calculate(const std::vector<RoundTrip>& trips) {
194 if (trips.empty()) {
195 return Result<RoundTripStatistics>::error(ErrorCode::InsufficientData, "No round trips to analyze");
196 }
197
198 RoundTripStatistics stats{};
199 stats.total_trips = static_cast<int>(trips.size());
200
201 double total_wins = 0.0;
202 double total_losses = 0.0;
203 double total_duration = 0.0;
204 double total_return_pct = 0.0;
205
206 stats.best_trade_pnl = std::numeric_limits<double>::lowest();
207 stats.worst_trade_pnl = std::numeric_limits<double>::max();
208
209 for (const auto& trip : trips) {
210 double pnl = trip.pnl();
211 stats.total_pnl += pnl;
212
213 if (pnl > 0) {
214 stats.winning_trips++;
215 total_wins += pnl;
216 } else {
217 stats.losing_trips++;
218 total_losses += std::abs(pnl);
219 }
220
221 stats.best_trade_pnl = std::max(stats.best_trade_pnl, pnl);
222 stats.worst_trade_pnl = std::min(stats.worst_trade_pnl, pnl);
223
224 total_duration += trip.duration_days();
225 total_return_pct += trip.return_pct();
226 }
227
228 stats.win_rate = static_cast<double>(stats.winning_trips) / stats.total_trips;
229 stats.average_pnl = stats.total_pnl / stats.total_trips;
230 stats.average_return = total_return_pct / stats.total_trips;
231 stats.average_duration_days = total_duration / stats.total_trips;
232
233 // Profit factor = gross wins / gross losses
234 stats.profit_factor =
235 total_losses > 0 ? total_wins / total_losses : (total_wins > 0 ? std::numeric_limits<double>::max() : 1.0);
236
237 return Result<RoundTripStatistics>::success(stats);
238 }
239};

291inline Result<DataFrame> round_trip_performance_by_period(const std::vector<RoundTrip>& trips, Frequency freq) {
292 if (trips.empty()) {
293 return Result<DataFrame>::error(ErrorCode::InsufficientData, "No round trips to analyze");
294 }
295
296 // Group by close date period
297 std::map<DateTime, std::vector<RoundTrip>> period_trips;
298
299 for (const auto& trip : trips) {
300 DateTime period_start;
301 auto close_date = trip.close_date.to_date();
302
303 switch (freq) {
304 case Frequency::Monthly:
305 period_start = DateTime{close_date.year() / close_date.month() / std::chrono::day{1}};
306 break;
307 case Frequency::Quarterly: {
308 auto month = close_date.month();
309 auto quarter_start_month = ((static_cast<unsigned>(month) - 1) / 3) * 3 + 1;
310 period_start =
311 DateTime{close_date.year() / std::chrono::month{quarter_start_month} / std::chrono::day{1}};
312 } break;
313 case Frequency::Yearly:
314 period_start = DateTime{close_date.year() / std::chrono::January / std::chrono::day{1}};
315 break;
316 default:
317 period_start = trip.close_date;
318 }
319
320 period_trips[period_start].push_back(trip);
321 }
322
323 // Calculate statistics for each period
324 std::vector<DateTime> periods;
325 std::vector<double> total_pnls;
326 std::vector<double> win_rates;
327 std::vector<double> avg_returns;
328 std::vector<int> trip_counts;
329
330 for (const auto& [period, trips_in_period] : period_trips) {
331 auto stats = RoundTripStatistics::calculate(trips_in_period);
332 if (stats.is_ok()) {
333 periods.push_back(period);
334 total_pnls.push_back(stats.value().total_pnl);
335 win_rates.push_back(stats.value().win_rate);
336 avg_returns.push_back(stats.value().average_return);
337 trip_counts.push_back(stats.value().total_trips);
338 }
339 }
340
341 DataFrame df{periods};
342 df.add_column("total_pnl", total_pnls);
343 df.add_column("win_rate", win_rates);
344 df.add_column("avg_return", avg_returns);
345 df.add_column("trip_count", trip_counts);
346
347 return Result<DataFrame>::success(std::move(df));
348}
