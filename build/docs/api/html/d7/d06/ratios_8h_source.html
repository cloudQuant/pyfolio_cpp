15inline Result<double> sharpe_ratio(const ReturnSeries& returns,
16 double risk_free_rate = constants::DEFAULT_RISK_FREE_RATE,
17 Frequency frequency = Frequency::Daily) {
18 if (returns.empty()) {
19 return Result<double>::error(ErrorCode::InsufficientData,
20 "Cannot calculate Sharpe ratio for empty return series");
21 }
22
23 // Calculate excess returns
24 auto excess_returns_result = calculate_excess_returns(returns, risk_free_rate);
25 if (excess_returns_result.is_error()) {
26 return Result<double>::error(excess_returns_result.error().code, excess_returns_result.error().message);
27 }
28
29 const auto& excess_returns = excess_returns_result.value();
30
31 // Calculate mean excess return
32 auto mean_result = stats::mean(std::span<const Return>{excess_returns.values()});
33 if (mean_result.is_error()) {
34 return Result<double>::error(mean_result.error().code, mean_result.error().message);
35 }
36
37 // Calculate volatility of excess returns
38 auto volatility_result = calculate_volatility(excess_returns, frequency);
39 if (volatility_result.is_error()) {
40 return Result<double>::error(volatility_result.error().code, volatility_result.error().message);
41 }
42
43 double mean_excess_return = mean_result.value();
44 double volatility = volatility_result.value();
45
46 if (volatility == 0.0) {
47 return Result<double>::error(ErrorCode::DivisionByZero, "Cannot calculate Sharpe ratio with zero volatility");
48 }
49
50 // Annualize the mean excess return
51 double periods_per_year = frequency::to_annual_factor(frequency);
52 double annualized_mean = mean_excess_return * periods_per_year;
53
54 double sharpe = annualized_mean / volatility;
55
56 return Result<double>::success(sharpe);
57}

62inline Result<double> sortino_ratio(const ReturnSeries& returns, double target_return = 0.0,
63 Frequency frequency = Frequency::Daily) {
64 if (returns.empty()) {
65 return Result<double>::error(ErrorCode::InsufficientData,
66 "Cannot calculate Sortino ratio for empty return series");
67 }
68
69 // Convert annual target return to period return
70 double periods_per_year = frequency::to_annual_factor(frequency);
71 double period_target = target_return / periods_per_year;
72
73 // Calculate mean return
74 auto mean_result = stats::mean(std::span<const Return>{returns.values()});
75 if (mean_result.is_error()) {
76 return Result<double>::error(mean_result.error().code, mean_result.error().message);
77 }
78
79 double mean_return = mean_result.value();
80
81 // Calculate downside deviation
82 std::vector<double> downside_deviations;
83 for (const auto& ret : returns) {
84 if (ret < period_target) {
85 double deviation = ret - period_target;
86 downside_deviations.push_back(deviation * deviation);
87 }
88 }
89
90 if (downside_deviations.empty()) {
91 return Result<double>::error(ErrorCode::InsufficientData,
92 "No downside deviations found for Sortino ratio calculation");
93 }
94
95 double downside_variance =
96 std::accumulate(downside_deviations.begin(), downside_deviations.end(), 0.0) / downside_deviations.size();
97 double downside_deviation = std::sqrt(downside_variance);
98
99 if (downside_deviation == 0.0) {
100 return Result<double>::error(ErrorCode::DivisionByZero,
101 "Cannot calculate Sortino ratio with zero downside deviation");
102 }
103
104 // Annualize
105 double annualized_mean = mean_return * periods_per_year;
106 double annualized_downside_dev = downside_deviation * std::sqrt(periods_per_year);
107
108 double sortino = (annualized_mean - target_return) / annualized_downside_dev;
109
110 return Result<double>::success(sortino);
111}

116inline Result<double> calmar_ratio(const ReturnSeries& returns, Frequency frequency = Frequency::Daily) {
117 if (returns.empty()) {
118 return Result<double>::error(ErrorCode::InsufficientData,
119 "Cannot calculate Calmar ratio for empty return series");
120 }
121
122 // Calculate annualized return
123 auto annualized_return_result = annualize_returns(returns, frequency);
124 if (annualized_return_result.is_error()) {
125 return Result<double>::error(annualized_return_result.error().code, annualized_return_result.error().message);
126 }
127
128 // Calculate maximum drawdown (we'll implement this in drawdown.h)
129 // For now, simplified calculation
130 auto cumulative_result = calculate_cumulative_returns(returns);
131 if (cumulative_result.is_error()) {
132 return Result<double>::error(cumulative_result.error().code, cumulative_result.error().message);
133 }
134
135 const auto& cumulative_returns = cumulative_result.value();
136
137 double peak = cumulative_returns[0];
138 double max_drawdown = 0.0;
139
140 for (const auto& ret : cumulative_returns) {
141 if (ret > peak) {
142 peak = ret;
143 }
144
145 double drawdown = (peak - ret) / (1.0 + peak);
146 if (drawdown > max_drawdown) {
147 max_drawdown = drawdown;
148 }
149 }
150
151 if (max_drawdown == 0.0) {
152 return Result<double>::error(ErrorCode::DivisionByZero,
153 "Cannot calculate Calmar ratio with zero maximum drawdown");
154 }
155
156 double calmar = annualized_return_result.value() / max_drawdown;

158 return Result<double>::success(calmar);

164inline Result<double> information_ratio(const ReturnSeries& portfolio_returns, const ReturnSeries& benchmark_returns,
165 Frequency frequency = Frequency::Daily) {
166 if (portfolio_returns.size() != benchmark_returns.size()) {
167 return Result<double>::error(ErrorCode::InvalidInput,
168 "Portfolio and benchmark returns must have the same length");
169 }
170
171 if (portfolio_returns.empty()) {
172 return Result<double>::error(ErrorCode::InsufficientData,
173 "Cannot calculate Information ratio for empty return series");
174 }
175
176 // Calculate active returns (portfolio - benchmark)
177 std::vector<Return> active_returns;
178 active_returns.reserve(portfolio_returns.size());
179
180 for (size_t i = 0; i < portfolio_returns.size(); ++i) {
181 active_returns.push_back(portfolio_returns[i] - benchmark_returns[i]);
182 }
183
184 // Calculate mean active return
185 auto mean_result = stats::mean(std::span<const Return>{active_returns});
186 if (mean_result.is_error()) {
187 return Result<double>::error(mean_result.error().code, mean_result.error().message);
188 }
189
190 // Calculate tracking error (volatility of active returns)
191 auto std_result = stats::standard_deviation(std::span<const Return>{active_returns});
192 if (std_result.is_error()) {
193 return Result<double>::error(std_result.error().code, std_result.error().message);
194 }
195
196 double mean_active_return = mean_result.value();
197 double tracking_error = std_result.value();
198
199 if (tracking_error == 0.0) {
200 return Result<double>::error(ErrorCode::DivisionByZero,
201 "Cannot calculate Information ratio with zero tracking error");
202 }
203
204 // Annualize
205 double periods_per_year = frequency::to_annual_factor(frequency);
206 double annualized_active_return = mean_active_return * periods_per_year;
207 double annualized_tracking_error = tracking_error * std::sqrt(periods_per_year);
208
209 double information_ratio_value = annualized_active_return / annualized_tracking_error;
210
211 return Result<double>::success(information_ratio_value);
212}

217inline Result<double> omega_ratio(const ReturnSeries& returns, double threshold = 0.0,
218 Frequency frequency = Frequency::Daily) {
219 if (returns.empty()) {
220 return Result<double>::error(ErrorCode::InsufficientData,
221 "Cannot calculate Omega ratio for empty return series");
222 }
223
224 // Convert annual threshold to period threshold
225 double periods_per_year = frequency::to_annual_factor(frequency);
226 double period_threshold = threshold / periods_per_year;
227
228 double gains_sum = 0.0;
229 double losses_sum = 0.0;
230
231 for (const auto& ret : returns) {
232 if (ret > period_threshold) {
233 gains_sum += (ret - period_threshold);
234 } else {
235 losses_sum += (period_threshold - ret);
236 }
237 }
238
239 if (losses_sum == 0.0) {
240 if (gains_sum > 0.0) {
241 return Result<double>::success(std::numeric_limits<double>::infinity());
242 } else {
243 return Result<double>::error(ErrorCode::DivisionByZero,
244 "Cannot calculate Omega ratio with zero gains and losses");
245 }
246 }
247
248 double omega = gains_sum / losses_sum;
249
250 return Result<double>::success(omega);
251}

256inline Result<double> treynor_ratio(const ReturnSeries& portfolio_returns, const ReturnSeries& benchmark_returns,
257 double risk_free_rate = constants::DEFAULT_RISK_FREE_RATE,
258 Frequency frequency = Frequency::Daily) {
259 if (portfolio_returns.size() != benchmark_returns.size()) {
260 return Result<double>::error(ErrorCode::InvalidInput,
261 "Portfolio and benchmark returns must have the same length");
262 }
263
264 if (portfolio_returns.empty()) {
265 return Result<double>::error(ErrorCode::InsufficientData,
266 "Cannot calculate Treynor ratio for empty return series");
267 }
268
269 // Calculate portfolio excess returns
270 auto portfolio_excess_result = calculate_excess_returns(portfolio_returns, risk_free_rate);
271 if (portfolio_excess_result.is_error()) {
272 return Result<double>::error(portfolio_excess_result.error().code, portfolio_excess_result.error().message);
273 }
274
275 // Calculate benchmark excess returns
276 auto benchmark_excess_result = calculate_excess_returns(benchmark_returns, risk_free_rate);
277 if (benchmark_excess_result.is_error()) {
278 return Result<double>::error(benchmark_excess_result.error().code, benchmark_excess_result.error().message);
279 }
280
281 const auto& portfolio_excess = portfolio_excess_result.value();
282 const auto& benchmark_excess = benchmark_excess_result.value();
283
284 // Calculate beta (correlation * (portfolio_vol / benchmark_vol))
285 auto correlation_result = stats::correlation(std::span<const Return>{portfolio_excess.values()},
286 std::span<const Return>{benchmark_excess.values()});
287
288 if (correlation_result.is_error()) {
289 return Result<double>::error(correlation_result.error().code, correlation_result.error().message);
290 }
291
292 auto portfolio_vol_result = calculate_volatility(portfolio_excess, frequency);
293 auto benchmark_vol_result = calculate_volatility(benchmark_excess, frequency);
294
295 if (portfolio_vol_result.is_error() || benchmark_vol_result.is_error()) {
296 return Result<double>::error(ErrorCode::CalculationError, "Failed to calculate volatilities for Treynor ratio");
297 }
298
299 double correlation = correlation_result.value();
300 double portfolio_vol = portfolio_vol_result.value();
301 double benchmark_vol = benchmark_vol_result.value();
302
303 if (benchmark_vol == 0.0) {
304 return Result<double>::error(ErrorCode::DivisionByZero, "Cannot calculate beta with zero benchmark volatility");
305 }
306
307 double beta = correlation * (portfolio_vol / benchmark_vol);
308
309 if (beta == 0.0) {
310 return Result<double>::error(ErrorCode::DivisionByZero, "Cannot calculate Treynor ratio with zero beta");
311 }
312
313 // Calculate mean portfolio excess return
314 auto mean_result = stats::mean(std::span<const Return>{portfolio_excess.values()});
315 if (mean_result.is_error()) {
316 return Result<double>::error(mean_result.error().code, mean_result.error().message);
317 }
318
319 double periods_per_year = frequency::to_annual_factor(frequency);
320 double annualized_excess_return = mean_result.value() * periods_per_year;
321
322 double treynor = annualized_excess_return / beta;
323
324 return Result<double>::success(treynor);
325}

330inline Result<ReturnSeries> rolling_sharpe_ratio(const ReturnSeries& returns, size_t window_size,
331 double risk_free_rate = constants::DEFAULT_RISK_FREE_RATE,
332 Frequency frequency = Frequency::Daily) {
333 if (window_size == 0 || window_size > returns.size()) {
334 return Result<ReturnSeries>::error(ErrorCode::InvalidInput, "Invalid window size for rolling Sharpe ratio");
335 }
336
337 std::vector<Return> rolling_sharpe_values;
338 std::vector<DateTime> rolling_timestamps;
339
340 rolling_sharpe_values.reserve(returns.size() - window_size + 1);
341 rolling_timestamps.reserve(returns.size() - window_size + 1);
342
343 for (size_t i = window_size - 1; i < returns.size(); ++i) {
344 // Extract window
345 std::vector<DateTime> window_timestamps;
346 std::vector<Return> window_returns;
347
348 for (size_t j = i - window_size + 1; j <= i; ++j) {
349 window_timestamps.push_back(returns.timestamp(j));
350 window_returns.push_back(returns[j]);
351 }
352
353 ReturnSeries window_series{std::move(window_timestamps), std::move(window_returns)};
354
355 auto sharpe_result = sharpe_ratio(window_series, risk_free_rate, frequency);
356 if (sharpe_result.is_error()) {
357 // Use NaN for invalid calculations
358 rolling_sharpe_values.push_back(std::numeric_limits<double>::quiet_NaN());
359 } else {
360 rolling_sharpe_values.push_back(sharpe_result.value());
361 }
362
363 rolling_timestamps.push_back(returns.timestamp(i));
364 }
365
366 return Result<ReturnSeries>::success(ReturnSeries{std::move(rolling_timestamps), std::move(rolling_sharpe_values),
367 returns.name() + "_rolling_sharpe"});
368}
